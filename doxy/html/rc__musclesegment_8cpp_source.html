42 int32 rcMuscleSegmenter::getEMI(const vector<rcWindow>& movie,
43  int32 startFrame, int32 endFrame)
44 {
45  /* First, generate the energy curve for the sample period. This will
46  * be used by calcExtendedMuscleCellFrame() to calculate the
47  * location of a frame with the muscle cell fully expanded.
48  */
49  deque<double> signal;
50 
51  const uint32 windowSz = 2, cacheSz = 0;
52 
53  rcSimilarator energy(rcSimilarator::eExhaustive,
54  rcPixel8, windowSz, cacheSz);
55 
56  vector<rcWindow> eMovie;
57 
58  int32 fi;
59  int32 lastWinFrame = startFrame + (int32)windowSz;
60  for (fi = startFrame; fi < lastWinFrame; fi++)
61  eMovie.push_back(movie[fi]);
62 
63  bool rval = energy.fill(eMovie);
64  rmAssert(rval);
65 
66  const int32 targetValue = windowSz/2;
67  deque<double> tempSig;
68  energy.entropies(tempSig, rcSimilarator::eVisualEntropy);
69  signal.push_back(tempSig[targetValue]);
70 
71  for ( ; fi < endFrame; fi++) {
72  rval = energy.update(movie[fi]);
73  rmAssert(rval);
74 
75  energy.entropies(tempSig, rcSimilarator::eVisualEntropy);
76  signal.push_back(tempSig[targetValue]);
77  }
78 
79  return calcExtendedMuscleCellFrame(signal, movie, startFrame);

128 void rcMuscleSegmenter::mergePolygons(vector<rcPolygon>& poly,
129  vector<rcPolygon>& polyCH,
130  vector<rcPolygon>& mergedCH,
131  vector<rcIRect>& mergedCHR,
132  vector<polyInterInfo>& mergedInfo) const
133 {
134  rmAssert(poly.size() == polyCH.size());
135 
136  mergedCH.clear();
137  mergedCHR.clear();
138  mergedInfo.clear();
139 
140  //@note less<> is the default sorting
141  // merging start with largest. Hence the reverse iterators
142 
143  multimap<double, uint32> areaToIndex;
144  for (uint32 si = 0; si < poly.size(); si++) {
145  double area = poly[si].area();
146  areaToIndex.insert(pair<double, uint32>(area,si));
147  }
148 
149  vector<uint32> mergedToPoly(poly.size());
150 
151  for (multimap<double, uint32>::reverse_iterator si = areaToIndex.rbegin();
152  si != areaToIndex.rend(); si++) {
153  rcPolygon& pCH = polyCH[si->second];
154  rcIRect boxCH = pCH.orthogonalEnclosingRect();
155  double sAreaCH = -1.0;
156  uint32 mi;
157  for (mi = 0; mi < mergedCH.size(); mi++) {
158  if (mergedCHR[mi].overlaps(boxCH, true)) {
159  if (sAreaCH < 0.0)
160  sAreaCH = pCH.area()*0.5;
161  uint32 mpi = mergedToPoly[mi];
162  rcPolygon intPoly;
163  polyCH[mpi].convexIntersection(pCH, intPoly);
164  if (intPoly.area() > sAreaCH) {
165  mergedInfo[mi].area += si->first;
166  mergedInfo[mi].polyInter.push_back(poly[si->second]);
167 
168  if (intPoly != pCH) {
169  rcPolygon unionPoly;
170  mergedCH[mi].convexUnion(pCH, unionPoly);
171  mergedCH[mi] = unionPoly;
172  mergedCHR[mi] = mergedCH[mi].orthogonalEnclosingRect();
173  }
174  break;
175  }
176  }
177  }
178 
179  if (mi == mergedCH.size()) {
180  mergedCH.push_back(pCH);
181  mergedCHR.push_back(boxCH);
182  mergedInfo.push_back(polyInterInfo());
183  mergedInfo.back().area = si->first;
184  mergedInfo.back().polyInter.push_back(poly[si->second]);
185  mergedToPoly[mergedCH.size()-1] = si->second;
186  }
187  }
188 
189  if (MSDBG1) printf("merged count %d\n", (int)mergedCH.size());
190 }

306 rcMuscleSegmenter::genSpatialSegmentation(const vector<rcWindow>& movie,
307  int32 startFrame, int32 endFrame,
308  int32& emi, rcWindow& segSdImg)
309 {
310  rmAssert(startFrame >= 0);
311  rmAssert(endFrame < (int32)movie.size());
312  rmAssert((endFrame - startFrame) >= 2);
313 
314  /* Use the following algorithm to generate a segmented image:
315  *
316  * 1) Calculate the index of a frame that has fully extended
317  * muscle cells.
318  *
319  * 2) Generate the std dev image for the emi frame. Calculate a
320  * segmentation point and generate a segmented version of of the
321  * std dev image based on this. The foreground pixels in the
322  * segmented std dev image serve as the candidate muscle cell
323  * areas.
324  */
325 
326  /* Step 1
327  */
328  emi = getEMI(movie, startFrame, endFrame);
329  rmAssert(emi >= startFrame);
330  rmAssert(emi < endFrame);
331 
332  /* Step 2
333  */
334  const int32 radius = 1;
335  const int32 szOffset = radius+1;
336  const int32 szAdjust = 2*szOffset;
337  const int32 segWidth = movie[0].width() - szAdjust;
338  const int32 segHeight = movie[0].height() - szAdjust;
339  rcWindow sdImg(segWidth, segHeight, rcPixel32);
340 
341  genSpatialSDImg(movie[emi], sdImg, radius);
342 
343  const float minBinVal = 3.0, binSz = 0.2;
344  const int32 binCnt = 40;
345  const bool getPeak = true;
346  const float segSdVal = segPoint(sdImg, minBinVal, binSz, binCnt, getPeak);
347 
348  const uint8 bg = 0, fg = 0xFF;
349  rcWindow segSdImgBase(movie[0].width(), movie[0].height());
350  segSdImgBase.setAllPixels(bg);
351  segSdImg = rcWindow(segSdImgBase, 1, 1, segWidth + 2, segHeight + 2);
352  rcWindow segSdWin(segSdImgBase, szAdjust, szAdjust, segWidth, segHeight);
353  binarizeFloatWindow(sdImg, segSdWin, segSdVal, fg, bg);
354  //@todo rfImageConvertFloat8 (sdImg, segSdWin, segSdVal - 1e-10, segSdVal + 1e-10);
355 }

359  int32 startFrame,

377  * frequency image serve as the candidate active muscle cell
378  * areas.
379  */

421  rcWindow segFreqWin(segFreqImg, 1, 1, segWidth, segHeight);

847 rcMuscleSegmenter::rejectRange(int32& startIndex, int32& endIndex,
848  rejectionCode reason, int32 instance) const
849 {
850  rmAssert(reason < cRejCodeCount);
851  const int32 rejSz = (int32)_rejectReason.size();
852  rmAssert(rejSz == (int32)_rejectIndex.size());
853 
854  startIndex = endIndex = 0;
855 
856  if (instance < 1)
857  return;
858 
859  int32 ri = -1;
860 
861  while (instance--) {
862  ri++;

864  if (_rejectReason[ri] == reason)
865  break;
866  }
867  }
868 
869  if (ri != rejSz) {
870  if (ri == 0)
871  startIndex = 0;
872  else

874  endIndex = _rejectIndex[ri];
875  }
