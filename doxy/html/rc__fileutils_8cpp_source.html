237 static int numericsort(const void *A, const void *B, int cs) {
238  struct dirent *dir1 = *(struct dirent **)A;
239  struct dirent *dir2 = *(struct dirent **)B;
240 
241  const char* a = dir1->d_name;
242  const char* b = dir2->d_name;
243  int ret = 0;
244  for (;;) {
245  if (isdigit(*a & 255) && isdigit(*b & 255)) {
246  int diff,magdiff;
247  while (*a == '0') a++;
248  while (*b == '0') b++;
249  while (isdigit(*a & 255) && *a == *b) {a++; b++;}
250  diff = (isdigit(*a & 255) && isdigit(*b & 255)) ? *a - *b : 0;
251  magdiff = 0;
252  while (isdigit(*a & 255)) {magdiff++; a++;}
253  while (isdigit(*b & 255)) {magdiff--; b++;}
254  if (magdiff) {ret = magdiff; break;} /* compare # of significant digits*/
255  if (diff) {ret = diff; break;} /* compare first non-zero digit */
256  } else {
257  if (cs) {
258  /* compare case-sensitive */
259  if ((ret = *a-*b)) break;
260  } else {
261  /* compare case-insensitve */
262  if ((ret = tolower(*a & 255)-tolower(*b & 255))) break;
263  }
264 
265  if (!*a) break;
266  a++; b++;
267  }
268  }
269  if (!ret) return 0;
270  else return (ret < 0) ? -1 : 1;
271 }

290 int rfFilename_match(const char *s, const char *p)
291 {
292  int matched;
293 
294  for (;;) {
295  switch(*p++) {
296 
297  case '?' : // match any single character
298  if (!*s++) return 0;
299  break;
300 
301  case '*' : // match 0-n of any characters
302  if (!*p) return 1; // do trailing * quickly
303  while (!rfFilename_match(s, p)) if (!*s++) return 0;
304  return 1;
305 
306  case '[': { // match one character in set of form [abc-d] or [^a-b]
307  if (!*s) return 0;
308  int reverse = (*p=='^' || *p=='!'); if (reverse) p++;
309  matched = 0;
310  char last = 0;
311  while (*p) {
312  if (*p=='-' && last) {
313  if (*s <= *++p && *s >= last ) matched = 1;
314  last = 0;
315  } else {
316  if (*s == *p) matched = 1;
317  }
318  last = *p++;
319  if (*p==']') break;
320  }
321  if (matched == reverse) return 0;
322  s++; p++;}
323  break;
324 
325  case '{' : // {pattern1|pattern2|pattern3}
326  NEXTCASE:
327  if (rfFilename_match(s,p)) return 1;
328  for (matched = 0;;) {
329  switch (*p++) {
330  case '\\': if (*p) p++; break;
331  case '{': matched++; break;
332  case '}': if (!matched--) return 0; break;
333  case '|': case ',': if (matched==0) goto NEXTCASE;
334  case 0: return 0;
335  }
336  }
337  case '|': // skip rest of |pattern|pattern} when called recursively
338  case ',':
339  for (matched = 0; *p && matched >= 0;) {
340  switch (*p++) {
341  case '\\': if (*p) p++; break;
342  case '{': matched++; break;
343  case '}': matched--; break;
344  }
345  }
346  break;
347  case '}':
348  break;
349 
350  case 0: // end of pattern
351  return !*s;
352 
353  case '\\': // quote next character
354  if (*p) p++;
355  default:
356  if (tolower(*s) != tolower(*(p-1))) return 0;
357  s++;
358  break;
359  }
360  }
361 }
