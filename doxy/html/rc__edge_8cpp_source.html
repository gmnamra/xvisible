192  {
193  y = -y;
194  if (x >= 0)

196  if (y <= x) return - ThetaTable[(y << precision) / x];
197  else return ( m3PIover2 + ThetaTable[(x << precision) / y]);
198  }
199  else
200  {
201  x = -x;
202  if (y <= x) return ( mPI + ThetaTable[(y << precision) / x]);
203  else return ( m3PIover2 - ThetaTable[(x << precision) / y]);
204  }
205  }
