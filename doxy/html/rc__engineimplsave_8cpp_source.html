677 /* wait for capture thread to decide whether or not to restart */
678 /* the save state machine. */

681 rcMovieFileState rcEngineMovieSaver::doneHdlr(rcMovieFileState)
682 {
683  /* Regardless of the outcome of the movie save, the engine should be
684  * stopped.
685  */
686  if (_engine->getState() == eEngineRunning) {
687  _engine->stop();
688  }
689 
690  rmAssert(_engine->getState() != eEngineRunning);
691 
692  uint32 curResetValue;
693 
694  /* Note: This would best be done with a condition variable instead
695  * of a polling loop.
696  */
697  while (_curResetValue.getValue(curResetValue) == _prevResetValue) {
698  if (seppukuRequested()) {
699  _saveStatus.status(eMovieFileErr_Aborted);
700  return eMovieFileState_Exit;
701  }
702 
703  usleep(100000); // Sleep for 100ms
704  }
705 
706  _prevResetValue = curResetValue;
707  _saveStatus.delayLeft(0);
708  _saveStatus.captureTime(0);
709  _saveStatus.status(eMovieFileErr_NoErr);
710  _hist.reset();
711  _histOverflow = 0;
712 
713  return eMovieFileState_Wait;
714 }

721 void rcEngineMovieSaver::run()
722 {
723  rcMovieFileState curState = _saveStatus.getState();
724  rcMovieFileState prevState = curState;
725 
726  for ( ; curState != eMovieFileState_Exit; ) {
727  rcMovieFileState newState = eMovieFileState_Uninitialized; // Quiet compiler
728 
729  switch (curState) {
730  case eMovieFileState_Uninitialized:
731  newState = uninitializedHdlr(prevState);
732  break;
733 
734  case eMovieFileState_Wait:
735  newState = waitHdlr(prevState);
736  break;
737 
738  case eMovieFileState_Delay:
739  newState = delayHdlr(prevState);
740  break;
741 
742  case eMovieFileState_Open:
743  newState = openHdlr(prevState);
744  break;
745 
746  case eMovieFileState_Save:
747  newState = saveHdlr(prevState);
748  break;
749 
750  case eMovieFileState_Close:
751  newState = closeHdlr(prevState);
752  break;
753 
754  case eMovieFileState_Error:
755  newState = errorHdlr(prevState);
756  break;
757 
758  case eMovieFileState_Done:
759  newState = doneHdlr(prevState);
760  break;
761 
762  case eMovieFileState_Exit:
763  rmAssert(0);
764  break;
765  } // End of: switch (curState)
766 
767  rmAssert(newState != eMovieFileState_Uninitialized);
768  prevState = curState;
769  curState = newState;
770  _saveStatus.state(curState);
771  } // for ( ; curState != eMovieFileState_Done; )
772 
773  return;
774 }
