11 /*
12  * Generic Kalman filter implementation
13  *
14  * The Kalman filter is the optimal state estimator
15  * for the following state (x) evolution and observation (z) equations
16  *
17  * x_1 = Ax_0 + q, q ~ N(r; 0, Q)
18  * z = Hx + r, r ~ N(r; 0, R)
19  *
20  * i.e. the state dynamics is linear with gaussian distributed noise (with
21  * covariance Q); the observation is also linear (with noise covariance R).
22  *
23  * The state is preserved as mean x and covariance P.
24  * The state is update as follows:
25  *
26  * First dynamics update (using only previous state and dynamics)
27  * x_1^ = Ax_0
28  * P_1^ = A * P_0 * A' + Q
29  *
30  * Then observation update (incorporate dynamics)
31  * K = P_1^ * H' / (H * P_1^ * H' + R);
32  * x_1 = x_1^ + K * (z_1 - H * x_1^)
33  * P_1 = (eye(size(K, 1)) - K * H) * P_1^;
34  *
35  * The effect is that if P_0 is small, then the gain matrix K would also be
36  * small, and so the state would depend more of dynamics than on the
37  * observation.
38  */

196 template<int DX, int DZ>
197 inline void rcKalmanFilter<DX, DZ>::update(const rcVector<double, DZ>& z,
198  const rc_Matrix<double, DZ, DZ>& zR)
199 {
200  rcVector<double, DX>
201  x1_;
202  rc_Matrix<double, DX, DX>
203  P1_;
204 
205  x1_ = _AA * _x;
206  P1_ = _AA * _PP * _AT + _QQ;
207 
208  _K = (P1_ * _HT) * rfMInvert(_H * P1_ * _HT + zR);
209  _x = x1_ + _K * (z - _H * x1_);
210  _PP = (_II - _K * _H) * P1_;
211 }
