27 // enum to define the basic behaviour of an argument
28 // - a switch is an option with no value but which can be switched on or off e.g. -help and -nohelp
29 // - a value is an option followed by a value e.g. -output results.txt
30 // (a default value can be removed by using the option as a negated switch e.g. -nooutput)
31 // - command-line values (i.e. any strings not preceded by '-') are treated internally as an option with no name and must be values
32 enum cli_kind_t {cli_switch_kind, cli_value_kind};
33 
34 // the mode controls the behaviour if an option appears more than once in either the command-line or the ini files
35 // - a single mode option overrides all previous values so will only be found once in the parsed result
36 // - a multiple mode option can be repeated to define multiple values, but overrides values from ini files
37 // - a cumulative mode option is a multiple mode option which keeps ini file values as well
38 enum cli_mode_t {cli_single_mode, cli_multiple_mode, cli_cumulative_mode};
