279 
280  void add (const rcFeature& feature)
281  {
282  mNstarted++;
283  mNtracked++;
284  const rc2Dvector& pos = feature.pos();
285 
286  // If the motion vector indicates no motion, the current decision is to add them
287  // but not add to the statistics (TBD)
288  if (pos.x() < 0.0001 && pos.y() < 0.0001)
289  return;
290 
291  double angle = pos.angle().Double();
292  mSumPos += pos;
293  mSumAngle += angle;
294  mSumSqAngle += angle * angle;
295  mSumAngle += feature.quality();
296  mSumSqQuality += feature.quality() * feature.quality();
297 
298  if (mNtracked)
299  {
300  double bsd = angle - mLastAngle;
301  mSumAngledT += bsd;
302  mSumSqAngledT += bsd*bsd;
303  mLastAngle = angle;
304  }
305 
306  // Finally if mode is not zero, add this feature to the features list
307  // Note: a copy will made
308  if (mode() == eKeepFeatures) features.push_front (feature);
309  state (eTracking);
310  mLastFeature = feature;
311 
312  }
313 
314  void add ()
315  {
316  mNstarted++;
317  mNtracked = 0;
318  state (eNoFeature);
319  }
