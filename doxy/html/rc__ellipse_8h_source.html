61  rcEllipseNormUnit unit() const
62  {
63  if (degenerate())
64  return rcEllipseNormUnit (mCenter, rcMatrix_2d());
65 
66  mOrient.norm();
67 
68  double sn = sin (mOrient) / mRadii.x();
69  double cs = cos (mOrient) / mRadii.x();
70  double tg = tan (rcRadian (0.0));
71 
72  rcMatrix_2d mm (cs + sn * tg,
73  sn - cs * tg,
74  -sn / (mRadii.y() / mRadii.x()), // y / x is aspect ratio
75  cs / (mRadii.y() / mRadii.x()));
76 
77  return rcEllipseNormUnit (mCenter, mm );
78  }

88  rcDRect boundingOrthRectangle () const
89  {
90  double c, s, x2, y2;
91  c = rmSquare (cos (mOrient));
92  s = rmSquare (sin (mOrient));
93  x2 = rmSquare (mRadii.x());
94  y2 = rmSquare (mRadii.y());
95  rc2Dvector r (sqrt (y2 * s + x2 * c), sqrt (x2 * s + y2 * c));
96  rc2Dvector tl = center() - r;
97  rc2Dvector size = r * 2.;
98  return rcDRect (tl.x(), tl.y(), size.x(), size.y());
