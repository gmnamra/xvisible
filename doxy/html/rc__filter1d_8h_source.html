49 template<class P>
50 void rf1DPeakDetect(const vector<P>& signal, vector<uint32>& peakBin,
51  vector<P>& peakLoc, vector<P>& peakVal,
52  const P hysteresisPct, bool DEBUG_PD = false)
53 {
54  peakBin.clear();
55  peakLoc.clear();
56  peakVal.clear();
57 
58  if (signal.empty())
59  return;
60 
61  /* Calculate peaks using hysteresisPct to threshold for true valleys.
62  */
63  int32 pPeak = 0, pValley = 0, min = 0, max = 0;
64  P peakThresh = signal[pPeak]*(1 - hysteresisPct);
65  P valleyThresh = signal[pValley]*(1 + hysteresisPct);
66 
67  if (DEBUG_PD) printf("signal[%d] = %f peakThresh %f valleyThresh %f\n",
68  0, signal[0], peakThresh, valleyThresh);
69 
70  for (int32 i = 1; i < (int32)signal.size(); i++) {
71  if (DEBUG_PD) printf("signal[%d] = %f", i, signal[i]);
72 
73  if ((pPeak >= 0) && signal[i] < peakThresh) {
74  // Found peak
75  peakBin.push_back((uint32)pPeak);
76  P peakValue;
77  peakLoc.push_back(rfInterpolatePeak(signal, (uint32)pPeak, &peakValue));
78  peakVal.push_back(peakValue);
79  if (DEBUG_PD) printf(" Peak (%f) @ %d.", signal[pPeak], pPeak);
80  pPeak = -1;
81  pValley = i;
82  min = max = i;
83  valleyThresh = signal[pValley]*(1 + hysteresisPct);
84  if (DEBUG_PD)
85  printf(" New pPeak %d pValley %d min %d max %d valleyThresh %f.\n",
86  pPeak, pValley, min, max, valleyThresh);
87  }
88  else if (signal[i] < signal[min]) {
89  if (pValley >= 0) {
90  pValley = i;
91  valleyThresh = signal[pValley]*(1 + hysteresisPct);
92  if (DEBUG_PD) printf(" New pValley %d valleyThresh %f.",
93  pValley, valleyThresh);
94  }
95  min = i;
96  if (DEBUG_PD) printf(" New min %d.\n", min);
97  }
98 
99  if ((pValley >= 0) && signal[i] > valleyThresh) {
100  // Found valley
101  if (DEBUG_PD) printf(" Valley (%f) @ %d.", signal[pValley], pValley);
102  pPeak = i;
103  pValley = -1;
104  min = max = i;
105  peakThresh = signal[pPeak]*(1 - hysteresisPct);
106  if (DEBUG_PD)
107  printf(" New pPeak %d pValley %d min %d max %d peakThresh %f.\n",
108  pPeak, pValley, min, max, peakThresh);
109  }
110  else if (signal[i] > signal[max]) {
111  if (pPeak >= 0) {
112  pPeak = i;
113  peakThresh = signal[pPeak]*(1 - hysteresisPct);
114  if (DEBUG_PD) printf(" New pPeak %d peakThresh %f.",
115  pPeak, peakThresh);
116  }
117  max = i;
118  if (DEBUG_PD) printf(" New max %d.\n", max);
119  }
120 
121  } // End of: for (int32 i = 1; i < signal.size(); i++) {
122 
123  if (pPeak >= 0) {
124  if (DEBUG_PD) printf("Last Peak (%f) @ %d.\n", signal[pPeak], pPeak);
125  peakBin.push_back((uint32)pPeak);
126  P peakValue;
127  peakLoc.push_back(rfInterpolatePeak(signal, (uint32)pPeak, &peakValue));
128  peakVal.push_back(peakValue);
129  }
130 }

224

226  int32 size = dst.size();
227  const S* front1ptr = &src[0];
228  const S* back1ptr = front1ptr;
229  const S* front2ptr = front1ptr;
230  const S* back2ptr = front1ptr;

233  D dummy(0);
234

236  int32 leftSize_ (halfWidth);
237

239  {
240  if (rightSize_ == 1 && leftSize_ == 1)
241  {
242  if (direction)
243  {
244  for (i=0; i<size; i++)
245  outptr[i] = rfRound((T)back1ptr[i+1] - (T)back1ptr[i], dummy);
246  } else
247  {
248  for (i=0; i<size; i++)
249  outptr[i] = rfRound((T)back1ptr[i] - (T)back1ptr[i+1], dummy);
250  }
251  } else
252  {
253  tmp1 = 0;
254  if (direction) for (i = 0; i < leftSize_; i++) tmp1 = tmp1 - (T) *front2ptr++;
255  else for (i = 0; i < leftSize_; i++) tmp1 = tmp1 + (T) *front2ptr++;
256 
257  front1ptr += leftSize_ + 1;
258  back1ptr = front1ptr;
259 
260  if (direction) for (i = 0; i < rightSize_; i++) tmp1 = tmp1 + (T) *front1ptr++;
261  else for (i = 0; i < rightSize_; i++) tmp1 = tmp1 - (T) *front1ptr++;
262 
263  *outptr++ = rfRound(tmp1, dummy);
264  --size;
265 
266  if (direction)
267  {

269  {
270  tmp1 = tmp1 + (T) front1ptr[i] - (T) back1ptr[i] - (T) front2ptr[i] + (T) back2ptr[i];

272  }
