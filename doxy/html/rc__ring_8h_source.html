13 /* rcRingBuffer - Thread-safe implementation of ring buffer. Thread-safety
14  * is based upon the use of atomicValue type integers for
15  * head and tail indices. A ring of pointers to T is
16  * created. The following restrictions apply:
17  *
18  * 1) There should be exactly one consumer thread (thread calling
19  * get()) and exactly one producer thread (thread calling put()).
20  *
21  * The interface consists of the following public functions:
22  *
23  * rcRingBuffer<T>(uint32 sz) - Construct empty ring buffer with sz free slots
24  *
25  * T* getValue() - Get next value from ring buffer
26  *
27  * bool putValue(T*) - Store next value in ring buffer
28  *
29  * size_t size() - Number of slots in ring
30  *
31  * uint32 availValues() - Number of values available to be read
32  *
33  * uint32 availFreeSlots() - Number of free slots available to put values in
34  */
35 template <class T>
36 class rcRingBuffer
37 {
38 public:

40  rcRingBuffer(uint32 sz) : _sz(sz+1), _specAlloc(0), _head(0), _tail(0)
41  {
42  /* Note: It was necessary to allocate an extra element in the
43  * ring, since the code assumes (head == tail) means the ring is
44  * empty and we don't want to maintain a counter to distinguish
45  * between the two cases.
46  */
47  assert(_sz != 0); // Paranoia reigns supreme
48  _ring = (T** volatile)malloc(_sz*sizeof(T*));
49  assert(_ring != 0);
50  }

55  * ring, since the code assumes (head == tail) means the ring is
56  * empty and we don't want to maintain a counter to distinguish
57  * between the two cases.
58  */

76  T* getValue(void)
77  {
78  uint32 readIndex, temp;
79 
80  if (_head.getValue(readIndex) == _tail.getValue(temp))
81  return 0;
82 
83  // Read value from ring
84  //
85  T* retVal = _ring[readIndex];
86 
87  // Then update head pointer
88  //

98  {
99  uint32 writeIndex, headIndex, newIndex;
100 
101  _head.getValue(headIndex);
102  newIndex = _tail.getValue(writeIndex) + 1;
103 
104  if (newIndex == _sz)
105  newIndex = 0;
106 
107  if (newIndex == headIndex)
108  return false;
109 
110  // Write value to ring
111  //
112  _ring[writeIndex] = tp;
113 
114  // Then update tail pointer
115  //
116  _tail.setValue(newIndex);
117 
118  return true;

165 /* rcBidirectionalRing - A pair of ring buffers that allows threads to share
166  * recylcable resources (like frame buffers) using the
167  * following model:
168  *
169  * 1) "producer thread" adds resource to ring.
170  *
171  * 2) "consumer thread" takes the resource from the
172  * ring and uses it during its processing.
173  *
174  * 3) "consumer thread" releases resource back to ring.
175  *
176  * 4) "producer thread" recovers the resource for later
177  * reuse.
178  *
179  * This is implemented by using a pair of
180  * rcRingBuffer class template objects. All
181  * operations are thread-safe. A bidirectional
182  * ring of pointers to T is created. The
183  * following restrictions apply:
184  *
185  * 1) For each of the 4 functions: giveResource(), takeResource(),
186  * releaseResource(), and recoverResource(), one and only one
187  * thread can call that function. That is, for example, if thread A
188  * is calling takeResource(), thread B can never. Thread B could call
189  * releaseResource(), but only if thread A (or for that matter, any
190  * other thread) will never call it.
191  *
192  * The interface consists of the following public functions:
193  *
194  * rcBidirectionalRing<T>(uint32 sz) - Construct empty pair of ring buffers,
195  * each with sz free slots
196  *
197  * bool giveResource(T*) - Add a resource to the consumer's
198  * "available resources ring"
199  *
200  * T* takeResource() - Take the next available resource from
201  * the "available resources ring"
202  *
203  * bool releaseResource(T*) - Add a resource to the producer's
204  * "freed resources ring"
205  *
206  * T* recoverResource() - Take the next freed resource from
207  * the "freed resources ring"
208  *
209  * size_t size() - Number of slots in ring
210  *
211  * uint32 availNewResources() - Number of resources available in the
212  * "available resources ring"
213  *
214  * uint32 availNewSlots() - Number of free slots available in the
215  * "available resources ring" for new
216  * resources
217  *
218  * uint32 availReleasedResources() - Number of resources available in the
219  * "freed resources ring"
220  *
221  * uint32 availReleasedSlots() - Number of free slots available in the
222  * "freed resources ring" for freed
223  * resources
224  *
225  * Note: The implementation of this class is trivial. Does this mean
226  * it shouldn't exist? I like the fact that it provides a template for
227  * how to use rcRingBuffers to share resources among different threads,
228  * so I'll leave it in for now.
229  */
