142 // Perform masked version of correlate.
143 // Only pixel locations that are set in the corresponding masked image are
144 // used in the calculation.
145 // All images must be the same size and depth and pixels in the mask image
146 // must all be either 0 or all ones.
147 void rfCorrelate(const rcWindow& img, const rcWindow& mdl,
148  const rcWindow& mask, const rsCorrParams& params, rcCorr& res,
149  int32 maskN)
150 {
151  rcWindow iMasked(img.width(), img.height(), img.depth());
152  rcWindow mMasked(img.width(), img.height(), img.depth());
153 
154  rfAndImage(img, mask, iMasked);
155  rfAndImage(mdl, mask, mMasked);
156  rfCorrelate(iMasked, mMasked, params, res);
157 
158  if (maskN < 0) {
159  uint32 allOnes = 0xFF;
160  if (img.depth() == rcPixel16)
161  allOnes = 0xFFFF;
162  else if (img.depth() == rcPixel32)
163  allOnes = 0xFFFFFFFF;
164  else
165  rmAssert(img.depth() == rcPixel8);
166 
167  maskN = 0;
168  for (int32 y = 0; y < mask.height(); y++)
169  for (int32 x = 0; x < mask.width(); x++) {
170  uint32 val = mask.getPixel(x, y);
171  if (val == allOnes)
172  maskN++;
173  else
174  rmAssert(val == 0);
175  }
176  }
177  res.n(maskN);
178  res.compute();
179 }

343 static void baligned_products_sumIM ( const unsigned char *input1, const unsigned char *input2, rcCorr& res, unsigned int m16)
344 {
345  vector unsigned char *current, *previous;
346  vector unsigned char t1, t2;
347  vector unsigned int zero, mom01, mom02, sumim;
348  vector unsigned int cross, mom11, mom22;
349  unsigned int result, sii, smm, si, sm;
350 
351 
352  current = (vector unsigned char *) input1;
353  previous = (vector unsigned char *) input2;
354 
355  // Sum IM
356  /* initialization */
357  INITCOMMON;
358  INITMOM1;
359  INITMOM2;
360 
361  // m16 is 16 multiples. Subtrack 1 to match numbering of product loop
362  assert (m16);
363  m16 -= 1;
364 
365  // Prime the pipe
366  t1 = current[0]; /* align current vector */
367 
368  // Process 16 time 16 Pels. Each step consists of cross product as well as 0, and first moments
369  // Results are written in to an int vector which has 4 partial sums in it
370  ALCROSSSUMIM(0);
371  ALCROSSSUMIM(1);
372  ALCROSSSUMIM(2);
373  ALCROSSSUMIM(3);
374  ALCROSSSUMIM(4);
375  ALCROSSSUMIM(5);
376  ALCROSSSUMIM(6);
377  ALCROSSSUMIM(7);
378  ALCROSSSUMIM(8);
379  ALCROSSSUMIM(9);
380  ALCROSSSUMIM(10);
381  ALCROSSSUMIM(11);
382  ALCROSSSUMIM(12);
383  ALCROSSSUMIM(13);
384  ALCROSSSUMIM(14);
385  ALCROSSSUMIM(15);
386 
387 accum:
388  // Sum 4 partial sums in a vector of int in to a single int of a int vector [s,s,s,s] = [-,-,-,S]
389  sumim = (vector unsigned int) (vec_sums((vector signed int) cross, (vector signed int) zero));
390  ACUMMOM1;
391  ACUMMOM2;
392 
393  // replicate the partial sums in to the other ints in an int vector
394  sumim = vec_splat( sumim, 3 );
395  REPLMOM1;
396  REPLMOM2;
397 
398  // Store the result in vector at an address
399  vec_ste( sumim, 0, &result );
400  STRMOM1;
401  STRMOM2;
402 
403  res.accumulate (result,sii,smm,si,sm);
404 
405 }
