701  if ((captureState == eVideoCaptureState_MovieSave) &&
702  (missedFramesCurValue != _acqInfo.missedFrames)) {
703  rmAssert(_acqInfo.missedFrames > missedFramesCurValue);
704  missedFramesOnCapture += _acqInfo.missedFrames - missedFramesCurValue;
705  missedFramesCurValue = _acqInfo.missedFrames;
706  }
707 
708  if (captureState == eVideoCaptureState_Uninitialized) {
709  captureState = eVideoCaptureState_MovieWait;
710  rcRect videoRect(0, 0, _acqInfo.maxFrameWidth,
711  _acqInfo.maxFrameHeight);
712 
713  if (_acqInfo.curResolution == rcAcqHalfResXY)
714  videoRect = rcRect(0, 0, _acqInfo.maxFrameWidth/2,
715  _acqInfo.maxFrameHeight/2);
716 
717  _captureStatus.captureRect(videoRect);
718  _captureStatus.captureDepth(_acqInfo.cameraPixelDepth);
719  _captureStatus.maxFramesPerSecond(_acqInfo.maxFramesPerSecond);
720 
721  _observer->notifyVideoRect(videoRect);
722  _observer->notifyAnalysisRect(videoRect);
723  _engine->setFrameSize(videoRect.width(), videoRect.height(), _acqInfo.cameraPixelDepth);
724  _observer->notifyMultiplier(1000/_acqInfo.maxFramesPerSecond);
725 
726  rcAnalyzerOptions newOpt(corrParams,
727  eAnalyzerFullCorrelation,
728  _captureCtrl.getSlidingWindowOrigin(),
729  _captureStatus.getCaptureRect(),
730  _captureCtrl.getSlidingWindowSize(),
731  _captureCtrl.getEntropyDefinition(),
732  _acqInfo.cameraPixelDepth);
733  if (newOpt != startOpt) {
734  analyzer.setOptions(newOpt);
735  }
736 
737  /* Format string to show on status line. First, make sure string
738  * is null terminated.
739  */
740 
741  std::string cameraName = std::string(_acqInfo.cameraInfo.name());
742 
743  cameraStatus = std::string("Camera: ") + cameraName +
744  std::string(" Action: ");
745  // Write camera info header
746  _engine->setCameraInfo( _acqInfo.cameraInfo );
747  // Start preview track
748  _engine->startPreview();
749  }
750 
751  /* When we first detect that the movie saver thread has
752  * entered the "save" state we want to start sending frames
753  * to the movie saver. Initialize the grabber's frame
754  * capture count and set the "saving movie" state variable
755  * to true.
756  */
757  if ((captureState == eVideoCaptureState_MovieWait) &&
758  (_movieSaver->saveStatus().getState() == eMovieFileState_Save)) {
759  uint32 captureTime = _movieSaver->saveStatus().getCaptureTime();
760 
761  // cMaxSaveFrameCount sets the ultimate upper limit for
762  // number of captured frames
763  if (captureTime) {
764  uint32 frameCaptureCount =
765  (uint32)(((captureTime*_acqInfo.maxFramesPerSecond)/
766  _acqInfo.curDecimationRate) + .4999);
767  if ( frameCaptureCount > cMaxSaveFrameCount-1 )
768  frameCaptureCount = cMaxSaveFrameCount-1;
769  grabber->setFrameCount(frameCaptureCount);
770  } else
771  grabber->setFrameCount(cMaxSaveFrameCount-1);
772 
773  missedFramesCurValue = _acqInfo.missedFrames;
774  missedFramesOnCapture = missedFramesOnSave = 0;
775  captureState = eVideoCaptureState_MovieSave;
776  } // End of: if ((captureState == eVideoCaptureState_MovieWait) && ...)
777 
778  /* See if frame grabber settings need to be changed, but only
779  * if we aren't currently saving a movie.
780  */
781  if ((captureState == eVideoCaptureState_MovieWait) &&
782  !updateCameraSettings(*grabber)) {
783  _acqInfo.acqState = rcAcqDisabled;
784  break;
785  }
786 
787  if ((captureState == eVideoCaptureState_MovieSave) &&
788  (_engine->getState() == eEngineStopped)) {
789  rcTimestamp now = rcTimestamp::now();
790  fprintf(stderr, "Capture stopped at %s\n",
791  now.localtime().c_str());
792  grabber->setFrameCount(-1);
793  closeSaveFile();
794  captureState = eVideoCaptureState_MovieClose;
795  }
796 
797  rcAnalyzerResult result;
798  status = analyzer.getNextResult(result, true);
799  if (status == eFrameStatusOK) {
800  /* If movie save is active handle this activity first
801  */
802  bool frameSaved = false; // Has this frame been saved to disk
803 
804  if ((captureState == eVideoCaptureState_MovieSave) &&
805  result.frameBuf()) {
806  if (_movieSaver->saveStatus().getState() != eMovieFileState_Save) {
807  rcTimestamp now = rcTimestamp::now();
808  fprintf(stderr, "Movie saver aborted at %s\n",
809  now.localtime().c_str());
810  grabber->setFrameCount(-1);
811  closeSaveFile();
812  captureState = eVideoCaptureState_MovieClose;
813  }
814  else if (_saveRing.availReleasedResources() != 0) {
815  rcSharedFrameBufPtr* sbp = _saveRing.recoverResource();
816  rmAssert(sbp);
817  *sbp = result.frameBuf();
818 
819  bool retval = _saveRing.giveResource(sbp);
820  rmAssert(retval == true);
821 
822  _buffersInTransmit.incrementVariable(1, 0);
823  frameSaved = true;
824  }
825  else
826  missedFramesOnSave++;
827  } // End of: if ((captureState==eVideoCaptureState_MovieSave) && ...
828 
829  /* Now give analyzer and GUI a chance to process result
830  */
831  uint32 missedFrames = _acqInfo.missedFrames;
832  if (captureState == eVideoCaptureState_MovieSave)
833  missedFrames = missedFramesOnCapture;
834 
835  _engine->frameInterrupt(result, cameraStatus, missedFrames,
836  missedFramesOnSave, frameSaved);
837  // Set new options if applicable
838  rcAnalyzerOptions newOpt(corrParams,
839  eAnalyzerFullCorrelation,
840  _captureCtrl.getSlidingWindowOrigin(),
841  _captureStatus.getCaptureRect(),
842  _captureCtrl.getSlidingWindowSize(),
843  _captureCtrl.getEntropyDefinition(),
844  _acqInfo.cameraPixelDepth );
845 
846  if (newOpt != analyzer.getOptions() ) {
847  analyzer.setOptions(newOpt);
848  }
849 
850  }
851  else if (status == eFrameStatusEOF) {
852  rcTimestamp now = rcTimestamp::now();
853  fprintf(stderr, "Grabber reached EOF at %s\n",
854  now.localtime().c_str());
855  grabber->setFrameCount(-1);
856  analyzer.reset();
857  closeSaveFile();
858  captureState = eVideoCaptureState_MovieClose;
859  }
860  else {
861  rcTimestamp now = rcTimestamp::now();
862  fprintf(stderr, "Got some other error at %s\n",
863  now.localtime().c_str());
864  grabber->setFrameCount(-1);
865  closeSaveFile();
866  captureState = eVideoCaptureState_MovieClose;
867  rcFrameGrabberError error = analyzer.getLastError();
868  _observer->notifyError(rcFrameGrabber::getErrorString(error).c_str());
869  break; // Bail out of while loop
870  } // End of: if (status == eFrameStatusOK) ...
871  // else if (status == eFrameStatusEOF) ... else ...
872 
873  /* Wait until movie saver says it is done before cleaning up shared
874  * ring.
875  */
876  if ((captureState == eVideoCaptureState_MovieClose) &&
877  (_movieSaver->saveStatus().getState() == eMovieFileState_Done)) {
878  saveComplete();
879  captureState = eVideoCaptureState_MovieWait;
880  }
881  } //End of: while (!seppukuRequested()&&(_acqInfo.acqState!=rcAcqDisabled))
882 
883  /* Before returning make sure that any pending movie save has a
884  * chance to complete.
885  */
886  if ((captureState == eVideoCaptureState_MovieSave)) {
887  closeSaveFile();
888  captureState = eVideoCaptureState_MovieClose;
889  }
890 
891  if (captureState == eVideoCaptureState_MovieClose) {
892  while (_movieSaver->saveStatus().getState() != eMovieFileState_Done)
893  usleep(10000); // sleep 10ms
894  saveComplete();
895  captureState = eVideoCaptureState_MovieWait;
896  }
897 
898  killSaveThread();
899  } else { // Error initing camera
900  rmAssert(errorString);
901 
902  std::string eMsg = "";
903 
904  if (!allocError)
905  std::string eMsg = rcFrameGrabber::getErrorString(grabber->getLastError());
906 
907  const size_t blen = 256;
908  char buf[blen];
909  snprintf(buf, blen, errorString, eMsg.c_str());
910  eMsg = std::string(buf);
911 
912  fprintf(stderr, "%s", eMsg.c_str());
913  _engine->start(); _engine->stop();
914  if (allocError)
915  _observer->notifyStatus("Cannot Initialize Camera");
916  else
917  _observer->notifyStatus("No Camera");
918  _observer->notifyError(eMsg.c_str());
919  _acqInfo.acqState = rcAcqDisabled;
920  } // End of: if (initializeCamera(*grabber)) ... else // Error initing camera
921  } else { // Grabber creation failure
922  _acqInfo.acqState = rcAcqDisabled;
923  }
924 
925  rmAssert(!_saveThread);
926  rmAssert(!_movieSaver);
927 
928  if (_acqInfo.acqState != rcAcqDisabled)
929  sendShutdown(*grabber);
930 
931  delete grabber;
932 
933  _acqInfo.acqState = rcAcqUnknown;
934 }
