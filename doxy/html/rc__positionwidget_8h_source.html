38  rcValueHistogram( uint32 minValue, uint32 maxValue ) : mMinValue( minValue ), mMaxValue( maxValue ), mAdded( 0 ), mCount( vector<uint32>(maxValue-minValue+1, 0) ) {};
39  ~rcValueHistogram() {}
40 
41  // Accessors
42  uint32 size() const { return mCount.size(); }
43  uint32 minValue() const { return mMinValue; }
44  uint32 maxValue() const { return mMaxValue; }
45  uint32 added() const { return mAdded; }
46  uint32 maxCount() const {
47  uint32 maxCount = 0;
48 
49  for ( uint32 i = mMinValue; i <= mMaxValue; ++i ) {
50  uint32 c = count( i );
51  if ( c > maxCount )
52  maxCount = c;
53  }
54  return maxCount;
55  }
56 
57  uint32 count( uint32 value ) const {
58  rmAssertDebug( value >= mMinValue );
59  rmAssertDebug( value <= mMaxValue );
60  return mCount[value-mMinValue];
61  }
62 
63  // Mutators
64  void add( uint32 value, uint32 count = 1 ) {
65  rmAssertDebug( value >= mMinValue );
66  rmAssertDebug( value <= mMaxValue );
67  mCount[value-mMinValue] += count;
68  mAdded += count;
69  }
70 
71  // Utilities

73  cout << "Value\tCount" << endl;

75  uint32 c = count( i );
76  if ( allValues || c > 0 )
77  cout << i << "\t" << c << endl;
78  }
79  }
