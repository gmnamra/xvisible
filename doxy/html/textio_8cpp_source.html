260 bool otext::put (int ch)
261 {
262  if (!m_buffer) return false;
263  bool success = true;
264  // its illegal to put a value outside the range of unsigned char
265  if (ch < 0 || ch > 255)
266  {
267  set_error(textio_put_failed);
268  success = false;
269  }
270  else if (ch != newline)
271  {
272  // no special handling for non-newline characters
273  if (m_buffer->put((unsigned char)ch))
274  m_buffer->increment();
275  else
276  {
277  set_error(textio_put_failed);
278  success = false;
279  }
280  }
281  else
282  {
283  switch(m_buffer->newline_mode())
284  {
285  case otext::unix_mode:
286  // convert the abstract newline character to the literal linefeed character
287  if (m_buffer->put('\xa'))
288  m_buffer->increment(true);
289  else
290  {
291  set_error(textio_put_failed);
292  success = false;
293  }
294  // test for line buffer condition only on end-of-line
295  if (m_buffer->line_buffer())
296  flush();
297  break;
298  case otext::msdos_mode:
299  // convert the abstract newline character to the literal carriage-return/linefeed pair
300  if (m_buffer->put('\xd'))
301  m_buffer->increment();
302  else
303  {
304  set_error(textio_put_failed);
305  success = false;
306  }
307  if (m_buffer->put('\xa'))
308  m_buffer->increment(true);
309  else
310  {
311  set_error(textio_put_failed);
312  success = false;
313  }
314  // test for line buffer condition only on end-of-line
315  if (m_buffer->line_buffer())
316  flush();
317  break;
318  case otext::macos_mode:
319  // convert the abstract newline character to the literal carriage-return character
320  if (m_buffer->put('\xd'))
321  m_buffer->increment(true);
322  else
323  {
324  set_error(textio_put_failed);
325  success = false;
326  }
327  // test for line buffer condition only on end-of-line
328  if (m_buffer->line_buffer())
329  flush();
330  break;
331  case otext::binary_mode:
332  // raw data mode so no special handling of newlines
333  if (m_buffer->put((unsigned char)ch))
334  m_buffer->increment(true);
335  else
336  {
337  set_error(textio_put_failed);
338  success = false;
339  }
340  break;
341  }
342  }
343  return success;
344 }

347 {
348  // treat a null string as acceptable - the method does nothing but returns a success flag
349  if (!str) return true;

745 {

747  int ch = m_buffer->peek();
748  if (m_buffer->newline_mode() == itext::convert_mode)

763 int itext::get (void)
764 {
765  if (!m_buffer) return -1;
766  // get a raw character from the input buffer
767  int ch = m_buffer->get();
768  // Note: it used to be an error to get an eof
769  // However, this seems unnecessary - input buffers should be designed to just return -1 repeatedly
770  // It is then legal to get an eof any number of times
771  if (ch == -1) return -1;
772  // optionally do line-end conversion if this cg=haracter is a line-end character
773  if (m_buffer->newline_mode() == itext::convert_mode)
774  {
775  switch(ch)
776  {
777  case '\xd': // carriage-return - optionally followed by linefeed
778  ch = newline;
779  if (m_buffer->peek() == '\xa')
780  {
781  m_buffer->get();
782  m_buffer->increment();
783  }
784  break;
785  case '\xa': // linefeed
786  ch = newline;
787  break;
788  default:
789  break;
790  }
791  }
792  m_buffer->increment(ch == newline);
793  return ch;
794 }

847 static std::string get_token(itext& str)
848 {
849  // Keep number handling simple by just getting a token
850  // let the string-to-numeric conversions detect any errors
851  // TODO - be fussier and allow other delimiters
852  std::string result;
853  int ch = 0;
854  skipwhite(str);
855  // get token
