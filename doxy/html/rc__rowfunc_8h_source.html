405 template <class T>
406 void rcBasicCorrRowFunc<T>::rowFuncNoneCached ()
407 {
408  double Si(0), Sm(0), Sii(0), Smm(0), Sim(0);
409  const T *pFirst (rcRowFuncTwoSource<T>::mFirst);
410  const T *pSecond (rcRowFuncTwoSource<T>::mSecond);
411 
412  for (const T *pEnd = rcRowFuncTwoSource<T>::mFirst + rcRowFuncTwoSource<T>::mWidth; pFirst < pEnd; ++pFirst, ++pSecond)
413  {
414  const T i = *pFirst;
415  const T j = *pSecond;
416 
417  Si += i;
418  Sm += j;
419  Sii += i * i;
420  Smm += j * j;
421  Sim += i * j;
422  }
423  mRes.accumulate (Sim, Sii, Smm, Si, Sm);
424 
425  rcRowFuncTwoSource<T>::mFirst += mRUP.x();
426  rcRowFuncTwoSource<T>::mSecond += mRUP.y();
427 }
