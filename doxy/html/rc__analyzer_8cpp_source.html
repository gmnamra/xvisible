55 rcFrameGrabberStatus rcAnalyzer::getNextResult( rcAnalyzerResult& result, bool isBlocking )
56 {
57  result.setEntropy( cUndefinedEntropy );
58  result.setFrameBuf( 0 );
59  rcRect clipRect;
60  static double zd (0.0);
61  deque <double> results (mOptions.windowSize(), zd);
62  vector <double> vresults (results.size(), zd);
63 
64  // Error checks
65  if ( !isValid( mOptions ) )
66  return eFrameStatusError;
67  if ( mState == eAnalyzerStopped )
68  return eFrameStatusEOF;
69  if ( mState == eAnalyzerError )
70  return eFrameStatusError;
71 
72  if ( mState == eAnalyzerInitialized )
73  start();
74 
75  if ( mState == eAnalyzerCached ) {
76  rmAssert( mFrames.size() == mResults.size() );
77 
78  if ( mFrames.empty() )
79  return eFrameStatusEOF;
80  else
81  return popResult( result );
82  }
83 
84  int32 i = 0;
85  if ( mState == eAnalyzerStarted ) {
86  rcSharedFrameBufPtr framePtr;
87  rcFrameGrabberStatus status = eFrameStatusError;
88 
89  if ( isBlocking ) {
90  for( ;; ) {
91  status = mInputSource.getNextFrame( framePtr, isBlocking );
92 
93  if ( status == eFrameStatusOK ) {
94  mLastError = eFrameErrorOK;
95  // Add new frame to queue
96  mFrames.push_back( framePtr );
97 
98  if ( mOptions.bound().empty() ) {
99  // If bounding rect is empty, analyze whole image
100  // Use the size of the first frame
101  clipRect = rcRect( 0, 0, mFrames.front()->width(), mFrames.front()->height() );
102  }
103  else
104  {
105  clipRect = mOptions.bound();
106 
107  if (mOffsets.size())
108  {
109  static const int32 dummy (0);
110  clipRect.translate (rcIPair (rfRound (mOffsets[i].x(), dummy), rfRound (mOffsets[i].y(), dummy)));
111  }
112  }
113 
114  rcWindow w(framePtr, clipRect );
115  i++;
116 
117  // mImages is only used for initial images to fill the pipe
118  if (!mFilled)
119  mImages.push_back( w );
120 
121  if (mImages.size() < mOptions.windowSize() )
122  {
123  pushUndefinedResult();
124  }
125 
126  else if (!mFilled && mImages.size() == mOptions.windowSize() )
127  {
128  if (mImages.size() > 1)
129  {
130  mEnergyCalc->fill (mImages);
131  mFilled = mEnergyCalc->entropies (results, mOptions.entropyDefinition());
132  rmAssert(mFilled);
133  vresults.assign (results.begin(), results.end());
134  }
135 
136  // Push the appropriate result(s)
137  pushResult( vresults );
138  // Pop a result
139  popResult( result );
140  break;
141  }
142  else
143  {
144  // Queue is full enough for popping
145 
146  rmAssert(mEnergyCalc);
147  mEnergyCalc->update (w);
148  bool gotSignal = mEnergyCalc->entropies (results, mOptions.entropyDefinition());
149  rmAssert(gotSignal);
150  vresults.assign (results.begin(), results.end());
151 
152  // Push the appropriate result(s)
153  pushResult( vresults );
154  // Pop a result
155  popResult( result );
156  break;
157  }
158  } else if ( status == eFrameStatusEOF )
159  {
160  // No more frames are available.
161  // Compute results and cache them. Subsequent calls will
162  // pull results from the cache.
163  if ( !mFrames.empty() )
164  {
165  mState = eAnalyzerCached;
166 
167  // Produce the necessary undefined values
168  while ( mResults.size() < mFrames.size())
169  {
170  mResults.push_back( cUndefinedEntropy );
171  }
172  status = popResult( result );
173  }
174  else
175  {
176  // Never got any frames
177  mLastError = eFrameErrorOK;
178  }
179  break;
180  }
181  else {
182  // getNextFrame() failed, bail out
183  mLastError = mInputSource.getLastError();
184  break;
185  }
186  }
187  } else {
188  // Non-blocking operation not implemented yet
189  mLastError = eFrameErrorNotImplemented;
190  status = eFrameStatusError;
191  }
192 
193  return status;
194  } else {
195  // Frame grabber never started!
196  }
197 
198  return eFrameStatusError;
199 }

284 rcFrameGrabberStatus rcAnalyzer::popResult( rcAnalyzerResult& result )
285 {
286  rmAssert( !mFrames.empty() );
287  rmAssert( !mResults.empty() );
288 
289  rcFrameGrabberStatus status = eFrameStatusError;
290 
291  if ( !mFrames.empty() && !mResults.empty() ) {
292  // Get first available result
293  result.setEntropy( mResults.front() );
294  result.setFrameBuf( mFrames.front() );
295 
296  // Remove front elements
297  mFrames.pop_front();
298  deque<rcSharedFrameBufPtr>(mFrames).swap(mFrames);
299 
300  mResults.pop_front();
301  deque<double>(mResults).swap(mResults);
302  mLastError = eFrameErrorOK;
303  status = eFrameStatusOK;
304  }
305  else {
306  mLastError = eFrameErrorInternal;
307  }
308 
309  return status;

313 void rcAnalyzer::pushResult( const vector<double>& results )
314 {
315  if ( results.empty() ) {
316  mResults.push_back( cUndefinedEntropy );
317  } else {
318  // Choose result based on origin
319  switch ( mOptions.origin() ) {
320  case eAnalyzerResultOriginLeft:
321  mResults.push_back( results.front() );
322  break;
323  case eAnalyzerResultOriginCenter:
324  mResults.push_back( results[ mOptions.windowSize()/2] );
325  break;
326  case eAnalyzerResultOriginRight:
327  mResults.push_back( results.back() );
328  break;
329  default:
330  rmAssert( 0 );
331  break;
332  }
333  }
