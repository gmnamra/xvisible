57 // 1D correlation: Normalized Correlation
58 //
59 template <class Iterator>
60 double rf1DNormalizedCorr (Iterator Ib, Iterator Ie, Iterator Mb, Iterator Me)
61 {
62  double sii(0.0), si(0.0), smm(0.0), sm(0.0), sim (0.0);
63  int32 n;
64  double rsq;
65 
66  n = Ie - Ib;
67  rmAssert (n);
68  rmAssert ((Me - Mb) == n);
69 
70  // Get container value type
71  typedef typename std::iterator_traits<Iterator>::value_type value_type;
72 
73  Iterator ip = Ib;
74  Iterator mp = Mb;
75 
76  while (ip < Ie)
77  {
78  double iv (*ip);
79  double mv (*mp);
80  si += iv;
81  sm += mv;
82  sii += (iv * iv);
83  smm += (mv * mv);
84  sim += (iv * mv);
85 
86  ip++;
87  mp++;
88  }
89 
90  double cross = ((n * sim) - (si * sm));
91  double energyA = ((n * sii) - (si * si));
92  double energyB = ((n * smm) - (sm * sm));
93 
94  energyA = energyA * energyB;
95  rsq = 0.;
96  if (energyA != 0.)
97  rsq = (cross * cross) / energyA;
98 
99  return rsq;
100 }

206  vector<double> space (slide + slide + 1);
207  vector<double>::iterator orgItr = space.begin() + slide;
208  vector<double>::iterator slidItr;
209 
210  // Sliding I to the right
211  slidItr = orgItr;
212  for (uint32 i = 0; i < (slide + 1); i++)
213  {
214  *slidItr-- = rf1DNormalizedCorr (Ib + i, Ie, Mb, Me - i);
215  }
216 
217  // Sliding I to the left (orgin already done)

219  for (uint32 i = 1; i < (slide + 1); i++)

222  }

364 template <class Iterator>
365 double rf1dSignalIntersection (Iterator Ib, Iterator Ie, Iterator Mb, Iterator Me)
366 {
367  double sm(0.0), sim (0.0);
368  int32 n;
369 
370  n = Ie - Ib;
371  rmAssert (n);
372  rmAssert ((Me - Mb) == n);
373 
374  // Get container value type
375  typedef typename std::iterator_traits<Iterator>::value_type value_type;
376 
377  Iterator ip = Ib;
378  Iterator mp = Mb;
379 
380  while (ip < Ie)
381  {
382  value_type iv (*ip);
383  value_type mv (*mp);
384  sm += mv;
385  sim += rmMin (iv, mv);
386  ip++;
387  mp++;
388  }
389 
390  double cross = sim / sm;
391  return cross;
392 }
