128 void rcMainWindow::settingChanged()
129 {
130  rcModelDomain* domain = rcModelDomain::getModelDomain();
131  // Get current experiment attributes
132  rcExperimentAttributes attr = domain->getExperimentAttributes();
133  uint32 maxX = attr.frameWidth;
134  uint32 maxY = attr.frameHeight;
135  int depth = attr.frameDepth * 8; // Bits per pixel
136  std::string title = attr.title;
137  if (title == std::string ("Untitled"))
138  title = attr.inputName;
139  string::size_type pos = title.find (';', 0);
140  if ( pos != string::npos)
141  {
142  //make a string up to the ;
143  title = std::string (title, 0, pos);
144  pos = title.find_last_of ('/', string::npos);
145  if ( pos != string::npos)
146  {
147  title = std::string (title, 0, pos);
148  pos = title.find ('/', 0);
149  if ( pos != string::npos)
150  {
151  title = std::string (title, pos, title.size());
152  }
153  }
154  }
155 
156  strstream caption;
157  // Concatenate experiment title, video frame size and depth
158  caption << title;
159  if ( depth > 0 )
160  caption << " (" << maxX << "x" << maxY << "x" << depth << ")";
161  caption << ends;
162  // Set window bar caption
163  setCaption( caption.str() );
164  caption.freeze( false ); // Without freeze(), strstream::str() leaks the string it returns
165 
166  // Disable resizing for camera
167  if ( attr.liveInput ) {
168  mStatusBar->setSizeGripEnabled( FALSE );
169  } else {
170  mStatusBar->setSizeGripEnabled( TRUE );
171  }
172 }

176 void rcMainWindow::closeEvent( QCloseEvent* event )
177 {
178  rcModelDomain* domain = rcModelDomain::getModelDomain();
179  rcExperimentState state = domain->getExperimentState();
180 
181  if ( state != eExperimentEmpty && state != eExperimentLocked ) {
182  // No data has been saved, prompt user
183  switch( QMessageBox::warning( 0, cAppName,
184  "Do you want to save experiment data before quitting?\n\n"
185  "If you don't save, your data will be lost.",
186  "&Don't Save", "Cancel", "&Save",
187  2, 1 ) ) {
188  case 0: // Don't Save clicked or Alt+D pressed
189  event->accept();
190  break;
191  case 1: // Cancel clicked or Alt+C pressed or Escape pressed
192  // Do nothing
193  break;
194  case 2: // Save clicked or Alt+S pressed
195  domain->requestSave( eExperimentNativeFormat );
196  if ( domain->getExperimentState() == eExperimentLocked )
197  event->accept();
198  break;
199  }
200 
201  } else
202  event->accept();
203 }
