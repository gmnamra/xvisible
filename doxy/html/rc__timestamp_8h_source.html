207 // Class for calculating a sliding average FPS (frames per second) speed
208 //
209 
210 class rcFpsCalculator {
211  public:
212  // Ctor
213  rcFpsCalculator( uint32 size ) : mLastTime( cZeroTime ), mWindowSize( size ), mTotalTime( -1.0 ) {
214  rmAssert( mWindowSize > 0 ); };
215 
216  // Accessor: get current speed
217  double fps() const {
218  if ( mTotalTime > 0.0 ) {
219  if ( !mFrameIntervals.empty() ) {
220  return 1.0 / (mTotalTime / (mFrameIntervals.size()));
221  }
222  }
223 
224  return 0.0;
225  }
226 
227  // Mutator: add new time
228  void addTime( const rcTimestamp& time ) {
229  if ( mTotalTime >= 0.0 ) {
230  // Add interval
231  rcTimestamp newInterval = time - mLastTime;
232  addInterval( newInterval.secs() );
233  } else {
234  // First time ever, just store the time
235  mTotalTime = 0.0;
236  }
237  mLastTime = time;
238  }
239 
240  // Mutator: reset queue
241  void reset() {
242  mLastTime = cZeroTime;
243  mTotalTime = 0.0;
244  mFrameIntervals.clear();
245  };
246 
247  private:
248  // Mutator: add new interval
249  void addInterval( double frameInterval ) {
250  // Check for window overflow
251  if ( mFrameIntervals.size() > mWindowSize ) {
252  // Remove first interval
253  mTotalTime -= mFrameIntervals.front();
254  mFrameIntervals.pop_front();
255  }
256  // Add new interval
257  mFrameIntervals.push_back( frameInterval );
258  mTotalTime += frameInterval;
259  }
260 
261  rcTimestamp mLastTime; // Timestamp of last update
262  uint32 mWindowSize; // Size of sliding window
263  double mTotalTime; // Total of all frame intervals
264  deque<double> mFrameIntervals; // Queue of frame intervals
265 };
