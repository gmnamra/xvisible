106 // Update the setting value with the string just entered in
107 // the QLineEdit widget.
108 void rcRectSettingWidget::valueChanged( void )
109 {
110  if ( _ignoreValueChanged )
111  return;
112 
113  setMaxValues();
114  // set the new value from the state of the spin boxes
115  int x = _xWidget->value();
116  int y = _yWidget->value();
117  int w = _wWidget->value();
118  int h = _hWidget->value();
119  rcRect newRect( x , y , w , h );
120 
121  rcModelDomain* domain = rcModelDomain::getModelDomain();
122  int32 maxX = domain->getExperimentAttributes().frameWidth;
123  int32 maxY = domain->getExperimentAttributes().frameHeight;
124 
125  // Throttle values
126  if ( newRect.x() + newRect.width() > maxX )
127  newRect.setWidth( maxX - newRect.x() );
128  if ( newRect.y() + newRect.height() > maxY )
129  newRect.setHeight( maxY - newRect.y() );
130 
131  mSetting.setValue( newRect );
132  // notify everybody about the change
133  rcModelDomain::getModelDomain()->notifySettingChange();
134  // notify widgets who want to draw this rect
135  rcModelDomain::getModelDomain()->notifyAnalysisRect( newRect );
136 }
