76  rmAssert (areaRatio > 0.0 || areaRatio <= 1.0);

78  rcIRect br = orthogonalEnclosingRect ();
79

81  {
82  rcIRect ar = mpolys->orthogonalEnclosingRect ();

123 void rfPolygonGroupMutualMerge (const vector<rcPolygon>& fixed, const vector<rcPolygon>& moving, vector<rcPolygon>& merged, double discDiameter, float areaRatio)
124 {
125  vector<rcPolygon>::const_iterator polys = fixed.begin();
126  merged.clear ();
127  for (; polys != fixed.end(); polys++)
128  {
129  rcPolygon buffer = polys->discBuffer (discDiameter);
130 
131  vector<rcPolygon > ipolPtrs, uniquePolPtrs;
132 
133  // The intersected list would include me
134  buffer.intersects (moving, ipolPtrs, areaRatio);
135 
136  // Check if the intersect group intersects with any other polygon
137  // @note Combinatorics of this could be improved
138  bool intersects (false);
139  for (uint32 pi = 0; pi < ipolPtrs.size(); pi++)
140  {
141  // Skip me. Since we know that it does and we want to get it out
142  if (ipolPtrs[pi] == *polys) continue;
143 
144  vector<rcPolygon> tmpPtrs;
145  intersects = ipolPtrs[pi].intersects (moving, tmpPtrs, areaRatio);
146  if (intersects)
147  {
148  break;
149  }
150  uniquePolPtrs.push_back (ipolPtrs[pi]);
151  }
152 
153  // Either we will have the original poly or a union of the merged.
154  rcPolygon unionPoly = (uniquePolPtrs.size() > 1) ? uniquePolPtrs[0] : *polys;
155 
156  if (uniquePolPtrs.size() > 1)
157  {
158  rmAssert (!intersects);
159  // There were more than one
160  for (uint32 pi = 1; pi < uniquePolPtrs.size(); pi++)
161  {
162  rcPolygon tmpPoly;
163  unionPoly .convexUnion(uniquePolPtrs[pi], tmpPoly);
164  unionPoly = tmpPoly;
165  }
166  }
167 
168  merged.push_back (unionPoly);
169  }
170 }
