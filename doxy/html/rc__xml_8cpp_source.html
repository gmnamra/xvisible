181 // Output tree to stream
182 ostream& output( ostream& os, const rcXMLElementTree& tree,
183  rcProgressIndicator* progress, double completedAlready )
184 {
185  // Pre-order iterators
186  rcXMLElementTree::iterator sibBegin = tree.begin();
187  rcXMLElementTree::iterator sibEnd = tree.end();
188  const uint32 totalElements = tree.size();
189  uint32 processedElements = 0;
190 
191  // Stack for close tags
192  stack<rcIndentedCloseTag> closeStack;
193 
194  while ( sibBegin != sibEnd ) {
195  const rcXMLElement& e = *sibBegin;
196  const rcXMLElementType type = e.type();
197  const int32 depth = tree.depth( sibBegin );
198 
199  // Report progress
200  if ( progress ) {
201  if ( !(processedElements % 256) ) {
202  progress->progress( completedAlready + (100.0-completedAlready) * processedElements / totalElements );
203  }
204  ++processedElements;
205  }
206  // Pop close tags if necessary
207  while ( !closeStack.empty() ) {
208  const rcIndentedCloseTag& t = closeStack.top();
209  if ( depth <= t.second ) {
210  os << t;
211  closeStack.pop();
212  } else
213  break;
214  }
215  // Open tag with content
216  os << e;
217 
218  // Optimization, produce close tag immediately if there are no children
219  if ( ! tree.number_of_children( sibBegin ) ) {
220  // Close tag
221  os << rcXMLCloseTag( type );
222  }
223  else {
224  closeStack.push( rcIndentedCloseTag( type, depth ) );
225  }

227  }
228

231  closeStack.pop();
232  }
