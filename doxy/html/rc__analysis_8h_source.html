56  inline double compute ()
57  {
58  rmAssertDebug( mSi >= 0.0 );
59  rmAssertDebug( mSm >= 0.0 );
60 
61  const double ei = ((mN * mSii) - (mSi * mSi));
62  const double em = ((mN * mSmm) - (mSm * mSm));
63  const double mEim = ei * em;
64 
65  // Modified Normalized Correlation. If both image and model have zero standard deviation and equal moments then
66  // they are identical. This case produces 0 in regression based correlation because you can not fit a line
67  // through a single point. However, from image matching point of view it is a perfect match.
68 
69  mRp = (mEim == 0.) && (mSmm == mSii) && (mSi == mSm);
70 
71  mCosine = ((mN * mSim) - (mSi * mSm));
72  mCosine = mCosine < 0 ? 0 : mCosine;
73  mR = (mCosine * mCosine ) / mEim;
74  return mR;
75  }
76 
77  inline double cosine ()
78  {
79  return acos (mCosine / std::sqrt (mEim));
80  }

384 /* rf1DFourierAnalysis - Takes the 1D input signal and performs a
385  * fourier analysis of the data, leaving all the results in rslt. The
386  * real and imag vectors in the result structure are both resized to
387  * the size of the input signal. The phase and amplitude vectors only
388  * contain the entries for frequencies that are within the nyqist
389  * limit (approximately half the size of the input signal).
390  *
391  * In addition, the result amplitudes are searched. The location of
392  * the frequency at which the peak amplitude occurred is returned as
393  * the peak frequency. A parabolic fit is done to find the final vaues
394  * for peak frequency and peak amplitude. The peak phase is computed
395  * by computing a linear fit of the two phase angles closest to the
396  * computed peak frequency.
397  *
398  * The bits in the control word works as follows:
399  *
400  * rc1DFourierForceFFT - If set, the input signal will be rounded up
401  * in size to a power of 2. This "rounding" is done by adding zeros
402  * to the end of the signal. If rc1DFourierZeroFill is also set, the
403  * input signal will be rounded up to the nearest power of 2 plus 1,
404  * otherwise it will be rounded up to the nearest power of 2.
405  *
406  * rc1DFourierZeroFill - If set, zero padding is added to the end of
407  * the signal. If rc1DFourierForceFFT is also set, then zero filling
408  * is as described above, otherwise signal size number of zeros are
409  * added.
410  *
411  * rc1DFourierRaisedCosine - If set, a Hanning filter is applied to
412  * the input signal.
413  *
414  * Note: If the modified signal (with any zero filling added) has a
415  * size that is a power of 2, then an FFT is performed, otherwise a
416  * DFT is performed.
417  *
418  * Note: In all cases, the DC component is subtracted from the input
419  * signal before FFT processing is done.
420  *
421  * Note: Choosing to support an input signal that is a deque of
422  * doubles was chosen for compatibility with the rcSimilarity's output
423  * signal.
424  */
425 enum rc1DFourierControl {
426  rc1DFourierForceFFT = 1,
427  rc1DFourierZeroFill = 2,
428  rc1DFourierRaisedCosine = 4,
429  rc1DFourierGenPhase = 8
430 };
