63 // Start tag of element encountered, ie <experiment>
64 bool rcXMLParser::startElement( const QString&,
65  const QString&,
66  const QString& name,
67  const QXmlAttributes& attributes )
68 {
69  if ( mProgress ) {
70  ++mParsedElements;
71  if ( !(mParsedElements % 32) )
72  mProgress->progress( 100.0 * mParsedElements/mExpectedElements );
73  }
74  rcXMLElementType type = mNameMapper->elementType( name.latin1() );
75 
76  if ( type != eXMLElementMax ) {
77  // Control mode
78  setIgnoreMode( type, true );
79 
80  if ( !mIgnoreAll ) {
81  rcAttributeList attrs;
82  const int attrCount = attributes.count();
83 
84  if ( attrCount > 0 ) {
85  attrs.reserve( attrCount );
86  // Produce attributes
87  for ( int i = 0; i < attrCount; ++i ) {
88  const std::string attributeName = attributes.qName( i ).latin1();
89  // Check validity
90  rcXMLAttributeType type = mNameMapper->attributeType( attributeName );
91  if ( type != eXMLAttributeMax ) {
92  const std::string attributeValue( attributes.value( i ).latin1() );
93  attrs.push_back( rcAttribute( type, attributeValue ) );
94  } else {
95  // Unknown attribute
96  cerr << "rcXMLParser error: unknown attribute \"" << attributeName << "\"" << endl;
97  }
98  }
99  }
100  // Create element
101  rcXMLElement e( type, attrs );
102  // Populate tree
103  if ( mEmptyTree ) {
104  // Add first element as a special case
105  mElementLatest = mElementTree.insert( mElementLatest, e );
106  if ( mElementTree.size() > 0 )
107  mEmptyTree = false;
108  } else {
109  mElementLatest = mElementTree.append_child( mElementLatest, e );
110  }
111  }
112  }
113  else {
114  // Error, unknown element
115 // cerr << "rcXMLParser error: unknown element <" << name << ">" << endl;
116  }
117 
118  return !mErrors;
119 }

121 // End tag of element encountered, ie </experiment>
122 bool rcXMLParser::endElement( const QString&,
123  const QString&,
124  const QString& name )
125 {
126  if ( !mEmptyTree ) {
127  rcXMLElementType elementType = mNameMapper->elementType( name.latin1() );
128 
129  if ( elementType == eXMLElementVersion && !mVersionDetected ) {
130  rcXMLElement& e = *mElementLatest;
131  std::string& content = e.content();
132  rcValue fileVer( content.c_str() );
133  int v = fileVer;
134  if ( v < rcXMLOldestSupportedFileVersion ) {
135  ostrstream estr;
136  // File version too obsolete to parse, bail out now
137  ++mErrors;
138  estr << "\nobsolete experiment file version " << v
139  << " not supported.";
140  estr << ends;
141  estr.freeze();
142  mErrorString = estr.str();
143  } else if ( v > rcXMLFileVersion ) {
144  ostrstream estr;
145  // File version too new to parse, bail out now
146  ++mErrors;
147  estr << "\nexperiment file version " << v
148  << " not supported.";
149  estr << ends;
150  estr.freeze();
151  mErrorString = estr.str();
152  }
153  mNameMapper = rfGetXMLNameMapper( v );
154  mVersionDetected = true;
155  }
156 
157  // Control mode
158  setIgnoreMode( elementType, false );
159 
160  if ( ! mIgnoreAll ) {
161  if ( mElementLatest->type() == elementType ) {
162  // Pop latest
163  mElementLatest = mElementTree.parent( mElementLatest );
164  } else {
165  // Mismatched start and end tags, serious error!
166  // This will be reported by rcXMLErrorHandler so don't count it here
167  }
168  }
169  } else {
170  // Empty tree, serious error!
171  // This will be reported by rcXMLErrorHandler so don't count it here
172  }
173 
174  return !mErrors;
175 }

178 bool rcXMLParser::characters( const QString& ch )
179 {
180  if ( !mIgnoreAll ) {

198 void rcXMLParser::setIgnoreMode( rcXMLElementType elementType,
199  bool startElement )
200 {
201  switch ( mMode ) {
202  case eExperimentImportAllSettings:
203  case eExperimentImportExperimentSettings:
204  if ( elementType == eXMLElementExperimentData ) {
205  // We want to load settings only, ignore all measurement data
206  mIgnoreAll = startElement;

209

212  // We want to load measurements only, ignore all settings data
