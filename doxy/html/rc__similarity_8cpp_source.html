459 template <class T>
460 bool rcSimilarator::ssMatrixFill(deque<rcCorrelationWindow<T> >& tWin)
461 {
462  rmAssert(_SMatrix.size() == _matrixSz);
463 
464  const int32 tWinSz = tWin.size();
465  rmAssert(tWinSz <= (int32)_matrixSz);
466 
467  int32 cacheSz = _cacheSz;
468  if (cacheSz <= 2)
469  cacheSz = tWinSz + 2;
470  const int32 cacheBlkSz = cacheSz - 2;
471  progressNotification* progress = 0;
472  if (_notify) progress =
473  new progressNotification(tWinSz*(tWinSz-1)/2, _guiUpdate);
474 
475  /* The following was designed to work well when the number of images
476  * that can be stored in memory at any one time is space limited.
477  * In the absence of other consumers of cache resources, it reads in
478  * O((N^2)/C) images from disk, where N is the number of images in
479  * the movie fifo and C is the count of cache entries available.
480  * The basic algorithm works as follows:
481  *
482  * Step 1 - Use the first cache block size number of images
483  * remaining and perform cross correlations on all of them.
484  *
485  * Step 2 - For all the remaining images, read in one image at a
486  * time, correlating each with every image from step 1 before moving
487  * on to the next image.
488  *
489  * Note: In both steps 1) and 2), passes through the cached images
490  * are done in alternating directions to guarantee that the most
491  * recently used images get processed first. This is done to make
492  * performance degradation more linear in the presence of other
493  * consumers of the cache. In particular, it limits pathological
494  * cases where reading needed images into the cache causes other
495  * needed images to be flushed from the cache before they get
496  * used. Note that this assumes an LRU type caching policy.
497  *
498  * Step 3 - Update the current starting point by the cache block
499  * size.
500  *
501  * Repeat the entire process until the new starting point is >= the
502  * fifo size.
503  */
504  for (int32 i = 0; i < tWinSz;
505  /* Step 3 - Update start */ i += cacheBlkSz) {
506  int32 cacheIncr = 1, cacheBegin, cacheEnd;
507  int32 firstUncachedFrame = i + cacheBlkSz;
508  if (firstUncachedFrame > tWinSz)
509  firstUncachedFrame = tWinSz;
510 
511  /* Step 1 - Fill cache.
512  */
513  for (int32 j = i + 1; j < firstUncachedFrame; j++) {
514  if (cacheIncr == 1) {
515  cacheIncr = -1; cacheBegin = j - 1; cacheEnd = i - 1;
516  }
517  else {
518  cacheIncr = 1; cacheBegin = i; cacheEnd = j;
519  }
520 
521  for (int32 k = cacheBegin; k != cacheEnd; k += cacheIncr) {
522  rmAssert((j >= 0) && (j < tWinSz));
523  rmAssert((k >= 0) && (k < tWinSz));
524  const double r = correlate(tWin[j], tWin[k]);
525  _SMatrix[j][k] = _SMatrix[k][j] = r;
526  tWin[k].frameBuf().unlock();
527  if (progress && progress->update()) {
528  tWin[j].frameBuf().unlock();
529  delete progress;
530  return false;
531  }
532  } // End of: for ( k = cacheBegin; k != cacheEnd; k += cacheIncr)
533  tWin[j].frameBuf().unlock();
534  } // End of: for (int32 j = i + 1; j < firstUncachedFrame; j++)
535 
536  /* Step 2 - Correlate remaining images against cached images.
537  */
538  for (int32 j = firstUncachedFrame; j < tWinSz; j++) {
539  if (cacheIncr == 1) {
540  cacheIncr = -1; cacheBegin = firstUncachedFrame-1; cacheEnd = i-1;
541  }
542  else {
543  cacheIncr = 1; cacheBegin = i; cacheEnd = firstUncachedFrame;
544  }
545 
546  for (int32 k = cacheBegin; k != cacheEnd; k += cacheIncr) {
547  rmAssert((j >= 0) && (j < tWinSz));
548  rmAssert((k >= 0) && (k < tWinSz));
549  const double r = correlate(tWin[j], tWin[k]);
550  _SMatrix[j][k] = _SMatrix[k][j] = r;
551  tWin[k].frameBuf().unlock();
552  if (progress && progress->update()) {
553  tWin[j].frameBuf().unlock();
554  delete progress;
555  return false;
556  }
557  } // End of: for (k = cacheBegin; k != cacheEnd; k += cacheIncr)
558  tWin[j].frameBuf().unlock();
559  } // End of: for (int32 j = i + 1; j < firstUncachedFrame; j++)
560  }
561 
562  if (progress) delete progress;
563 
564  return true;
565 }

853  }
854 
855  for (uint32 i = 0; i < (_matrixSz-1); i++)
856  for (uint32 j = (i+1); j < _matrixSz; j++) {
857  _sums[i] += _SMatrix[i][j];
858  _sums[j] += _SMatrix[i][j];
859  }
860 
861  for (uint32 i = 0; i < _matrixSz; i++) {
862  for (uint32 j = i; j < _matrixSz; j++) {
863  double rr =
864  _SMatrix[i][j]/_sums[i]; // Normalize for total energy in samples
865  _entropies[i] += shannon(rr);
866 
867  if (i != j) {
868  rr = _SMatrix[i][j]/_sums[j];//Normalize for total energy in samples
869  _entropies[j] += shannon(rr);
870  }

873  }
