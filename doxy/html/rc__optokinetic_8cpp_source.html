54 bool rfOptoKineticEnergy (vector<rcWindow>& images, vector<double>& energy, const rsCorrParams& params,
55  rcProgressIndicator* pIndicator )
56 {
57  assert (images.size () > 1);
58  assert (images.size () == energy.size ());
59  vector<rcWindow> locals(images.size());
60  vector<double> scoreTotals (images.size());
61  vector<vector<double> > scores(images.size());
62  const double tinyNum = 1e-10; // When using rcSumPlogPNorm, added to every score to guarantee non-zero value
63  bool finished = true;
64 
65  // O (n^2/2 - n)
66  // Pre-initialize with unity for identity cross correlate
67  for (unsigned int i = 0; i < images.size(); i++) energy[i] = 1.0;
68 
69  if (params.pp != rcPreNone)
70  {
71  rmAssert (1); // Not implemented
72  }
73  else if (images[0].depth() > rcPixel8 ) {
74  switch ( params.rc ) {
75  case rcColorReduceAll32to8:
76  // Reduce all color and gray 24/32-bit images to 8-bit gray images
77  rfRcWindow32to8(images, locals);
78  break;
79  case rcColorReduceGray32to8:
80  // Reduce depth of all gray 24/32-bit images to 8-bit gray images
81  if ( images[0].isGray() )
82  rfRcWindow32to8(images, locals);
83  else
84  locals = images; // Analyze all bytes of color originals
85  break;
86  case rcColorUse32:
87  // Analyze all bytes of all originals
88  locals = images;
89  break;
90  default:
91  rmAssert( 1 );
92  break;
93  }
94  }
95  else
96  locals = images;
97 
98  if (params.em == rcSumPlogPNorm)
99  {
100  for (unsigned int i = 0; i < images.size(); i++)
101  {
102  scores[i].resize(scores.size());
103  scores[i][i] = scoreTotals[i] = 1 + tinyNum;
104  energy[i] = 0;
105  }
106  }
107  else if (params.em == rcSumPlogP)
108  // Pre-initialize with unity for identity cross correlate
109  for (unsigned int i = 0; i < images.size(); i++) energy[i] = 0.0; // Because log2(1) == 0
110  else
111  // Pre-initialize with unity for identity cross correlate
112  for (unsigned int i = 0; i < images.size(); i++) energy[i] = 1.0;
113 
114  // Progress update variables
115  uint32 lastProgress = 0;
116  const uint32 imageSize = images[0].width() * images[0].height();
117  // Progress update after this many pixels have been processed
118  const uint32 updateLimit = 2000000;
119  uint32 processedPixels = 0;
120 
121 
122  for (unsigned int i = 0; i < images.size (); i++)
123  {
124  for (unsigned int j = i + 1; j < images.size(); j++)
125  {
126  rcCorr res;
127 
128  switch (params.match)
129  {
130  default:
131  case rcNormalizedCorr:
132  rfCorrelate (locals[i], locals[j], params, res);
133  break;
134  case rcIntFrequencyDist:
135  {
136  rcWindow iw (locals[j], 1, 1, locals[j].width() - 2, locals[j].height() - 2);
137  double minn (1.0), maxx (0.0);
138  for (uint32 p = 0; p < 3; p++)
139  for (uint32 q = 0; q < 3; q++)
140  {
141  rcWindow jw (locals[i], p, q, locals[i].width() - 2, locals[i].height() - 2);
142  rfCorrelate (iw, jw, params, res);
143  if (res.r() <= minn) minn = res.r();
144  if (res.r() >= maxx) maxx = res.r();
145  }
146  res.r(maxx - minn);
147  assert (res.r() >= 0 && res.r() <= 1.0);
148  }
149  break;
150  }
151 
152  switch (params.em)
153  {
154  case rcSumSquare:
155  energy[i] += res.r();
156  energy[j] += res.r();
157  break;
158  case rcSumPlogP:
159  {
160  double rr = res.r() + 0.0000000000001;
161  energy[i] += -1.0 * rr * log2 (rr);
162  energy[j] += -1.0 * rr * log2 (rr);
163  }
164  break;
165  default:
166  case rcSumPlogPNorm:
167  {
168  double rr = res.r() + tinyNum;
169  scoreTotals[i] += rr;
170  scoreTotals[j] += rr;
171  scores[i][j] = rr;
172  scores[j][i] = rr;
173  }
174  break;
175  }
176 
177  if ( pIndicator ) {
178  processedPixels += imageSize;
179  // Update frequency depends on total number of processed pixels
180  if ( processedPixels > updateLimit ) {
181  processedPixels -= updateLimit;
182  uint32 percent = uint32(double(i+1)/images.size() * 100.0);
183  // However, minimum progress increment is 1%
184  if ( percent > lastProgress ) {
185  bool abort = pIndicator->progress( percent );
186  lastProgress = percent;
187  if ( abort ) {
188  finished = false;
189  goto abort;
190  }
191  }
192  }
193  }
194  }
195  }
196 
197  if (params.em == rcSumPlogPNorm)
198  {
199  for (unsigned int i = 0; i < images.size(); i++)
200  {
201  for (unsigned int j = i; j < images.size(); j++)
202  {
203  double rr = scores[i][j]/scoreTotals[i]; // Normalize for total energy in samples
204  energy[i] += -1.0 * rr * log2 (rr);
205 
206  if (i != j)
207  {
208  rr = scores[i][j]/scoreTotals[j]; // Normalize for total energy in samples
209  energy[j] += -1.0 * rr * log2 (rr);
210  }
211  } // End of : for (unsigned int j = i; j < images.size(); j++)
212 
213  energy[i] = energy[i]/log2(images.size()); // Normalize for number of samples
214  } // End of: for (unsigned int i = 0; i < images.size(); i++)
215  }
216  else
217  {
218  double n (images.size());
219 
220  for (unsigned int i = 0; i < energy.size(); i++) energy[i] = fabs(energy[i]) / n;
221  }
222 
223 
224  // Reporting: Derivative (raw) or cumulative
225  if (params.rp == rcTimeIntegral)
226  {
227  rfIntegrateEnergy (energy);
228  }
229  abort:
230  // Abort analysis, results are not guaranteed to be in a consistent state
231 
232  return finished;
233 }

251 void rfOptoKineticVariance (vector <rcWindow>& sequence, rcWindow& varImage)
252 {
253  rmAssert (sequence.size());
254  rmAssert (sequence.size() <= 257); // guarantee no numerical overflow
255  uint32 n (sequence.size());
256 
257  // Check the sizes
258  uint32 width = sequence[0].width();
259  uint32 height = sequence[0].height();
260 
261  rcWindow sum (width, height, rcPixel16);
262  rcWindow sumsq (width, height, rcPixel32);
263  sum.setAllPixels (0);
264  sumsq.setAllPixels (0);
265 
266  // Produce Sum and Sum sq
267 
268  for (vector<rcWindow>::iterator img = sequence.begin(); img != sequence.end(); img++)
269  {
270  for (uint32 j = 0; j < height; j++)
271  {
272  uint8 *pelRow = img->rowPointer (j);
273  uint16 *sumRow = (uint16 *) sum.rowPointer (j);
274  uint32 *sumSqRow = (uint32 *) sumsq.rowPointer (j);
275 
276  for (uint32 i = 0; i < width; i++, pelRow++, sumRow++, sumSqRow++)
277  {
278  *sumRow += *pelRow;
279  *sumSqRow += squareTable [*pelRow];
280  }
281  }
282  }
283 
284  // Now produce variance using: variance = n * sumSq - sum * sum
285  for (uint32 j = 0; j < height; j++)
286  {
287  uint16 *sumRow = (uint16 *) sum.rowPointer (j);
288  uint32 *sumSqRow = (uint32 *) sumsq.rowPointer (j);
289 
290  for (uint32 i = 0; i < width; i++, sumRow++, sumSqRow++)
291  {
292  uint32 var = *sumSqRow * n;
293  uint32 tmp = *sumRow;
294  var = var - tmp * tmp;
295  *sumSqRow = var;
296  }
297  }
298 
299  varImage = sumsq;
300 
301 }
