101 template<class T>
102 rcLineSegment<T>::rcLineSegment (const rc2dVector<T>& pa, const rc2dVector<T>& pb)
103 {
104  if (pa == pb)
105  throw general_exception ("Degenerate Line");
106 
107  mAngle = (pa - pb).angle();
108  T sn = sin (- mAngle);
109  T cs = cos (- mAngle);
110  rc2dVector<T> p (cs*pa.x()-sn*pa.y(),sn*pa.x()+cs*pa.y());
111  mDist = p.y();
112  if (mDist < 0)
113  {
114  mDist = - mDist;
115  mAngle = rcRadian (rkPI) + mAngle;
116  }
117  mSin = sin(mAngle);
118  mCos = cos(mAngle);
119 }

155 template<class T>
156 vector<rc2dVector<T> >
157 rcLineSegment<T>::intersection(const rcLineSegment<T> & line) const
158 {
159  vector<rc2dVector<T> > pts;
160  double denom = line.mCos * mSin - mCos * line.mSin;
161  if(denom == 0.0)
162  return pts;
163  pts.push_back((rc2dVector<T> (mCos,mSin)* line.mDist -
164  rc2dVector<T> (line.mCos,line.mSin) * mDist)/denom);
165  return pts;
166 }
