281  // Call this to update status bar with completion percentage
282  virtual bool progress( double percentComplete ) {
283  if ( percentComplete != mLastPercentage )
284  {
285  rcTimestamp now = rcTimestamp::now();
286  rcTimestamp timeSinceLastUpdate = now - mLastUpdate;
287 
288  // Update only if specified interval has passed
289  if ( timeSinceLastUpdate.secs() > mUpdateInterval )
290  {
291  snprintf( mMessageBuf, rmDim( mMessageBuf ), mFormatMessage.c_str(), percentComplete );
292  mDomain->notifyStatus( mMessageBuf );
293  mLastPercentage = percentComplete;
294  if ( mUpdateDisplay ) {
295  mDomain->notifyTime( mDomain->getExperimentLength() );
296  mApp->processEvents(QEventLoop::ExcludeUserInputEvents, 5 ); // Allow 5 ms for event processing
297  } else {
298  mApp->processEvents( QEventLoop::ExcludeUserInputEvents, 1 ); // Allow 1 ms for event processing
299  }
300  mLastUpdate = now;
301  // TODO: add abort logic for shutting down
302  }
303  }
304  return false;
305  }
