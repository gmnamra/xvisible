43 template<class P, class A>
44 void rfImageIntegrals (const P* pixelTypePtr, const A* accTypePtr, const rcWindow& image, rcWindow& sumColInt, rcWindow& sumRowInt)
45 {
46  rmUnused (pixelTypePtr);
47  rmUnused (accTypePtr);
48  rmAssert (image.isBound());
49  rmAssert (sumColInt.isBound());
50  rmAssert (sumRowInt.isBound());
51  rmAssert (image.size() == sumColInt.size());
52  rmAssert (image.size() == sumRowInt.size());
53 
54  sumRowInt.set (0.0);
55  sumColInt.set (0.0);
56 
57  // Fillup the first row of col integrators, and the first col of row integrators
58  const P *p16 = (P *) image.pelPointer (0, 0);
59  A *r32 = (A *) sumColInt.pelPointer (0, 0);
60  for (int32 i = 0; i < image.width(); i++, p16++, r32++)
61  {
62  *r32 = *p16;
63  }
64 
65  A *c32 = (A *) sumRowInt.rowPointer (0);
66  uint32 crup = sumRowInt.rowPixelUpdate();
67  for (int32 j = 0; j < image.height(); j++, c32+=crup)
68  {
69  A pel = A ( *((P *) image.rowPointer (j) ) );
70  *c32 = pel * pel;
71  }
72 
73 
74  // Fillup the first column of the column integrators
75  A acc = sumColInt.getPixel (0,0);
76  r32 = (A *) sumColInt.rowPointer (1);
77  p16 = (P *) image.rowPointer (1);
78  uint32 rrup = sumColInt.rowPixelUpdate();
79  uint32 irup = image.rowPixelUpdate();
80  for (int32 j = 1; j < image.height(); j++, r32+=rrup, p16+=irup)
81  {
82  acc += *p16;
83  *r32 = acc;
84  }
85 
86  // Fillup the first row of the row integrators
87  acc = sumRowInt.getPixel (0,0);
88  c32 = (A *) sumRowInt.pelPointer (1, 0);
89  p16 = (P *) image.pelPointer (1, 0);
90  for (int32 i = 1; i < image.width(); i++, p16++, c32++)
91  {
92  A pel = A (*p16);
93  acc += pel * pel;
94  *c32 = acc;
95  }
96 
97 
98  // Now process from 1,1 onward.
99 
100 
101  for (int32 j = 1; j < image.height(); j++)
102  {
103  P *p16 = (P *) image.pelPointer (1, j);
104 
105  // current and previous rows for col integrators
106  A *rprev32 = (A *) sumColInt.pelPointer (1, j-1);
107  A *r32 = (A *) sumColInt.pelPointer (1, j);
108 
109  // current and previous integration pixel for row integrators
110  A *cprev32 = (A *) sumRowInt.rowPointer (j);
111  A *c32 = cprev32+1;
112 
113  // As we travel through pixels we are also traveling thorugh
114  // equal size integrators. Update all the column integrators
115  // and integration pixel of this row's integrator
116 
117  for (int32 i = 1; i < image.width(); i++, p16++, r32++, c32++, rprev32++)
118  {
119  A pel = A (*p16);
120 
121  // integrate row j of col integrators
122  *r32 = *rprev32 + pel;
123 
124  // integrate col i of row integartors
125  *c32 = *cprev32 + pel * pel;
126 
127  // Roll over
128  cprev32 = c32;
129  }
130  }
131 }

153 float rfImageACfield (const rcWindow& image, const rcIPair& target, rcWindow& floatDest)
154 {
155  rcWindow L (image.width(), image.height(), rcPixel32);
156  rcWindow L2 (image.width(), image.height(), rcPixel32);
157  rcWindow V2; rcWindow& dest = floatDest;
158  rcIRect field = rfImageACfieldArea (image, target);
159 
160  // @note Window into the passed in destination if it is bound and of the right depth and size
161  // else create one
162  if (dest.isBound() && dest.size() == image.size () && dest.frameBuf()->isD32Float ())
163  {
164  V2 = rcWindow (dest, target.x() / 2, target.y() / 2, field.width(), field.height());
165  }
166  else
167  {
168  dest = rcWindow (image.width(), image.height(), rcPixel32);
169  dest.frameBuf()->markD32Float (true);
170  V2 = rcWindow (dest, field.ul().x(), field.ul().y(), field.width (), field.height());
171  }
172 
173  uint32 lrup = L.rowUpdate() / L.depth();
174  uint32 l2rup = L2.rowUpdate() / L2.depth();
175  int64 maxvar (0);
176 
177  rfImageIntegrals (image, L, L2);
178  int64 n = target.x() * target.y();
179  int32 targetHeightbyRUP = (target.y()-1) * lrup;
180  rcIPair steps = L2.size() / target;
181 
182  for (uint32 j = 0; j < V2.height(); j++)
183  {
184  uint32 *leftSqRowPtr = (uint32 *) L2.rowPointer (j);
185  uint32 *topSumRowPtr = (uint32 *) L.rowPointer (j);
186  float *varFloatPtr = (float *) V2.rowPointer (j);
187 
188  // Row Processing
189  int32 cols(0);
190  for (cols = 0; cols < V2.width(); cols++, leftSqRowPtr++, topSumRowPtr++, varFloatPtr++)
191  {
192  uint32 *leftSumSq = leftSqRowPtr, *rightSumSq = leftSumSq + target.x() - 1;
193  uint32 *topSum = topSumRowPtr, *botSum = topSum + targetHeightbyRUP;
194  double d (0.0), dd (0.0);
195 
196  // The squares are arranged progressively in X
197  // S E
198  //[ 0] 1 2 3 4 5 9
199  //[ 1] 1 2 3 4 5 6
200  //[ 2] 1 2 3 4 5 6
201  //[ 3] 1 2 3 4 5 6
202  //[ 4] 1 2 3 4 5 6
203  //[ 5] 4 5 6 7 8 9
204  //[ 6] 1 2 3 4 5 6
205 
206  // If we are at the top left bin use the cached val (since we are writing results out to the same bin)
207  // If we are at other bins but at the first position in the row just subtract the bin
208  for (uint32 cc = 0; cc < target.y(); cc++, leftSumSq+= l2rup, rightSumSq+=l2rup)
209  {
210  dd += (*rightSumSq);
211  dd -= cols ? leftSumSq[-1] : 0;
212  }
213 
214  // The sums are arrange progressively in Y
215  // [ 0] 1 1 1 1 1 2 S
216  // [ 1] 2 2 2 2 2 3
217  // [ 2] 3 3 3 3 3 4
218  // [ 3] 4 4 4 4 4 5
219  // [ 4] 5 5 5 5 5 6 E
220  // [ 5] 7 6 6 6 6 7
221  // [ 6] 8 7 7 7 7 8
222 
223  for (uint32 bb = 0; bb < target.y(); bb++, topSum ++, botSum ++)
224  {
225  d += (*botSum);
226  d -= j ? topSum[-lrup] : 0;
227  }
228 
229  // Compute n SumOfSq - Sum^2
230  dd = dd * n - d * d;
231  *varFloatPtr = (float) dd;
232  if (dd > maxvar) maxvar = dd;
233  }
234  }
235 
236  // Window out the results
237  floatDest = dest;
238  return (float) maxvar;
239 }
