46 void rfGradientImageToVisualSegment (const rcWindow& peaks, const rcWindow& angle, rcVisualSegmentCollection& v)
47 {
48  rmAssert (peaks.size() == angle.size());
49  for (int32 j = 2; j < angle.height () - 2; j++)
50  {
51  const uint8 *mag = peaks.pelPointer(2, j); // 5x5 processing
52  const uint8 *ang = angle.pelPointer(2, j);
53  for (int32 i = 2; i < angle.width () - 2; i++, mag++, ang++)
54  {
55  if (! (*mag) ) continue;
56  rcAngle8 d (*ang);

58  rc2Fvector p2 = p1 + rc2Fvector (1.0f, rcRadian (d));
