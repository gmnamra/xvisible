48 (const T *begin, int len, const T &val)
49 {

113 bool rfQuadradic (T a, T b, T c, T& pos, T& neg)
114 {

400 
401 template < typename T >
402 int32 rcNearbyint(const T& d)
403 {
404  int32 z = int32 (d);
405  T frac = d - z;
406 
407  rmAssert (abs(frac) < T(1.0));
408 
409  if (frac > 0.5)
410  ++z;
411  else if (frac < -0.5)
412  --z;
413  else if (frac == 0.5 && (z&1) != 0) // NB: We also need the round-to-even rule.
414  ++z;
415  else if (frac == -0.5 && (z&1) != 0)
416  --z;
417 
418  rmAssert (abs(T(z) - d) < T(0.5) ||
419  (abs(T(z) - d) == T(0.5) && ((z&1) == 0)));
420  return z;
421 }
