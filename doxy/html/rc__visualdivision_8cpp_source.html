111 bool rcVisualFunction:: isIsoMorphic ()
112 {
113  vector<rcPolygon*>corr;
114 
115  // How many region polys intersect with me?
116  intersectSet (corr);
117  if (!corr.size() == 1)
118  {
119  return false;
120  }
121 
122  // So only one region poly intersects with this cell.
123  // How many cells intersect with this region poly.
124  list<rcVisualFunction>::const_iterator cell = sharedKin()->visualBodies ().begin();
125  list<rcVisualFunction>::const_iterator cellend = sharedKin()->visualBodies ().end();
126  for (; cell != cellend; cell++)
127  {
128  if (cell->isState(eIsOutSide)) continue;
129  if (cell->isState(eIsDivided)) continue;
130 
131  if (cell->id() == id()) continue; // skip me
132  rcPolygon dispoly (cell->polygons().back());
133  if (corr[0]->intersects (dispoly))
134  {
135  return false;
136  }
137  }
138  return true;
139 }
