310 // Set observed min/max values
311 void rcPositionTrackImpl::setMinMaxValues( const rcFPair& value )
312 {
313  // First time through, set current min and max values
314  if ( _currentMinimumValue.x() > _currentMaximumValue.x() )
315  {
316  _currentMinimumValue.x() = value.x();
317  _currentMaximumValue.x() = value.x();
318  }
319  // First time through, set current min and max values
320  if ( _currentMinimumValue.y() > _currentMaximumValue.y() )
321  {
322  _currentMinimumValue.y() = value.y();
323  _currentMaximumValue.y() = value.y();
324  }
325 
326  // Update min and max current values
327  if ( value.x() < _currentMinimumValue.x() )
328  _currentMinimumValue.x() = value.x();
329  if ( value.x() > _currentMaximumValue.x() )
330  _currentMaximumValue.x() = value.x();
331  if ( value.y() < _currentMinimumValue.y() )
332  _currentMinimumValue.y() = value.y();
333  if ( value.y() > _currentMaximumValue.y() )
334  _currentMaximumValue.y() = value.y();
335 }
