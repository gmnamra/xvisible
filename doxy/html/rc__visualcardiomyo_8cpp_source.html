231 
232  clearState (eHasFullyExtended);
233  rc2Fvector csum;
234  for (uint32 i = 0; i < 2; i++)
235  {
236  rcIPair ic ((int32) mMajorEndPoints[i].x(), (int32) mMajorEndPoints[i].y());
237  csum += mMajorEndPoints[i];
238  int32 umin = rmMin (ic.x(), ic.y());
239  if (umin < mRad) mRad =umin - 1;

241  umin = use4fixed().width() - (1 + ic.x()+mRad); // -1 moved to - side
242  umin -= 1; // border of 1 from the edge of the image
243  if (umin < 0) mRad += umin; // reduce to fit
244  rmAssert (mRad >= (int32) eMinMyoOtherDimensionInPixels);
245  umin = use4fixed().height() - (1 + ic.y()+mRad); // -1 moved to - side
246  umin -= 1; // border of 1 from the edge of the image
247  if (umin < 0) mRad += umin; // reduce to fit
248  rmAssert (mRad >= (int32) eMinMyoOtherDimensionInPixels);
249  rcPolygon pg (mRad /2.0f, 8);
250  rc2Xform xt;
251  xt.trans (mMajorEndPoints[i]);

253  rcFRect fr; rcIRect ir;
254  pg.orthogonalEnclosingRect (fr);
255  ir = roundRectangle (fr);

257  rcWindow tmp (use4fixed (), ir, true);

294 void rcVisualFunction::beatProcess ()
295  {
296  // As long as the size is less than frames per period
297  // built up the time series
298  if ((int32) mDimensionTs.size() < ((int32)framesPerPeriod ()))
299  {
300  mDimensionTs.push_back (age());
301  mDimensions.push_back (mAffineSize.x());
302  rmAssert (mDimensionTs.size() == mDimensions.size());
303  }
304 
305  if (mDimensionTs.size() >= (uint32) integralFramesPerPeriod ())
306  {
307  rmAssert (mDimensionTs.size() == mDimensions.size());
308  clear (eBeatInfo);
309 
310  // What is the median dimension
311  // Must be after the first length measurement.
312  // @note: if we have less than 12 frames per
313  if (mDimensionTs.size() < ((uint32)eMinFramesPerPeriod4Stat))
314  mBeatMedian = mFirstBeatLength;
315  else
316  mBeatMedian = rfMedian (mDimensions);
317 
318  // What is the minimum and when is it reached
319  vector<float>::iterator minL = min_element (mDimensions.begin(),
320  mDimensions.end());
321  uint32 index = minL - mDimensions.begin();
322 
323  // interpolate and get min length and time
324  // Report
325  // Interpolation is on Median - Length (t - 1),
326  // Median - Length (t),
327  // Median - Length (t + 1)
328 
329  float minIntL (0), minIntT (0), minIntNormT (0);
330  if (index > 0 && index < (mDimensions.size() - 1))
331  {
332  minIntNormT = minIntT = parabolicFit (mBeatMedian - mDimensions[index-1],
333  mBeatMedian - *minL,
334  mBeatMedian - mDimensions[index+1],
335  &minIntL);
336  }
337 
338  // Peak return is mBeatMedian - minimumLength.
339  // Scale Time according to the actual
340  if (minIntT > 0.0) minIntT *= (mDimensionTs[index+1] - mDimensionTs[index]);
341  if (minIntT < 0.0) minIntT *= (mDimensionTs[index] - mDimensionTs[index-1]);
342  minIntL = mBeatMedian - minIntL;
343 
344  // If Interpolated is too close or inconclusive/degenerate use the measured minimum
345  if (rfRealEq (minIntL, 0.0f) || rfRealEq (minIntT, 0.0f) || rfRealEq (*minL, minIntL, 1.0f))
346  minIntL = *minL;
347  mBeatShortenning = 1.0f - minIntL / mBeatMedian;
348 
349  // Construct Contraction and Relaxation immediate angles
350 
351  // Actual time stamp at the minimum is sub-frame time + time at the center.
352 
353  minIntT += mDimensionTs[index];
354  rcLineSegment<float> cLine (rc2Fvector (minIntNormT , minIntL), rc2Fvector(-1.0f,
355  mDimensions[index-1]));
356  rcLineSegment<float> rLine (rc2Fvector (minIntNormT , minIntL), rc2Fvector(+1.0f,
357  mDimensions[index+1]));
358  // Put all the LTs out including the interpolated value, in order of increasing time.
359  // A little inefficient for now
360  uint32 fi;
361  for (fi = 0; fi <= index; fi++)
362  {
363  mInterpolatedMin.push_back (rcFPair (mDimensionTs[fi], mDimensions[fi]));
364  }
365 
366  if (minIntL != *minL)
367  mInterpolatedMin.push_back (rcFPair (minIntT , minIntL ));
368 
369  // Assert that iL is within +/- 1 from minL
370 
371  for (; fi < mDimensions.size(); fi++)
372  {
373  mInterpolatedMin.push_back (rcFPair (mDimensionTs[fi], mDimensions[fi]));
374  }
375 
376  // Transform Angles to our definition:
377  // ContractionAngle = - (90 + (- cLine.angle())
378  // RelaxationAngle = 90 - rLine.angle()
379  rcRadian contraction (cLine.angle());
380  rcRadian relaxation (rLine.angle());
381  contraction.normSigned();
382  relaxation.normSigned ();
383  contraction = - (rkPI/2.0 + (- contraction.Double()));
384  relaxation = rkPI/2.0 - relaxation.Double();
385  mPeakAngles.push_back (rcPair<rcRadian> (contraction, relaxation));
386 
387  // Clear temporal buffer
388  mDimensionTs.clear ();
389  mDimensions.clear ();
390  mark (eBeatInfo);
391  mBeatNumber += 1;
392  }
393 
394  // If we have at least 2 beats worth of frames
395  if (mBeatNumber > 1 && similarity().size() > mBeatAC.size())
396  {
397  clear (ePacingInfo);
398  vector<float>::const_iterator Begin = similarity().end();
399  advance (Begin, - (int32) mBeatAC.size());
400  rf1DAutoCorr (Begin, similarity().end(), mBeatAC);
401  rs1DFourierResult rslt;
402  rf1DFourierAnalysis(mBeatAC, rslt, rc1DFourierForceFFT);
403  mContractionFrequency = (float) mBeatAC.size() / rslt.peakFrequency;
404  mark (ePacingInfo);
405  }
406  }
