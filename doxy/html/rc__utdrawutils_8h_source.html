103 static void rfDrawCylinder (deque<rcWindow>& cylinder, rc2Dvector center, double ra, double rb)
104 {
105  for (uint32 i = 0; i < cylinder.size(); i++)
106  cylinder[i].setAllPixels (0);
107 
108  int32 width (cylinder[0].width());
109  int32 height (cylinder[0].height());
110 
111  double dIdT = cylinder.size() / 255.0;
112  double iDiDt (0.0);
113 
114  double minR = rmMin (ra, rb);
115  double maxR = rmMax (ra, rb);
116 
117  // Basic Linear Model
118  // TBD: Support power law models
119  // Forward Diffusion
120  for (uint32 tt = 0; tt < cylinder.size(); tt++)
121  for(int32 i=0;i< width;i++)
122  {
123  for(int32 j=0;j< height; j++)
124  {
125  rc2Dvector pos(i+0.5,j+0.5);
126  if (center.distance (pos) <= maxR)
127  cylinder[tt].setPixel (i,j,uint32 (iDiDt));
128  iDiDt += dIdT;
129  }
130  }
131 
132  // Reverse Diffusion
133  iDiDt = 0.0;
134  for (uint32 tt = 0; tt < cylinder.size(); tt++)
135  for(int32 i=0;i< width;i++)
136  {

140  if (center.distance (pos) <= minR)
141  cylinder[tt].setPixel (i,j,255 - uint32 (iDiDt));

203 static void rfDrawLine(rcWindow& image, rcIPair& p1, rcIPair& p2,uint8 ink)
204 {
205  int32& x1 = p1.x();
206  int32& y1 = p1.y();
207  int32& x2 = p2.x();
208  int32& y2 = p2.y();
209 
210  int32 d, x, y, ax, ay, sx, sy, dx, dy;
211  dx = x2-x1; ax = rmABS(dx)<<1; sx = rmSGN(dx);
212  dy = y2-y1; ay = rmABS(dy)<<1; sy = rmSGN(dy);
213  x = x1;
214  y = y1;
215  if (ax>ay) { /* x dominant */
216  d = ay-(ax>>1);
217  for (;;) {
218  image.setPixel (x, y, ink);
219  if (x==x2) return;
220  if (d>=0) {
221  y += sy;
222  d -= ax;
223  }
224  x += sx;
225  d += ay;
226  }
227  }
228  else { /* y dominant */
229  d = ax-(ay>>1);
230  for (;;) {
231  image.setPixel (x, y, ink);
232  if (y==y2) return;
233  if (d>=0) {
234  x += sx;
235  d -= ay;
236  }
237  y += sy;
238  d += ax;
239  }
240  }
241 }
