205 void rcEngineMoviePlayback::genMovieData(vector<pbRec>& movieData,
206  double msPerUpdate)
207 {
208  uint32 updateCnt = 1;
209  uint32 curFrame = 0xFFFFFFFF;
210  movieData.resize(0);
211  double curTime = (_startTimeInMS < 0.0) ? 0.0 : _startTimeInMS;
212  uint32 slotIndex = (uint32)(curTime/_msPerFrame + .5);
213  if (slotIndex >= _timeline.size()) {
214  slotIndex = _timeline.size() - 1;
215  curTime = slotIndex * _msPerFrame;
216  }
217 
218  while (slotIndex < _timeline.size()) {
219  if (curFrame != _timeline[slotIndex]) {
220  if (!movieData.empty()) {
221  rmAssert(movieData.back().frameIndex == curFrame);
222  movieData.back().nextUpdateCnt = updateCnt;
223  }
224  updateCnt = 0;
225  curFrame = _timeline[slotIndex];
226  movieData.push_back(pbRec(curFrame, 1));
227  }
228  updateCnt++;
229  curTime += msPerUpdate * _rateMultiplier;
230  slotIndex = (uint32)(curTime/_msPerFrame + .5);
231  }
232 }
