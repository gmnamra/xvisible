199 rcFrameGrabberStatus rcQVideoGrabber::getNextFrame(rcSharedFrameBufPtr& fptr,
200  bool isBlocking)
201 {
202  if (!_isValid)
203  return eFrameStatusError;
204 
205  if (_frameCount > 0)
206  _frameCount--;
207  else if (_frameCount == 0)
208  return eFrameStatusEOF;
209 
210  /* fgStatus is a place to collect errors that happen before shared
211  * resources have been freed.
212  */
213  rcFrameGrabberStatus fgStatus = eFrameStatusOK;
214 
215  rcSharedMemoryUser& consumer = _shmemCtrlP->getShmemUser();
216  rcSharedMemError err;
217  bool needFrame = true;
218 
219  while (needFrame)
220  {
221  rcVideoSharedMemLayout* imgData =
222  (rcVideoSharedMemLayout*)consumer.acquireSharedMemory(err, isBlocking);
223 
224  if (!imgData)
225  {
226  if (err == rcSharedMemNotAvailable)
227  {
228  _lastError = eFrameErrorFrameNotAvailable;
229  return eFrameStatusNoFrame;
230  }
231 
232  fprintf(stderr, "Parent: error acquiring shared data. err: %d\n", err);
233  perror("Parent: ");
234 
235  _lastError = eFrameErrorIPC;
236  return eFrameStatusError;
237  }
238 
239  if ((imgData->videoState.acqFlags & ~rcACQFLAGS_FRAMESKIPPED) ==
240  rcACQFLAGS_FRAMEAVAIL)
241  {
242 
243  /* Copy pixels from shared data into a new frame buffer created from the
244  * heap.
245  */
246  try
247  {
248  /* First, try to find a buffer that is currently only
249  * referenced by the rcSharedFrameBufPtr _buffers in
250  * _buffers. Its reference count therefore should be 1.
251  */
252  vector<rcSharedFrameBufPtr>::iterator bufptr = _buffers.begin();
253  for ( ; bufptr != _buffers.end(); bufptr++)
254  if ((*bufptr).refCount() == 1)
255  break;
256 
257  if (bufptr == _buffers.end()) {
258  fptr = rcSharedFrameBufPtr(new rcFrame(imgData->rawPixels,
259  imgData->rowUpdate,
260  imgData->width,
261  imgData->height,
262  imgData->pixelDepth,
263  imgData->isGray));
264  // Init color map
265  if (imgData->pixelDepth == rcPixel8)
266  {
267  fptr->initGrayColorMap();
268  }
269  else if (imgData->pixelDepth == rcPixel16)
270  rfFillColorMap(0, fptr);
271  }
272  else {
273  fptr = *bufptr;
274  fptr->loadImage(imgData->rawPixels, imgData->rowUpdate, imgData->width,
275  imgData->height, imgData->pixelDepth, imgData->isGray);
276  }
277 
278  if (!fptr) {
279  _lastError = eFrameErrorSystemResources;
280  fgStatus = eFrameStatusError;
281  }
282  else
283  fptr->setTimestamp(imgData->timestamp);
284  }
285  catch (...)
286  {
287  _lastError = eFrameErrorSystemResources;
288  fgStatus = eFrameStatusError;
289  }
290  needFrame = false;
291  }
292  else if (imgData->videoState.acqFlags & rcACQFLAGS_FRAMEERROR)
293  {
294  _lastError = eFrameErrorCameraCapture;
295  fgStatus = eFrameStatusError;
296  }
297 #ifdef DEBUGGING
298  else
299  fprintf(stderr, "xyzzy looping...\n");
300 #endif
301 
302  /* All data read from shared buffer. Free it.
303  */
304  switch (err = consumer.releaseSharedMemory())
305  {
306  case rcSharedMemNoError:
307  break;
308 
309  default:
310  fprintf(stderr, "Parent: error freeing shared data, err: %d\n", err);
311  perror("Parent: ");
312  _lastError = eFrameErrorIPC;
313  fgStatus = eFrameStatusError;
314 
315  } // End of: switch (consumer.releaseSharedMemory())
316 
317  if (fgStatus != eFrameStatusOK)
318  return fgStatus;
319  } // End of: while (needFrame)
320 
321  return eFrameStatusOK;
322 }
