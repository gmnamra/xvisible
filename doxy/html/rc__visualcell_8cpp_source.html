204 void rcVisualFunction::populationMigration ()
205 {
206  clear (eDispersion);
207  if (sharedKin()->mFunctionMap.size() < 2) return;
208 
209  // else measure population migration
210  list<rcVisualFunction>::const_iterator cell;
211  vector<float> cellDiffusion;
212  vector<float> distances;
213 
214  distances.reserve (sharedKin()->mFunctionMap.size() - 1);
215  cellDiffusion.reserve (sharedKin()->mFunctionMap.size() - 1);
216 
217  for(cell = sharedKin()->mFunctionMap.begin(); cell != sharedKin()->mFunctionMap.end(); cell++)
218  {
219  // Skip me
220  if (cell->id() == id()) continue;
221 
222  // Vector from me to another cell
223  rc2Dvector c2c (position().x() - cell->position().x(),
224  position().y() - cell->position().y());
225 
226  // Distance from me to another cell
227  float f = position().distance (cell->position());
228  distances.push_back (f);
229 
230  // My velocity towards the other cell (dot product)
231  rc2Dvector vd (velocity().x(), velocity().y());
232  cellDiffusion.push_back (c2c * vd);
233  }
234 
235  // Now get min and max distances
236  vector<float>::const_iterator mind = min_element (distances.begin(),
237  distances.end());
238  vector<float>::const_iterator maxd = max_element (distances.begin(),
239  distances.end());
240  // If all cells are attached to me, we are undetermined
241  if (rfRealEq (*maxd, 0.0f)) return;
242 
243  // Find min and max distances; if there is no range, pick max
244  float scale (*maxd - *mind);
245  if (rfRealEq (scale, 0.0f)) scale = *maxd;
246 
247  // Sum persistence towards me or away from me scale pro closest to me.
248  // Use a linear map to use distance
249  float dispersion (0.0f);
250  for (uint32 i = 0; i < distances.size(); i++)
251  dispersion += (distances[i] * cellDiffusion[i]) / scale;
252 
253  mMotionPattern = dispersion;
254  mark (eDispersion);
255 }

365 void rcVisualFunction::measure ()
366 {
367  step();
368 
369  // Invalidate all measurements before we begin
370  reset ();
371 
372  if (sharedKin()->movingPolygons().size() < 1) return;
373  // sharedKin()->dumpPolys (cerr);
374 
375  //@note allow unknown.
376  if (rfCellNotViable (*this)) return;
377  static const rc2Fvector tinyMove (2.0f, 2.0f);
378 
379  // Get the associated Segmented Poly and most recent poly. Use wc as variable
380  vector<rcPolygon>::const_iterator pc = polygons().end(); std::advance (pc, -1);
381  vector<rcPolygon>::const_iterator vc = sharedKin()->movingPolygons().begin() + label();
382  vector<rcPolygon>::const_iterator wc;
383 
384  // If we have a reasonable correspondence use it
385  // Otherwise use the updated rect from last time
386  rcTimingInfo profile (10, sharedKin()->frameCount() + 1);
387  profile.nextPass (true);
388 
389  if (vc != sharedKin()->movingPolygons().end() && vc->isValid())
390  {
391  // We do not need this one around
392  //@note we use two different ShapeRefs here. For calculating center of mass we use the
393  // the orthogonal window. For shape we use the segmented polygon.
394  rcFRect fr;rcIRect ir;
395  vc->orthogonalEnclosingRect(fr);
396  ir = roundRectangle (fr);
397  rcWindow frame (use4moving(), ir);
398  rcShapeRef frameRef = boost::shared_ptr<rcShape> (new rcShape (frame));
399  rc2Fvector dD = frameRef->centerOfmass (rcShape::eImage);
400 
401  //@note this is poorely placed here. Change.
402  mMass = (float) frameRef->mass ();
403 
404  dD += rc2Fvector (fr.ul());
405  profile.touch (1);
406  if (!isCardiacCell () && !isLabelProtein ())
407  {
408  // @note: if we have motion vectors, compute transformation
409  if (vXform (eMotionCenterANDTP, mMotionXform))
410  {
411  dD = mMotionXform.transf ();
412  }
413 
414  rcFPair tolerance = predictedTolerance (dD);
415  correspond (dD, tolerance);
416  mPOLYs.push_back (*vc);
417  mark (eMotion);
418  mark (eDimensions);
419  }
420  else if (isCardiacCell ())
421  {
422  cardiacMeasure (vc);
423  }
424  else if (isLabelProtein ())
425  {
426  // @note
427  // dD is com of the most correponding region.
428  // mCurrentPosition is known position before this measure.
429  // Immediately after division, there might not be separate
430  // regions corresponding to each child.
431  // So a gaussian model is used to track the daughters until
432  // they form a isomorphic (one to one) correspondence with a region.
433  // The first time they do, daughters are updated to the
434  // region and the cell is moved to the com of the region.
435 
436  float distance = dD.distance (mCurrentPosition);
437  float radius = sqrt (area() / rkPI) * 1.5f;
438 
439  if (rfRealEq (cQuality (), 1.0f, 0.001f) &&
440  rfRealEq (circularity (), (float) vc->circularity(), 0.067f) &&
441  distance < radius)
442  {
443  direct (dD);
444  // @note we are accepting the perfect correspondence
445  wc = vc;
446  }
447  else
448  {
449  dD = rc2Fvector (adaptBias (), adaptBias ());
450  rcFPair tolerance = predictedTolerance (dD);
451  profile.touch (0);
452  if (sharedKin ()->channelAvailable ())
453  correspondMutualChannel (dD, tolerance);
454  else
455  correspond (dD, tolerance);
456  // @note we are moving the last poly forward
457  // pc->transform (instantTransformation ());
458  wc = pc;
459  profile.touch(1);
460  }
461 
462 
463  mPOLYs.push_back (*wc);
464  mark (eMotion);
465  mark (eDimensions);
466  }
467 
468  else
469  {
470  rmAssert (0);
471  }
472  }
473  vector<std::string> lables;
474  lables.push_back (std::string ("Correspondence"));
475  profile.printInfo (2, lables);
476 }

541

581  if (!isAtStep (1))
582  {
583  mLastAngle = angle;
584  mark (ePersistence);
585  }
586 
587  mark (eSpeed);
588  mark (eVelocity);
589  mark (eDistance);
590  }
