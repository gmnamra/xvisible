90 void rcCubicGeneratePixel::gen2ndD(const vector<float>& y, vector<float>& y2,
91  float yp1, float ypn, bool v1, bool vn)
92 {
93  uint32 n = y.size(), last = n - 1;
94 
95  if (n != y2.size())
96  y2.resize(n);
97 
98  vector<float> u(last);
99 
100  if (v1) {
101  y2[0] = -0.5;
102  u[0] = 3.0*((y[1]-y[0]) - yp1);
103  }
104  else
105  y2[0] = u[0] = 0.0;
106 
107  for (uint32 i = 1; i < last; i++) {
108  float p = 0.5 * y2[i-1] + 2.0;
109  y2[i] = -0.5/p;
110  u[i] = (y[i+1] - y[i]) - (y[i] - y[i-1]);
111  u[i] = (3*u[i] - 0.5*u[i-1])/p;
112  }
113 
114  float qn = 0.0, un = 0.0;
115  if (vn) {
116  qn = 0.5;
117  un = 3*(ypn - (y[last] - y[last-1]));
118  }
119 
120  y2[last] = (un - qn*u[last-1])/(qn*y2[last - 1] + 1.0);
121 
122  for (uint32 k = last-1; k != (uint32)-1; k--)
123  y2[k] = y2[k]*y2[k+1] + u[k];
124 }

257 double rcCubicGeneratePixel::interpolate(const vector<float>& y,
258  const vector<float>& y2d, float x)
259 {
260  uint32 left = (uint32)x;
261  uint32 right = left + 1;
262 
263  rmAssert(y.size() == y2d.size());
264 
265  float a = (float)right - x;
266  float b = 1.0 - a;
267 
268  double retVal;
269 
270  if (b == 0.0) {
271  rmAssert(left < y.size());
272  retVal = y[left];
273  }
274  else {
275  rmAssert(right < y.size());
276  retVal = a*y[left] + b*y[right] +
277  ((a*a*a - a)*y2d[left] + (b*b*b - b)*y2d[right])/6.0;
278  }
279 
280  if (retVal < 0.0)
281  return 0.0;
282 
283  return (_maxVal != 0.0 && (retVal > _maxVal)) ? _maxVal : retVal;
