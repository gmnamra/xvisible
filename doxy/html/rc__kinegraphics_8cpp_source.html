260 void rcKinetoscope::visualFunctionMap (rcVisualSegmentCollection& segments,
261  const rcVisualStyle& baseStyle )
262 {
263  rmUnused( baseStyle );
264  // Reserve space for one graphics token per motion vector
265  segments.reserve( 1 + mPathMap.size() );
266 
267 
268  // If there are motion vectors draw them.
269  rcMotionPathMap::iterator pos;
270  for (pos = mPathMap.begin(); pos != mPathMap.end(); pos++)
271  {
272 
273  rc2Fvector p2(pos->second.anchor().x(),
274  pos->second.anchor().y());
275  rc2Fvector p1 (pos->second.position().x(),
276  pos->second.position().y());
277  rc2Fvector p21 (p1 - p2);
278  if (p21.isNull())
279  continue;
280 
281  p1 = xform().mapPoint (p1);
282  p2 = xform().mapPoint (p2);
283  rcVisualArrow arrow( p2, p1);
284  segments.push_back( arrow );
285 
286  }
287 
288 
289 
290  list<rcVisualFunction>::const_iterator cell;
291  for( cell = mFunctionMap.begin(); cell != mFunctionMap.end(); ++cell )
292  {
293  if (cell->isCardiacCell ())
294  {
295  rcAffineRectangle ar = cell->affys().back();
296  segments.push_back (affy);
297  rfAffineRectangleToSegmentsCollection (ar, segments);
298 
299  // if we have dimensios, draw projections
300  if (cell->have(rcVisualFunction::eDimensions))
301  {
302  // Draw Length Calipers
303  rcFPair maj = cell->majorEnds();
304  bool side = cell->isState (rcVisualFunction::eIsAffineXdirected);
305  int32 d = side ? ar.cannonicalSize().x() : ar.cannonicalSize().y();
306  rcFPair size ((double) d, (double) d);
307  maj = maj / size;
308 
309  if (side)
310  {
311  segments.push_back (caliper);
312  segments.push_back (rcVisualLine (ar.affineToImage(rc2Dvector(maj.x(), 0.33)),
313  ar.affineToImage(rc2Dvector(maj.x(), 0.67))));
314  segments.push_back (caliper);
315  segments.push_back (rcVisualLine (ar.affineToImage(rc2Dvector(maj.y(), 0.33)),
316  ar.affineToImage(rc2Dvector(maj.y(), 0.67))));
317  segments.push_back (caliper);
318  segments.push_back (rcVisualLine (ar.affineToImage(rc2Dvector(maj.x(), 0.5)),
319  ar.affineToImage(rc2Dvector(maj.y(), 0.5))));
320 
321  segments.push_back (caliper);
322  segments.push_back (rcVisualLine (ar.affineToImage(rc2Dvector(0.0, 0.5)),
323  ar.affineToImage(rc2Dvector(0.0, 0.5))));
324 
325 
326  }
327  else
328  {
329  segments.push_back (caliper);
330  segments.push_back (rcVisualLine (ar.affineToImage(rc2Dvector(0.33, maj.x())),
331  ar.affineToImage(rc2Dvector(0.67, maj.x()))));
332  segments.push_back (caliper);
333  segments.push_back (rcVisualLine (ar.affineToImage(rc2Dvector(0.33, maj.y())),
334  ar.affineToImage(rc2Dvector(0.67, maj.y()))));
335  segments.push_back (caliper);
336  segments.push_back (rcVisualLine (ar.affineToImage(rc2Dvector(0.5, maj.x())),
337  ar.affineToImage(rc2Dvector(0.5, maj.y()))));
338  segments.push_back (caliper);
339  segments.push_back (rcVisualLine (ar.affineToImage(rc2Dvector(0.5, 0.0)),
340  ar.affineToImage(rc2Dvector(0.5, 0.0))));
341 
342  }
343 
344  // Draw the last framerPerPeriod in the affine Rectangle
345  vector<float> lengthHistory ((uint32) cell->framesPerPeriod());
346  const deque<rcFPair>& imin = cell->interpolatedMin ();
347 
348  deque<rcFPair>::const_iterator start = (imin.size() < lengthHistory.size()) ? imin.begin() :
349  imin.end() - lengthHistory.size();
350  deque<rcFPair>::const_iterator finish = imin.end();
351  vector<float>::iterator lIter;
352  vector<float>::iterator historyStart = lengthHistory.begin() + (imin.size() < lengthHistory.size() ?
353  lengthHistory.size() - imin.size() : 0);
354 
355  for (lIter = historyStart; lIter < lengthHistory.end() && start < finish; lIter++, start++)
356  *lIter = start->y();
357 
358  // Place a rectangle at cell position
359  rcFRect ctr (cell->position().x() - cell->framesPerPeriod() / 2,
360  cell->position().y() - cell->framesPerPeriod() / 2,
361  cell->framesPerPeriod(), cell->framesPerPeriod());
362  segments.push_back (graph);
363  rfPlot1Dsignal (historyStart, lengthHistory.end(), ctr, segments, false);
364  }
365  }
366  else
367  {
368  rcPolygon pog = cell->polygons().back();
369 
370  // @note: add other visual codes for cell function
371  if (cell->isState (rcVisualFunction::eIsMoving))
372  {
373  segments.push_back (affy);
374  }
375 
376  // @note: add other visual codes for cell function
377  if (cell->isState (rcVisualFunction::eIsDivided))
378  {
379  segments.push_back (affylow);
380  }
381 
382  if (cell->isState (rcVisualFunction::eIsDividing))
383  {
384  segments.push_back (caliper2);
385  }
386 
387  rfPolygonToSegmentsCollection (pog, segments);
388 
389  }
390  }
