35  if (pfsspecImage == NULL) return (bdNamErr);

37  // Get the file size. We use this to uncover quicktime's inability to tell us if it is 8 or 16 bit monochrom
38  // @todo move to util or separate function.
39 
40  FSRef filerf;
41  OSErr osErr;

43  FSCatalogInfoBitmap whichInfo;
44  FSCatalogInfo catalogInfo;
45  whichInfo = kFSCatInfoNodeFlags;
46  whichInfo |= kFSCatInfoDataSizes;
47  whichInfo |= kFSCatInfoRsrcSizes;
48  osErr = FSGetCatalogInfo(&filerf, whichInfo, &catalogInfo, NULL, NULL,NULL);
49  fileSize = catalogInfo.dataLogicalSize + catalogInfo.rsrcLogicalSize;
50 
51 
52  // Get the graphics importer
53  GetGraphicsImporterForFile(pfsspecImage, &gi);
54 
55  result = GraphicsImportGetImageDescription(gi, &imageDescH);
56 
57  if( noErr != result || imageDescH == NULL )
58  {
59  return (bdNamErr);
60  }
61  desc = *imageDescH;
62  getPixelInfoFromImageDesc (imageDescH, pd, pf, isGray, isWhiteReversed, &ctb);
63 
64 #ifdef DEBUG_LOG
65  for (int i = 0; i < 256; i++)
66  fprintf (stderr, "[%d]: %d\t%d\t%d\n", i, (*ctb)->ctTable[i].rgb.Qt::red, (*ctb)->ctTable[i].rgb.Qt::green, (*ctb)->ctTable[i].rgb.Qt::blue);
67 #endif
68 
69  // Get size and format information we need
70  Rect boundsRect = { 0, 0, desc->height, desc->width };
71  imageWidth = desc->width;
72  imageHeight = desc->height;
73  overEightSize = imageWidth * imageHeight;
74  overEightSize += overEightSize / 10; // add 10 percent
75 
76  // Check to see if file size and pixel format / depth match
77 
78  if (pd == rcPixel8 && pf == k8IndexedGrayPixelFormat && overEightSize < fileSize)
79  {

82  }

85  // We will allocate GWorld memory ourselves as a framebuf ===> Dispose will not free this memory
86  try
87  {
88  rcSharedFrameBufPtr buf(new rcFrame (imageWidth, imageHeight, rcPixel (pd)));
89  buf->setIsGray(isGray);

110  }

113  goto bail;
114  }
115

121  }

138  // }

258 
259  if ( status == noErr ) {
260  ptr = image.frameBuf();
261  if ( mFrameInterval > 0.0 ) {

264  mCurrentTimeStamp += mFrameInterval;
265  }
266  ret = eFrameStatusOK;
267  setLastError( eFrameErrorOK );
268  } else {
