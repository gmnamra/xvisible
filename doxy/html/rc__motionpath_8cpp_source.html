278 void rcMotionVectorPath::measure ()
279 {
280  // Anchor is the center of the origin pixel. This is
281  // Independent of target sizes and gradient filter sizes
282  // MovingSize is 1 + radius * 2 of the search area -- hence divide by 2
283  //
284 
285  rcIPair anc (iAnchor ());
286  rcIPair span = sharedKin()->movingSize ();
287  rcIPair target = sharedKin()->target ();
288 
289  rcIRect fixed (anc.x() - target.x(), anc.y() - target.y(),
290  target.x()+target.x()+1, target.y()+target.y()+1);
291 
292  rcIRect moving (anc.x() - span.x(), anc.y() - span.y(),
293  span.x()+span.x()+1, span.y()+span.y()+1);
294 
295  if (sharedKin()->isWithin(fixed) && sharedKin()->isWithin(moving))
296  parabolicDisparity (fixed, moving);
297 
298 }

308 void rcMotionVectorPath::parabolicDisparity (const rcRect& fixed, const rcRect& moving)
309 {
310  float maxScore (0.0);
311  rcPair<bool> edge;
312  rc2Fvector interp;
313  rcCorr maxCorr;
314  vector<vector<float> > mSpace;
315 
316  // Size and create search space
317  const rcIPair searchSpace (moving.width() - fixed.width() + 1,
318  moving.height() - fixed.height() + 1);
319 
320  rc2Fvector origin (searchSpace);
321  rcIPair maxPos = searchSpace;
322  vector<float> score (rcKinetoscope::eSplatArea);
323  origin /= 2.0f;
324 
325 
326  bool fastCorr = sharedKin()->use4fixed().depth() == rcPixel8 &&
327  searchSpace.y() == 5 && searchSpace.x() == 5 &&
328  sharedKin()->isPointCorrOptimizationAvailable ();
329 
330  {
331  rcWindow fixedw (sharedKin()->use4fixed(), fixed);
332  rcWindow movingw (sharedKin()->use4moving(), moving);
333  rmAssert (fixedw.depth() == movingw.depth());
334  maxScore = match (fixedw, movingw, searchSpace, maxCorr, maxPos, mSpace);
335 
336 
337  edge.y() = !(maxPos.y() > 0 && maxPos.y() < (searchSpace.y() - 1));
338  edge.x() = !(maxPos.x() > 0 && maxPos.x() < (searchSpace.x() - 1));
339 
340  if (!edge.x() && !edge.y())
341  {
342  score[0] = mSpace[maxPos.y()-1][maxPos.x()-1];
343  score[1] = mSpace[maxPos.y()-1][maxPos.x()];
344  score[2] = mSpace[maxPos.y()-1][maxPos.x()+1];
345  score[3] = mSpace[maxPos.y()][maxPos.x()-1];
346  score[4] = mSpace[maxPos.y()][maxPos.x()];
347  score[5] = mSpace[maxPos.y()][maxPos.x()+1];
348  score[6] = mSpace[maxPos.y()+1][maxPos.x()-1];
349  score[7] = mSpace[maxPos.y()+1][maxPos.x()];
350  score[8] = mSpace[maxPos.y()+1][maxPos.x()+1];
351  }
352 
353 
354  if (!edge.x() && !edge.y())
355  {
356  static float * dnull (0);
357  interp = rc2Fvector (parabolicFit (score [3], score [4], score [5], dnull),
358  parabolicFit (score[1], score [4], score [7], dnull));
359  for (uint32 i = 0; i < score.size();i++)
360  {
361  mSplat[i] = uint8 (255 * score[i]);
362  }
363 
364  interp += rc2Fvector (maxPos.x()+0.5f, maxPos.y()+0.5f);
365  interp -= origin;
366  }
367 
368  // @note We do this here to avoid calling update
369  move (interp);
370 
371  }
372 }

403 void rcMotionVectorPath::reconcile ()
404 {
405  // When we implement pyramid processing, this
406  // will be added the logic to take the low resolution
407  // motion vectors and check them at a higher
408  // resolution
409  // For now it is assumed that we only have one
410  // level pyramid
411  // See how prependicular are ACF and motion. We want them as prependicular as possible
412  // Which means we want to dot product between the two vectors to be as close to 0 as possible
413  // Adding Velocity Fields
414  // rcFeature d16 (motion ());
415  // rcIPair ip = iAnchor ();
416  // mKine->velocityXfield().setPixel (ip.x(), ip.y(), (uint32) (d16.pos16().x().real()));
417  // mKine->velocityYfield().setPixel (ip.x(), ip.y(), (uint32) (d16.pos16().y().real()));
418 
419  if (motion() == rc2Fvector()) return;
420 
421  float l2 = motion().l2();
422 
423  // @note default setting
424  if (l2 < 0.0025) return;
425 
426  rcWindow& connect = mKine->connect();
427  const rcWindow& moving = mKine->use4moving();
428  rcIPair pt2 ((int32) (anchor().x() + motion().x()),
429  (int32) (anchor().y() + motion().y()));
430  rcIPair target (rcKinetoscope::eSplatSize / 2);
431  vector<uint8>::const_iterator sp = mSplat.begin();
432 
433 
434  for (int32 j = - target.y(); j <= target.y(); j++)
435  for (int32 i = - target.x(); i <= target.x(); i++)
436  {
437  rcIPair pt = pt2 + rcIPair (i, j);
438 
439  rmAssertDebug (connect.isWithin(pt));
440  uint8 *pp = (uint8 *) connect.pelPointer (pt.x(), pt.y());
441  uint8 *mm = (uint8 *) moving.pelPointer (pt.x(), pt.y());
442  *pp = (*sp++ * *mm++) >> 8;
443  }
444 
445 
446 }
