410 template<class T>
411 void rcRectangle<T>::normalize()
412 {
413  T w = width();
414  T h = height();
415 
416  if (w<0 || h<0) {
417  T x = mUpperLeft.x();
418  T y = mUpperLeft.y();
419  if (w < 0)
420  {
421  x += w;
422  w = -w;
423  }
424  if (h < 0)
425  {
426  y += h;
427  h = -h;
428  }
429 
430  mUpperLeft = rcPair<T>(x, y);
431  mLowerRight = rcPair<T>(x+w, y+h);
432  }
433 }

486 {
487  rcPair<T> overlapUpperLeft(rmMax(mUpperLeft.x(),other.mUpperLeft.x()),

616 template<class P>
617 rcIRect rfRoundRectangle (const rcRectangle<P>& fr)
618 {
619  static const int32 dummy (0);
620 
621  rcIPair iSize (rfRound (fr.width(), dummy), rfRound (fr.height(), dummy));
622 
623  int32 right, left, top, bottom;
624 
625 
626  if ((fr.ul().x() - (int32) (fr.ul().x())) < ((int32) fr.lr().x() + P (1) -fr.lr().x()))
627  {
628  left = (int32) fr.origin().x(); right = left + iSize.x();
629  }
630  else
631  {
632  right = (int32) fr.lr().x() +1; left = right - iSize.x();
633  }
634 
635  if ((fr.ul().y() - (int32) fr.ul().y()) < ((int32) fr.lr().y() + P (1) -fr.lr().y()))
636  {
637  top = rfRound (fr.ul().y(), dummy); bottom = top + iSize.y();
638  }
639  else
640  {
641  bottom = (int32) fr.lr().y() + 1; top = bottom - iSize.y();
642  }
643 
644  return rcIRect (left, top, right - left, bottom - top);
645 }
