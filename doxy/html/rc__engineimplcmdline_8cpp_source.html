79  string ACI ("aci"), VEN ("vis"), DIF ("dif"), CELL ("cell"), CARDIO ("myo");

81  string BODY ("body"), MOTION ("motion"), ALL ("all"), EXPOSURE ("exp"), DRG ("drg");
82  string DRGAUTOMASK ("drgautomask");
83  string NOP (""), TEMPLATE ("target");

94  {
95  rmExceptionMacro (<< "-analysis -result options are required" << endl);

97 
98  try
99  {
100  movie = cmd.getArgument ("-movie", 0);
101  }
102  catch (...)
103  {
104  cout << "Not a movie, is it a directory of tiffs \n";

106  {
107  tiffdir = cmd.getArgument ("-dirOfTiffs", 0);
108  }
109  catch (...)
110  {
111  rmExceptionMacro (<< "either movie or tiff directory options are required" << endl);
112  }
113  }
114 
115  // Get the optional stuff
116  // @note: some parameters are set using setSettingId and some directly in the engine
117  // the choice largely mirrors what takes place in rc_engineimpl.cpp
118 
119  report = cmd.getSafeArgument ("-report", 0, "csv");
120  cell = cmd.getSafeArgument ("-type", 0, "general");
121  graphics = cmd.getSafeArgument ("-graphics", 0, "no");
122  advanced = cmd.getSafeArgument ("-advanced", 0, "no");
123  exportImages = cmd.getSafeArgument ("-exportImages", 0, "");

128  _cLineEndFrame = rmMax (_cLineStartFrame, _cLineEndFrame);
129  _cLineRectTL = rcIPair (cmd.atoi(cmd.getSafeArgument("-TopLeft", 0, "-1")),
130  cmd.atoi(cmd.getSafeArgument("-TopLeft", 1, "-1")));
131  _cLineRectSize = rcIPair (cmd.atoi(cmd.getSafeArgument("-Size", 0, "-1")),
132  cmd.atoi(cmd.getSafeArgument("-Size", 1, "-1")));

134  rcValue maxTrackDisp (maxDisp);
135  int32 channel = cmd.atoi (cmd.getSafeArgument("-Channel", 0, "0"));
136  rcValue channelVal (int(channel));

139  rcValue cmdline (eCmd);

164

216
