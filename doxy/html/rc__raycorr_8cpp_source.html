25 {
26  int32 misses (0);
27  int32 dir = mPathWays.size();
28  rcIPair size = site.size();
29  rcIPair origin ((int32) (org.x() * size.x()),
30  (int32) (org.y() * size.y()));
31  static rcIPair nowhere (rcINT32_MAX, rcINT32_MAX);
32 
33  // Note that rowUpdate is the same between a window and its
34  // framebuf
35  const uint32& fixedRUP = site.rowUpdate();
36  const uint32& movingRUP = image.rowUpdate();
37  mDiffusion = 0.0;
38  rcIPair& rmax = nowhere;
39  int32 rMaxIndex (-1);
40  float rMax (0.0);
41 
42  if (dir == 0) return false;
43 
44  if (mPathWays[direction].size() == 0) return false;
45  if (direction >= dir) return false;
46 
47  vector<rcIPair>::const_iterator ip = mPathWays[direction].begin();
48 
49  // Walk the line and fill up the correlation space
50  int32 i (0);
51  for (; ip != mPathWays[direction].end(); ip++, i++)
52  {
53  rcIPair spot (*ip - origin + start);
54  if (image.isWithin (spot))
55  {
56  rcCorr res;
57  int32 inSide (0);
58  rcIRect rect (spot, spot+size);
59  rcWindow rw (image, rect, inSide);
60  misses += !inSide;
61  if (!inSide) continue;
62  rfCorrelate (rw.pelPointer(0,0),
63  site.pelPointer(0,0),
64  fixedRUP, movingRUP,
65  size.x(), size.y(),
66  res);
67  float r = (float) res.r();
68  mDiffusion += r;
69  mSpace.push_back (r);
70  if (r > rMax)
71  {
72  rMax = r;
73  rmax = *ip;
74  rMaxIndex = i;
75  }
76  }
77  }
78 
79  // If we do have a max, interpolate in the ray direction
80  // Size of correlation space defines existence of resutls.
81  float ds (0.0);
82  int32 s = mSpace.size();
83 
84  if (s > 1)
85  {
86  // at least 3 and max not at the borders
87  if (s > 2 && rMaxIndex > 0 && rMaxIndex < (s - 1))
88  {
89  static float *fdum (0);
90 
91  ds = parabolicFit (mSpace[rMaxIndex - 1],
92  mSpace[rMaxIndex],
93  mSpace[rMaxIndex + 1], fdum);
94  }
95  else // at least 2 and Max at the borders
96  {
97  ds = mSpace[s - 2] / (mSpace[s-1] + mSpace[s-2]);
98  }
99  }
100 
101  mBest = rc2Fvector (mR + ds, (rk2PI * direction) / mDirs);
102 
103  return (bool (s));
104 }

121  for (int32 i = 0; i < dirQ; i++)
122  {
123  rc2Fvector point (mR, (rk2PI * i) / mDirs);
124  rcIPair ipoint ((point.x() < 0 ? rfRoundNeg (point.x(), dummy) :
125  rfRoundPlus (point.x(), dummy)),
126  (point.y() < 0 ? rfRoundNeg (point.x(), dummy) :

135  dy = y2-y1; ay = rmABS(dy)<<1; sy = rmSGN(dy);
136  x = x1;
137  y = y1;
138  if (ax>ay) { /* x dominant */
