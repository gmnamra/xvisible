553 // Create a 8 bit gray scale image from an image stored in a color format
554 static void _rfRcWindow32to8(const rcWindow& rgbInput, rcWindow& channelOutput, rcChannelConversion opt)
555 {
556  rmAssert( rgbInput.width() == channelOutput.width() );
557  rmAssert( rgbInput.height() == channelOutput.height() );
558  rmAssert( rgbInput.depth() == rcPixel32 );
559 
560  const uint32 width = rgbInput.width();
561  const uint32 height = rgbInput.height();
562 
563  // TODO: use AltiVec for this
564  // Separate color and gray cases for speed
565 
566  if ( rgbInput.isGray() ) {
567  // Gray scale image
568 
569  for (uint32 j = 0; j < height; j++)
570  {
571  uint32 *rgbRow = (uint32 *) rgbInput.rowPointer (j);
572  uint8 *oneRow = channelOutput.rowPointer (j);
573 
574  for (uint32 i = 0; i < width; i++, rgbRow++, oneRow++)
575  {
576  const uint8* rgbComponents = (uint8*) rgbRow;
577  // ARGB or RGBA we are assuming all channels are the same
578  *oneRow = rgbComponents[1];
579  }
580  }
581  }
582  else {
583  // Color image
584  for (uint32 j = 0; j < height; j++)
585  {
586  uint32 *rgbRow = (uint32 *) rgbInput.rowPointer (j);
587  uint8 *oneRow = channelOutput.rowPointer (j);
588 
589  for (uint32 i = 0; i < width; i++, rgbRow++, oneRow++)
590  {
591  const uint8* rgbComponents = (uint8*) rgbRow;
592 
593  switch (opt)
594  {
595  case rcSelectAverage:
596  *oneRow = rfGray( rgbComponents[1], rgbComponents[2], rgbComponents[3] );
597  break;
598  case rcSelectRed:
599  *oneRow = rgbComponents[1];
600  break;
601  case rcSelectGreen:
602  *oneRow = rgbComponents[2];
603  break;
604  case rcSelectBlue:
605  *oneRow = rgbComponents[3];
606  break;
607  case rcSelectMax:
608  *oneRow = rmMaxOf3 ( rgbComponents[1], rgbComponents[2], rgbComponents[3] );
609  case rcSelectAll:
610  default:
611  // Cannot keep all channels when going from 32 to 8
612  rmAssert( 0 );
613  }
614  }
615  }
616  }
