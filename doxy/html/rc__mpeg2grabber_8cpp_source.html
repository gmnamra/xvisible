105  // Generate a reifymovie from this mpeg file
106 bool rcMPEG2Grabber::getReifyMovie ( const std::string& fName, const rcChannelConversion& conv,
107  const movieFormatRev& rev, bool overWrite, const float& fInt,
108  const int32 sample)
109 {
110  lock();
111 
112  setLastError( eFrameErrorUnknown );
113  rmAssert (sample > 0 && sample < 9999); // some sanity checks on sample
114 
115  struct fbuf_s * current_fbuf;
116  const movieOriginType origin = movieOriginConversionExt;
117 
118  rcGenMovieFileError error = eGenMovieFileErrorOK;
119 
120  //@note the frame rate is set to 30 fps
121  rcGenMovieFile movie(fName, origin, conv, cCreator, rev, overWrite, fInt);
122  rmAssert( movie.valid() ); // @note replace with throw
123 
124  // Add an experiment header
125  rcMovieFileExpExt expHdr = generateHeader();
126  expHdr.title( "MPEG2Grabber" );
127 
128  do
129  {
130  mState = mpeg2_parse (mDecoder);
131  switch (mState)
132  {
133  case STATE_BUFFER:
134  mSize = fread (mBuffer, 1, rcMPEG2Grabber::buffer_size, mFilePtr);
135  mpeg2_buffer (mDecoder, mBuffer, mBuffer + mSize);
136  break;
137  case STATE_SEQUENCE:
138  mpeg2_convert (mDecoder, mpeg2convert_rgb24, NULL);
139  for (int32 i = 0; i < 3; i++)
140  {
141  fbuf[i].rgb[0] = (uint8_t *) malloc (3 * mInfo->sequence->width * mInfo->sequence->height);
142  fbuf[i].rgb[1] = fbuf[i].rgb[2] = NULL;
143  if (!fbuf[i].rgb[0])
144  {
145  fprintf (stderr, "Could not allocate an output buffer.\n");
146  exit (1);
147  }
148  fbuf[i].used = 0;
149  }
150  for (int32 i = 0; i < 2; i++)
151  {
152  current_fbuf = get_fbuf ();
153  mpeg2_set_buf (mDecoder, current_fbuf->rgb, current_fbuf);
154  }
155  break;
156  case STATE_PICTURE:
157  current_fbuf = get_fbuf ();
158  mpeg2_set_buf (mDecoder, current_fbuf->rgb, current_fbuf);
159  break;
160  case STATE_SLICE:
161  case STATE_END:
162  case STATE_INVALID_END:
163  if (mInfo->display_fbuf)
164  {
165  if (! ( frameCount() % sample ) )
166  movie.addFrame ( add_frame (mInfo->sequence->width, mInfo->sequence->height, mInfo->display_fbuf->buf[0]));
167  mFrameCount++;
168  }
169 
170  if (mInfo->discard_fbuf)
171  ((struct fbuf_s *)mInfo->discard_fbuf->id)->used = 0;
172  if (mState != STATE_SLICE)
173  for (int32 i = 0; i < 3; i++)
174  free (fbuf[i].rgb[0]);
175  break;
176  default:
177  break;
178  }
179  } while (mSize);
180 
181  setLastError( eFrameErrorOK );
182  // Add header and flush
183  movie.addHeader( expHdr );
184  error = movie.flush();
185  unlock();
186  return ( error == eGenMovieFileErrorOK );
187 }

217 rcWindow rcMPEG2Grabber::add_frame (int width, int height, uint8_t * data24)
218 {
219  const int32 bytesInPixel = 3;
220  rcWindow frame (width, height, rcPixel32);
221  for (int32 j = 0; j < height; j++)
222  {
223  uint32 *fourbytepels = (uint32 *) frame.rowPointer (j);
224  for (int32 i = 0; i < width; i++, data24+=bytesInPixel)
225  {
226 #ifdef __ppc__
227  *fourbytepels++ = rfRgb (data24[0],data24[1],data24[2]);
228 #else
229  uint32 tmp = rfRgb (data24[2],data24[1],data24[0]);
230  *fourbytepels++ = ( tmp & 0xff000000 ) | ( tmp & 0xff ) << 16 |
231  ( tmp & 0xff00 ) | ( tmp & 0xff0000 ) >> 16;
232 #endif
233  }
234  }
235  return frame;
236 }
