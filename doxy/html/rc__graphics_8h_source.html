29  // Type of graphics segment
30  enum rcVisualSegmentType {
31  eUnknown = 0, // Sentinel, causes an assert failure in drawing code
32  eEmpty, // Empty NOP to break a line strip or loop
33  ePoints, // Draw 2 points
34  eLine, // Draw 1 line segment
35  eLineStrip, // Draw 1 line segment connected to previous and next segment
36  eLineLoop, // Draw eLineStrip plus last point connected to first point
37  eArrow, // Draw 1 line with an arrow head
38  eRect, // Draw 1 unfilled rectangle
39  eEllipse, // Draw 1 unfilled ellipse (circle is a special case)
40  eCross, // Draw 1 cross
41  eStyle, // Specify drawing style, draw nothing
42  eLast // Sentinel, causes an assert failure in drawing code
43  };

78 //
79 // All graphics classes derived from rcVisualSegment have the following properties:
80 //
81 // 1. They utilize the same data members as base class.
82 // 2. They do not have data members of their own. This guarantees that
83 // derived classes can be stored in STL containers without risk of
84 // slicing.
85 // 3. They can have new ctors and accessors/mutators to improve ease
86 // of use.
87 // 4. Each derived class defines the semantics of data in mP1 and mP2.
