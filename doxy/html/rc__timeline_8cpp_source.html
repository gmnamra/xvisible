743  prevX = x;
744  }
745  }
746  }

871 // Paint vertical scale marks
872 void rcTimeline::paintScalarScaleMarks( QPainter& p,
873  rcTrackRenderInfo& tri )
874 {
875  rcScalarTrack* scalarTrack = dynamic_cast<rcScalarTrack*>( tri._track );
876  if (scalarTrack == 0)
877  return;
878 
879  double minValue = tri._minValue;
880  double maxValue = tri._maxValue;
881  if ( minValue == maxValue )
882  return;
883 
884  double incr = (maxValue - minValue) / 10.0;
885  // If incr <= 0, we'll loop indefinitely.
886  if ( incr <= 0.0 )
887  return;
888 
889  // Save current painter state
890  p.save();
891  // Draw in the margin
892  QRect cr = rect();
893  QRect dr = reducedContentsRect();
894  int x = cr.right();
895  // Use new pen
896  p.setPen( tri._tickMarkPen );
897  p.setFont( tri._tickMarkFont );
898 
899  // Draw full-width ticks and numbers
900  double value;
901  const char* formatString = tri._track->getDisplayFormatString();
902  int32 valueOffset = getValueOffset();
903  double valueScale = getValueScale( tri );
904 
905  // Draw min tick
906  int y = valueToPixel( minValue, tri, valueScale, valueOffset );
907  int textX = dr.right() + 4;
908  paintScalarScaleMark( p, formatString, minValue, x, y, textX, y + 4 );
909  // Draw max tick
910  y = valueToPixel( maxValue, tri, valueScale, valueOffset );
911  paintScalarScaleMark( p, formatString, maxValue, x, y, textX, y + 4 );
912  // Draw remaining ticks
913  for ( value = minValue + incr; value < maxValue; value += incr ) {
914  y = valueToPixel( value, tri, valueScale, valueOffset );
915  paintScalarScaleMark( p, formatString, value, x, y, textX, y + 4 );
916  }
917 
918  // Draw half-width ticks
919  for ( value = minValue + incr/2; value < maxValue; value += incr ) {
920  int y = valueToPixel( value, tri, valueScale, valueOffset );
921  p.drawLine( x, y, x - cScaleTickWidth/2, y );
922  }
923 
924  // Restore painter state
925  p.restore();
926 }
