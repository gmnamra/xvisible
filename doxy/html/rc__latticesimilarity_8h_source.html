75 template<class Iterator>
76 bool rcLatticeSimilarator::fill(Iterator starting, Iterator afterLast)
77 {
78  typedef typename std::iterator_traits<Iterator>::value_type value_type;
79 
80  if (mTemporal != distance (starting, afterLast)) return false;
81 
82  int32 width, height;
83  value_type image (*starting);
84  width = image.width(); height = image.height();
85  rcIPair tileSize (defaultTileSize, defaultTileSize);
86 
87  // @note overlapping tiles. Amount of overlap is determined by step size.
88  // if mLrad is set to defaultTileSize divided by 2, then neighboring tiles overlap 50 percent
89  // that is 100 - mLrad / defaultTileSize.
90  int32 rows ((height - defaultTileSize) / mLrad);
91  int32 columns ((width - defaultTileSize) / mLrad);
92 
93  if (rows <= 0 || columns <=0) return false;
94  mSize = rcIPair (columns, rows);
95  mImageSize = rcIPair (width, height);
96  mDepth = image.depth();
97 
98  // Allocate an array of tile rectangles
99  mTiles.resize (rows);
100  mSims.resize (rows);
101 
102 
103  int32 tly (0);
104 
105  for (int32 j = 0; j < rows; j++)
106  {
107  mTiles[j].resize (columns);
108  mSims[j].resize (columns);
109 
110  int32 tlx (0);
111  for (int32 i = 0; i < columns; i++, tlx += mLrad)
112  {
113  rcIPair tl (tlx, tly);
114  mTiles[j][i] = rcIRect (tl, tl + tileSize);
115  }
116  tly += mLrad;
117  }
118 
119  bool rtn (false);
120  for (int32 j = 0; j < rows; j++)
121  for (int32 i = 0; i < columns; i++)
122  {
123  vector<rcWindow> wins;
124 
125  Iterator w = starting;
126  while (w < afterLast)
127  {
128  value_type img (*w);
129  rtn = (img.size() == mImageSize);
130  if (rtn == false) return rtn;
131 
132 
133  static const int32 inside (0);
134  rcWindow tile (img, mTiles[j][i], inside);
135  wins.push_back (tile);
136  w++;
137  }
138 
139  mSims[j][i] = boost::shared_ptr<rcSimilarator> (new rcSimilarator (rcSimilarator::eExhaustive,
140  mDepth, mTemporal, mTemporal * 2));
141  rtn = (mSims[j][i] && mSims[j][i]->fill (wins));
142  if (rtn == false) return rtn;
143  rmAssert (mSims[j][i]->longTermCache () == false);
144  rtn = mSims[j][i]->longTermCache (true);
145  if (rtn == false) return rtn;
146  }
147 
148  if (rtn)
149  {
150  vector<rcWindow> wins;
151 
152  Iterator w = starting;
153  while (w < afterLast)
154  {
155  value_type img (*w);
156  rtn = (img.size() == mImageSize);
157  if (rtn == false) return rtn;
158 
159  wins.push_back (img);
160  w++;
161  }
162 
163  mWholeSim = boost::shared_ptr<rcSimilarator> (new rcSimilarator (rcSimilarator::eExhaustive,
164  mDepth, mTemporal, mTemporal * 2));
165 
166  rtn = (mWholeSim && mWholeSim->fill (wins));
167  if (rtn == false) return rtn;
168  rmAssert (mWholeSim->longTermCache () == false);
169  rtn = mWholeSim->longTermCache (true);
170  if (rtn == false) return rtn;
171  }
172 
173  return rtn;
174 }
