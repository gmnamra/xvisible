42 rcTails::rcTails(const vector<uint32>& histogram,
43  uint32 leftCutoff,
44  uint32 rightCutoff)
45  : mOffset(0), mCumulativeOffset(0)
46 {
47  uint32 count = 0;
48 
49  // go up from the left end
50  if(leftCutoff == 0) {
51  mLeftTail = histogram.begin();
52  }
53  else {
54  vector<uint32>::const_iterator p = histogram.begin();
55  for(count = 0; (count += *p) < leftCutoff; p++);
56  mLeftTail = ++p;
57  mCumulativeOffset = count;
58  }
59 
60  // down from the right end
61  if(rightCutoff == 0) {
62  mRightTail = histogram.end() - 1;
63  }
64  else {
65  vector<uint32>::const_iterator p = histogram.end() - 1;
66  for(count = 0; (count += *p) < rightCutoff; p--);
67  mRightTail = --p;
68  }
69 
70  rmAssert (distance (histogram.begin(), mLeftTail) < distance (histogram.begin(), mRightTail));
71 }

73 void
74 rcHornMoments::compute( vector<uint32>::const_iterator leftTail,
75  vector<uint32>::const_iterator rightTail)
76 {
77  // we shift the interval [left tail, right tail] to [0, right - left]
78  double m0 = 0.0, m1 = 0.0, m2 = 0.0;
79 
80  // Horn Accumulator
81  // the left tail is intentionally excluded
82  for( vector<uint32>::const_iterator p = rightTail;
83  p > leftTail;
84  p--){
85  m0 += *p;
86  m1 += m0;
87  m2 += m1;
88  }
89 
90  // since we assume left tail == 0, we do not have to add i * hist[i] to m1
91  mZeroth = m0 + *leftTail; // number of sample
92  mFirst = m1; // mean * number of sample
93  mSecond = m2 * 2 - m1; // var = second_/zeroth_ - (first_/zeroth_)^2
94 }

128 void
129 rcWithinGroupInv::compute( vector<uint32>::const_iterator leftTail,
130  vector<uint32>::const_iterator rightTail,
131  uint32 offset, uint32 cumulativeOffset,
132  rcLevelSets& results)
133 {
134  // Between Group Variance (BGV) is defined as:
135  // BGV = (m0_l/m0) * (m0_r/m0) * (m1_l/m0_l - m1_r/m0_r)^2
136  // oops, the above variable names are not quite descriptive
137  rcHornMoments moment(leftTail, rightTail);
138 
139  double zerothMomentLeft = 0.0;
140  double zerothMomentRight = 0.0;
141  double firstMomentLeft = 0.0;
142  double firstMomentRight = 0.0;
143 
144  double max = 0.0;
145  double cumulative = 0.0;
146  uint32 threshold = 0;
147  uint32 i = 0; // index to the (partial) histogram
148  vector<uint32>::const_iterator p = leftTail;
149  for(; p < rightTail; p++, i++) {
150  if(*p == 0) continue;
151  zerothMomentLeft += *p;
152  zerothMomentRight = moment.zeroth() - zerothMomentLeft;
153  firstMomentLeft += *p * i;
154  firstMomentRight = moment.first() - firstMomentLeft;
155  double dm = firstMomentRight * zerothMomentLeft -
156  firstMomentLeft * zerothMomentRight;
157  double score = dm * dm / (zerothMomentLeft * zerothMomentRight);
158 
159  if(score > max) {
160  threshold = i;
161  cumulative = zerothMomentLeft;
162  max = score;
163  }
164  }
165 
166  double denom = moment.second()*moment.zeroth()-moment.first()*moment.first();
167  results.update(threshold + offset + 1, max/denom, (int32) cumulative);
168 }
