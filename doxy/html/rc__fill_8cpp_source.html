309 static double
310 polygon_area (const polygon * poly /* array of pointers to input
311  vertices */
312 )
313 /* Returns the area bounded by <poly>. Assumes convex polygon. This method is
314  from standard CRC analytic geometry. */
315 {
316  poly_segment *s1;
317  poly_segment *s2;
318  double area = 0; /* return value */
319  int32 n;
320 
321  if (poly->num_segs < 3)
322  return 0;
323 
324  s1 = poly->segs;
325  s2 = s1->next;
326  for (n = poly->num_segs; n > 0; --n) {
327 
328  area += s1->x * s2->y - s1->y * s2->x;
329  s1 = s2;
330  s2 = s2->next;
331  }
332  return 0.5 * area;
333 }

338 static void
339 coords_to_poly (int32 num_pts, /* number of polygon vertices */
340  double *coords, /* coord array: x1, y1, x2, y2, ... */
341  polygon * poly, /* resulting polygon */
342  int32 max_row, /* maximum row */
343  int32 max_col /* maximum column */
344 )
345 /* Converts a list of vertex coordinates to an internal polygon data
346  structure. The resulting polygon wraps in the positive direction. */
347 {
348  poly_segment *seg; /* loop ptr */
349  int32 n; /* loop index */
350  double *c; /* ptr to current coordinate */
351  double area; /* polygon area */
352 
353  assert (num_pts <= poly->max_segs);
354 
355  /* Initialize subject polygon: */
356 
357  poly->num_segs = num_pts;
358  seg = poly->segs;
359  c = coords;
360 
361  /* Form a circular linked list with the vertex coordinates: */
362 
363  for (n = poly->num_segs; n > 0; --n, ++seg) {
364  seg->x = *c++;
365  seg->y = *c++;
366  seg->ix = (int32) floor (seg->x);
367  seg->iy = (int32) floor (seg->y);
368 
369  seg->vertex_in_image = TRUE;
370  if (seg->ix < 0
371  || seg->ix > max_col
372  || seg->iy < 0
373  || seg->iy > max_row) {
374  seg->vertex_in_image = FALSE;
375  }
376  if (n == 1)
377  seg->next = poly->segs;
378  else
379  seg->next = seg + 1;
380  }
381 
382  /* Compute the overall polygon wrapping direction. The segment vectors
383  must rotate plus or minus 360 degrees for one complete boundary
384  traversal. We check the sign of the area as determined by the cross
385  products to determine direction. This works for an arbitrary,
386  non-convex polygon. */
387 
388  area = polygon_area (poly);
389 
390  poly->bndry_wraps_pos = (area > 0);
391 
392  /* If the polygon wraps in the negative direction, reverse the segment
393  order. The routine clip_polygon_to_pel() requires the polygon to wrap
394  in the positive direction. */
395 
396  if (area < 0.0) {
397  poly->segs->next = &poly->segs[poly->num_segs - 1];
398  for (n = poly->num_segs - 1, seg = poly->segs + 1;
399  n > 0; --n, ++seg) {
400  seg->next = seg - 1;
401  }
402  }
403  poly->bndry_wraps_pos = TRUE;
404 
405  /* Now initialize the rest: */
406 
407  poly->min_col = poly->min_row = rcINT32_MAX;
408  poly->max_col = poly->max_row = 0;
409 
410  for (n = poly->num_segs, seg = poly->segs; n > 0; --n, seg = seg->next) {
411 
412  seg->vx = seg->next->x - seg->x;
413  seg->vy = seg->next->y - seg->y;
414 
415  /* Determine the slope of the segment: */
416 
417  if (seg->vy == 0.0)
418  seg->slope = SLOPE_HORZ;
419  else if (seg->vx == 0.0)
420  seg->slope = SLOPE_VERT;
421  else if ((seg->vx >= 0.0) ^ (seg->vy > 0.0))
422  seg->slope = SLOPE_NEG;
423  else
424  seg->slope = SLOPE_POS;
425 
426  /* Set the magnitudes of dx and dy so they reflect the change in
427  coordinate for an integral step in pixel coordinate. Set the
428  signs of dx and dy so they reflect the increase or decrease of the
429  intercepts as the pel row or column is incremented: */
430 
431  switch (seg->slope) {
432  case SLOPE_HORZ:
433  seg->dx = 1;
434  seg->dy = 0;
435  break;
436 
437  case SLOPE_VERT:
438  seg->dx = 0;
439  seg->dy = 1;
440  break;
441 
442  case SLOPE_POS:
443  seg->dx = fabs (seg->vx / seg->vy);
444  seg->dy = 1.0 / seg->dx;
445  break;
446 
447  case SLOPE_NEG:
448  seg->dx = -fabs (seg->vx / seg->vy);
449  seg->dy = 1.0 / seg->dx;
450  break;
451  }
452 
453  /* Compute the row and column limits containing the segment: */
454 
455 #define FLOOR(A) ((int32) floor(A))
456 
457  if (seg->y > seg->next->y) {
458  seg->min_row = FLOOR (seg->next->y);
459  seg->max_row = FLOOR (seg->y);
460  } else {
461  seg->min_row = FLOOR (seg->y);
462  seg->max_row = FLOOR (seg->next->y);
463  }
464  if (seg->x > seg->next->x) {
465  seg->first_col = FLOOR (seg->next->x);
466  seg->last_col = FLOOR (seg->x);
467  } else {
468 
469  seg->first_col = FLOOR (seg->x);
470  seg->last_col = FLOOR (seg->next->x);
471  }
472  if (seg->first_col < poly->min_col)
473  poly->min_col = seg->first_col;
474  if (seg->last_col > poly->max_col)
475  poly->max_col = seg->last_col;
476 
477  if (seg->min_row < poly->min_row)
478  poly->min_row = seg->min_row;
479  if (seg->max_row > poly->max_row)
480  poly->max_row = seg->max_row;
481 
482 #undef FLOOR
483 
484 #if 0
485  seg->name[0] = poly->num_segs - n + '1';
486  if (n == 1)
487  seg->name[1] = poly->segs->name[0];
488  else
489  seg->name[1] = seg->name[0] + 1;
490  seg->name[2] = '\0';
491 #endif
492 
493  }
494 
495  /* Determine the segment's boundary type. The type is used to decide in
496  which direction lies the polygon interior. This algorithm compares the
497  segments's direction, measured by both slope and vector direction,
498  with the overall direction that the boundary wraps to enclose the
499  polygon (the change in segment vector direction for one complete wrap
500  must be plus or minus 360 degrees). This algorithm works for
501  non-convex polygons. */
502 
503  for (n = poly->num_segs, seg = poly->segs; n > 0; --n, seg = seg->next) {
504 
505  switch (seg->slope) {
506  case SLOPE_HORZ:
507  if (seg->vx <= 0.0)
508  seg->type = BNDRY_TYPE_BOTTOM_LEFT;
509  else
510  seg->type = BNDRY_TYPE_TOP_RIGHT;
511  break;
512 
513  case SLOPE_VERT:
514  if (seg->vy <= 0.0)
515  seg->type = BNDRY_TYPE_TOP_LEFT;
516  else
517  seg->type = BNDRY_TYPE_BOTTOM_RIGHT;
518  break;
519 
520  case SLOPE_POS:
521  if (seg->vx <= 0.0)
522  seg->type = BNDRY_TYPE_BOTTOM_LEFT;
523  else
524  seg->type = BNDRY_TYPE_TOP_RIGHT;
525  break;
526 
527  case SLOPE_NEG:
528  if (seg->vx <= 0.0)
529  seg->type = BNDRY_TYPE_BOTTOM_RIGHT;
530  else
531  seg->type = BNDRY_TYPE_TOP_LEFT;
532  break;
533  }
534 
535  seg->isect_outside = blisect_outside_tab[seg->type];
536  }
537 }

541 /* Polygon clipping routine. This is used to compute the insersection of the
542  polygon and the corner pixels. We then take the area of this polygon to
543  determine its gray value. */
544 
545 static void
546 clip_polygon_to_pel (polygon * subj_poly, /* polygon to clip */
547  double pel_x, /* pel X coord */
548  double pel_y, /* pel Y coord */
549  polygon * new_poly /* resulting polygon */
550 )
551 /* Produces a clipped polygon in <new_poly> from <poly>, specified by a list
552  of points <poly>. The clipping region is pel_x <= x <= pel_x + 1 and pel_y
553  <= y <= pel_y + 1.
554 
555 This function clips against each of the four pel boundaries in turn,
556  successively splicing segments into the subject polygon as needed. We take
557  advantage of the following REQUIREd assumptions:
558 
559 The first point in <subj_poly> is assumed to be inside the pel boundaries.
560 
561 <new_poly> must be allocated by the caller with room for four more segments
562  than <subj_poly>.
563 
564 <subj_poly> must wrap in the positive direction.
565 
566 <subj_poly> is convex. Therefore, there can be only one resulting clipped
567  polygon. */
568 {
569  poly_segment *seg; /* current segment ptr */
570  poly_segment *new_segs; /* place to add new segments */
571  poly_segment *nseg; /* next-segment ptr */
572  poly_segment *seg1; /* first segment ptr */
573  poly_segment *seg2; /* secopd segment ptr */
574  int32 n; /* loop index */
575  double pel_x_p1; /* precomputation */
576  double pel_y_p1; /* precomputation */
577 
578  /* Validate input: */
579 
580  assert (subj_poly->bndry_wraps_pos);
581 
582  /* Make sure we have space for 4 segments. This is the most we'll add: */
583 
584  assert (new_poly->max_segs - subj_poly->num_segs >= 4);
585 
586  /* Simple initializations: */
587 
588  pel_x_p1 = pel_x + 1.0;
589  pel_y_p1 = pel_y + 1.0;
590 
591  /* Initialize output polygon from the subject polygon: */
592 
593  new_poly->bndry_wraps_pos = TRUE;
594  new_poly->num_segs = 0;
595  seg = new_poly->segs;
596  seg1 = subj_poly->segs;
597 
598  while (1) {
599 
600  *seg = *seg1;
601  ++new_poly->num_segs;
602 
603  seg->next = seg + 1;
604 
605  seg1 = seg1->next;
606  if (seg1 == subj_poly->segs)
607  break;
608  seg = seg->next;
609  }
610  seg->next = new_poly->segs;
611 
612  /* Initialize x and y intercepts: */
613 
614  for (n = new_poly->num_segs, seg = new_poly->segs;
615  n > 0;
616  --n, seg = seg->next) {
617 
618  switch (seg->slope) {
619  case SLOPE_HORZ:
620  seg->x1 = seg->x;
621  seg->x2 = seg->next->x;
622  seg->y1 = seg->y2 = seg->y;
623  break;
624 
625  case SLOPE_VERT:
626  seg->x1 = seg->x2 = seg->x;
627  seg->y1 = seg->y;
628  seg->y2 = seg->next->y;
629  break;
630 
631  default:
632  seg->x1 = seg->x + (pel_y - seg->y) * seg->dx;
633  seg->y1 = seg->y + (pel_x - seg->x) * seg->dy;
634  seg->x2 = seg->x1 + seg->dx;
635  seg->y2 = seg->y1 + seg->dy;
636  break;
637  }
638  }
639 
640  /* Initialize space to add new segments: */
641 
642  new_segs = &new_poly->segs[new_poly->num_segs];
643 
644 #if DEBUG
645  if (diag_flags & DIAG_POLY_CLIP) {
646  printf ("\nStarting Polygon:\n");
647  dump_poly (new_poly);
648  }
649 #endif
650 
651  /* TOP EDGE CLIP. */
652  /* Find the first left segment: */
653 
654  for (seg = new_poly->segs; seg->type & LEFT_BNDRY; seg = seg->next);
655  for (; !(seg->type & LEFT_BNDRY); seg = seg->next);
656 
657  /* Check for intersections of the top pel line with left boundary
658  segments. Use the first one we encounter: */
659 
660  seg1 = seg2 = NULL;
661  while (seg->type & LEFT_BNDRY) {
662  nseg = seg->next;
663  if ((seg->y <= pel_y && nseg->y >= pel_y)
664  || (seg->y >= pel_y && nseg->y <= pel_y)) {
665  seg1 = seg;
666  break;
667  }
668  seg = nseg;
669  }
670  /* If we have a left intersection, we must have a right intersection.
671  Find the last right segment that intersects the top: */
672 
673  if (seg1) {
674  for (; seg->type & LEFT_BNDRY; seg = seg->next);
675  while (!(seg->type & LEFT_BNDRY)) {
676  nseg = seg->next;
677  if ((seg->y <= pel_y && nseg->y >= pel_y)
678  || (seg->y >= pel_y && nseg->y <= pel_y)) {
679  seg2 = seg;
680  }
681  seg = nseg;
682  }
683  assert (seg2);
684 
685  /* Make the new segment, and splice it into the polygon: */
686 
687  nseg = new_segs++;
688  nseg->x = seg1->x1;
689  nseg->y = nseg->y1 = nseg->y2 = pel_y;
690  nseg->x1 = pel_x;
691  nseg->x2 = pel_x_p1;
692  nseg->type = BNDRY_TYPE_TOP_LEFT;
693 
694 #if DEBUG
695  nseg->y1_start = nseg->y2;
696  nseg->slope = SLOPE_HORZ;
697  strcpy (nseg->name, "-T");
698 #endif
699 
700  seg2->x = seg2->x1;
701  seg2->y = pel_y;
702 
703  seg1->next = nseg;
704  nseg->next = seg2;
705  new_poly->segs = seg1;
706 
707 #if DEBUG
708  if (diag_flags & DIAG_POLY_CLIP_FINE) {
709  printf ("\nAfter top clipping:\n");
710  printf ("seg1 - seg2: %2s - %2s\n", seg1->name, seg2->name);
711  dump_poly (new_poly);
712  }
713 #endif
714  }
715  /* BOTTOM EDGE CLIP. */
716  /* Find the first right segment: */
717 
718  for (seg = new_poly->segs; !(seg->type & LEFT_BNDRY); seg = seg->next);
719  for (; seg->type & LEFT_BNDRY; seg = seg->next);
720 
721  /* Check for intersections of the bottom pel line with right boundary
722  segments. Use the first one we encounter: */
723 
724  seg1 = seg2 = NULL;
725  while (!(seg->type & LEFT_BNDRY)) {
726  nseg = seg->next;
727  if ((seg->y <= pel_y_p1 && nseg->y >= pel_y_p1)
728  || (seg->y >= pel_y_p1 && nseg->y <= pel_y_p1)) {
729  seg1 = seg;
730  break;
731  }
732  seg = nseg;
733  }
734  /* If we have a right intersection, we must have a left intersection.
735  Find the last left segment that intersects the bottom: */
736 
737  if (seg1) {
738  for (; !(seg->type & LEFT_BNDRY); seg = seg->next);
739  while (seg->type & LEFT_BNDRY) {
740  nseg = seg->next;
741  if ((seg->y <= pel_y_p1 && nseg->y >= pel_y_p1)
742  || (seg->y >= pel_y_p1 && nseg->y <= pel_y_p1)) {
743  seg2 = seg;
744  }
745  seg = nseg;
746  }
747  assert (seg2);
748 
749  /* Make the new segment, and splice it into the polygon: */
750 
751  nseg = new_segs++;
752  nseg->x = seg1->x2;
753  nseg->y = nseg->y1 = nseg->y2 = pel_y_p1;
754  nseg->x1 = pel_x;
755  nseg->x2 = pel_x_p1;
756  nseg->type = BNDRY_TYPE_BOTTOM_RIGHT;
757 
758 #if DEBUG
759  nseg->y1_start = nseg->y2;
760  nseg->slope = SLOPE_HORZ;
761  strcpy (nseg->name, "-B");
762 #endif
763 
764  seg2->x = seg2->x2;
765  seg2->y = pel_y_p1;
766 
767  seg1->next = nseg;
768  nseg->next = seg2;
769  new_poly->segs = seg1;
770 
771 #if DEBUG
772  if (diag_flags & DIAG_POLY_CLIP_FINE) {
773  printf ("\nAfter bottom clipping:\n");
774  printf ("seg1 - seg2: %2s - %2s\n", seg1->name, seg2->name);
775  dump_poly (new_poly);
776  }
777 #endif
778  }
779  /* RIGHT EDGE CLIP. */
780  /* Find the first top segment: */
781 
782  for (seg = new_poly->segs; seg->type & TOP_BNDRY; seg = seg->next);
783  for (; !(seg->type & TOP_BNDRY); seg = seg->next);
784 
785  /* Check for intersections of the right pel line with top boundary
786  segments. Use the first one we encounter: */
787 
788  seg1 = seg2 = NULL;
789  while (seg->type & TOP_BNDRY) {
790  nseg = seg->next;
791  if ((seg->x <= pel_x_p1 && nseg->x >= pel_x_p1)
792  || (seg->x >= pel_x_p1 && nseg->x <= pel_x_p1)) {
793  seg1 = seg;
794  break;
795  }
796  seg = nseg;
797  }
798  /* If we have a top intersection, we must have a bottom intersection.
799  Find the last bottom segment that intersects the right: */
800 
801  if (seg1) {
802  for (; seg->type & TOP_BNDRY; seg = seg->next);
803  while (!(seg->type & TOP_BNDRY)) {
804  nseg = seg->next;
805  if ((seg->x <= pel_x_p1 && nseg->x >= pel_x_p1)
806  || (seg->x >= pel_x_p1 && nseg->x <= pel_x_p1)) {
807  seg2 = seg;
808  }
809  seg = nseg;
810  }
811  assert (seg2);
812 
813  /* Make the new segment, and splice it into the polygon: */
814 
815  nseg = new_segs++;
816  nseg->x = nseg->x1 = nseg->x2 = pel_x_p1;
817  nseg->y = seg1->y2;
818  nseg->y1 = pel_y;
819  nseg->y2 = pel_y_p1;
820  nseg->type = BNDRY_TYPE_TOP_RIGHT;
821 
822 #if DEBUG
823  nseg->y1_start = nseg->y2;
824  nseg->slope = SLOPE_VERT;
825  strcpy (nseg->name, "-R");
826 #endif
827 
828  seg2->x = pel_x_p1;
829  seg2->y = seg2->y2;
830 
831  seg1->next = nseg;
832  nseg->next = seg2;
833  new_poly->segs = seg1;
834 
835 #if DEBUG
836  if (diag_flags & DIAG_POLY_CLIP_FINE) {
837  printf ("\nAfter right clipping:\n");
838  printf ("seg1 - seg2: %2s - %2s\n", seg1->name, seg2->name);
839  dump_poly (new_poly);
840  }
841 #endif
842  }
843  /* LEFT EDGE CLIP. */
844  /* Find the first bottom segment: */
845 
846  for (seg = new_poly->segs; !(seg->type & TOP_BNDRY); seg = seg->next);
847  for (; seg->type & TOP_BNDRY; seg = seg->next);
848 
849  /* Check for intersections of the left pel line with bottom boundary
850  segments. Use the first one we encounter: */
851 
852  seg1 = seg2 = NULL;
853  while (!(seg->type & TOP_BNDRY)) {
854  nseg = seg->next;
855  if ((seg->x <= pel_x && nseg->x >= pel_x)
856  || (seg->x >= pel_x && nseg->x <= pel_x)) {
857  seg1 = seg;
858  break;
859  }
860  seg = nseg;
861  }
862  /* If we have a bottom intersection, we must have a top intersection.
863  Find the last top segment that intersects the left: */
864 
865  if (seg1) {
866  for (; !(seg->type & TOP_BNDRY); seg = seg->next);
867  while (seg->type & TOP_BNDRY) {
868  nseg = seg->next;
869  if ((seg->x <= pel_x && nseg->x >= pel_x)
870  || (seg->x >= pel_x && nseg->x <= pel_x)) {
871  seg2 = seg;
872  }
873  seg = nseg;
874  }
875  assert (seg2);
876 
877  /* Make the new segment, and splice it into the polygon: */
878 
879  nseg = new_segs++;
880  nseg->x = nseg->x1 = nseg->x2 = pel_x;
881  nseg->y = seg1->y1;
882  nseg->y1 = pel_y;
883  nseg->y2 = pel_y_p1;
884  nseg->type = BNDRY_TYPE_TOP_RIGHT;
885 
886 #if DEBUG
887  nseg->y1_start = nseg->y2;
888  nseg->slope = SLOPE_VERT;
889  strcpy (nseg->name, "-R");
890 #endif
891 
892  seg2->x = pel_x;
893  seg2->y = seg2->y1;
894 
895  seg1->next = nseg;
896  nseg->next = seg2;
897  new_poly->segs = seg1;
898 
899 #if DEBUG
900  if (diag_flags & DIAG_POLY_CLIP_FINE) {
901  printf ("\nAfter left clipping:\n");
902  printf ("seg1 - seg2: %2s - %2s\n", seg1->name, seg2->name);
903  dump_poly (new_poly);
904  }
905 #endif
906  }
907  /* Fix up the number of segments: */
908 
909  n = 0;
910  seg = new_poly->segs;
911  while (1) {
912  ++n;

916  }
917 
918  new_poly->num_segs = n;
919 }

924 static double
925 bndry_fract (const poly_segment * seg, /* boundary segment */
926  int32 x, /* pel X value as an integer */
927  double pel_x, /* integral pel X coordinate */
928  double pel_y /* integral pel Y coordinate */
929 )
930 /* Return the appropriate fraction of the pixel at <pel_x, pel_y> outside the
931  boundary segment <seg>. */
932 {
933  unsigned char itype; /* type of segment/pel intersection */
934  double area; /* return value */
935  double pel_x_p1; /* precomputation */
936  double pel_y_p1; /* precomputation */
937 
938  /* Don't use this segment if it doesn't intersect this pel: */
939 
940  if (seg->min_col > x
941  || seg->max_col < x)
942  return 0.0;
943 
944  /* There are six types of intersection. Two divide the pel into
945  trapezoids, 4 into triangles. Test for the type of intersection, and
946  compute the relevant area. The computations are eased by using the
947  knowledge that the pel is a unit square. */
948 
949  pel_x_p1 = pel_x + 1.0;
950  pel_y_p1 = pel_y + 1.0;
951 
952  if (seg->slope == SLOPE_VERT) {
953  if (seg->x1 <= pel_x) {
954  area = 0.0;
955  itype = ISECT_NONE;
956  } else if (seg->x1 > pel_x_p1) {
957  area = 1.0;
958  itype = ISECT_NONE;
959  } else {
960  area = seg->x1 - pel_x;
961  itype = ISECT_TOP_BOTTOM;
962  }
963  } else if (seg->y1 < pel_y) {
964 
965  if (seg->y2 < pel_y) {
966  area = 0.0;
967  itype = ISECT_NONE;
968  } else if (seg->y2 < pel_y_p1) {
969  /* top-right intersection, triangular region */
970  area = 0.5 * (1.0 - (seg->x1 - pel_x)) * (seg->y2 - pel_y);
971  itype = ISECT_TOP_RIGHT;
972  } else {
973  /* top-bottom intersection, trapezoidal region */
974  area = 0.5 * (seg->x1 + seg->x2) - pel_x;
975  itype = ISECT_TOP_BOTTOM;
976  }
977 
978  } else if (seg->y1 < pel_y_p1) {
979 
980  if (seg->y2 < pel_y) {
981  /* left-top intersection, triangular region */
982  area = 0.5 * (seg->x1 - pel_x) * (seg->y1 - pel_y);
983  itype = ISECT_TOP_LEFT;
984  } else if (seg->y2 < pel_y_p1) {
985  /* left-right intersection, trapezoidal region */
986  area = 0.5 * (seg->y1 + seg->y2) - pel_y;
987  itype = ISECT_LEFT_RIGHT;
988  } else {
989  /* left-bottom intersection, triangular region */
990  area = 0.5 * (seg->x2 - pel_x) * (1.0 - (seg->y1 - pel_y));
991  itype = ISECT_LEFT_BOTTOM;
992  }
993 
994  } else {
995 
996  if (seg->y2 < pel_y) {
997  /* bottom-top intersection, trapezoidal region */
998  area = 0.5 * (seg->x1 + seg->x2) - pel_x;
999  itype = ISECT_TOP_BOTTOM;
1000  } else if (seg->y2 < pel_y_p1) {
1001  /* bottom-right intersection, triangular region */
1002  area = 0.5 * (1.0 - (seg->x2 - pel_x))
1003  * (1.0 - (seg->y2 - pel_y));
1004  itype = ISECT_BOTTOM_RIGHT;
1005  } else {
1006  area = 1.0;
1007  itype = ISECT_NONE;
1008  }
1009  }
1010  /* We have computed the area. Depending on the boundary type, invert the
1011  area (subtract it from 1): */
1012 
1013  if (seg->isect_outside[itype] == -1) {
1014  printf ("Warning! Invalid bndry/isect combination: btype %d, isect %d\n",
1015  seg->type, itype);
1016  area = 0.0;
1017  } else if (!seg->isect_outside[itype]) {
1018  area = 1.0 - area;
1019  }
1020 
1021 #if DEBUG
1022  if (diag_flags & DIAG_FILL) {
1023  char x1buf[10];
1024  char x2buf[10];
1025  char y1buf[10];
1026  char y2buf[10];
1027 
1028  trunc_form (x1buf, "%6.2f", seg->x1);
1029  trunc_form (x2buf, "%6.2f", seg->x2);
1030  trunc_form (y1buf, "%6.2f", seg->y1);
1031  trunc_form (y2buf, "%6.2f", seg->y2);
1032 
1033  printf ("%31s %3.0f: %2s %6s %6s %6s %6s %2s %6.3f %1d\n",
1034  "", pel_x, seg->name,
1035  x1buf, x2buf, y1buf, y2buf,
1036  isect_type_name_tab[itype],
1037  area,
1038  seg->isect_outside[itype]);
1039  }
1040 #endif
1041 
1042  return area;
1043 }

1047 
1048 void
1049 rc_fill_poly (rcWindow& image, /* image to draw into */
1050  int32 num_pts, /* number of polygon vertices */
1051  double *coords, /* array: x1, y1, x2, y2, ... */
1052  uint32 colors /* colors to use */
1053 )
1054 /* Utility function to do gray-scale filling of a polygon defined by the
1055  vertices whose coordinates are given in <coords>.
1056 
1057 This routine REQUIRES that the points be ordered in circular fashion to
1058  outline a convex polygon. In otherwords, the centroid of the polygon may
1059  not be contained in any of the boundary segments. */
1060 {
1061  polygon_a poly; /* polygon to fill */
1062  polygon_a new_poly; /* 2nd, clipped, polygon */
1063  poly_segment _stk_segs[MAX_STK_SEGS]; /* array of segments on stack */
1064  poly_segment *segs = _stk_segs; /* array of segments */
1065  poly_segment _new_stk_segs[MAX_STK_SEGS + 4]; /* array of segments on
1066  stack */
1067  poly_segment *new_segs = _new_stk_segs; /* array of segments */
1068  int32 num_rows; /* number of rows containing rectangle */
1069  int32 num_cols; /* number of columns containing rect */
1070  int32 row_incr; /* address offset between rows */
1071  uint8 *row; /* pel row pointer */
1072  uint8 *pel; /* pel row pointer */
1073  int32 row_no; /* loop index */
1074  int32 col_no; /* loop index */
1075  uint8 fgnd; /* foreground color */
1076  uint8 color; /* pel color to set */
1077  double bgnd_fract; /* fraction of background color to use */
1078  int32 x; /* current pel X coord */
1079  int32 y; /* current pel Y coord */
1080  double f_x; /* x as a double */
1081  double f_y; /* y as a double */
1082  int32 col_shift; /* amount column # is shifted */
1083  int32 clamped_col1; /* col1 clamped to image limits */
1084  poly_segment *seg; /* ptr to current segment */
1085  int32 n; /* loop index */
1086  int32 max_row; /* maximum row */
1087  int32 max_col; /* maximum column */
1088 
1089  /* Validate input: */
1090 
1091  assert(num_pts >= 3);
1092 
1093  if (num_pts > MAX_STK_SEGS) {
1094  segs = (poly_segment *) malloc ((int32) (num_pts * sizeof (*segs)));
1095  new_segs = (poly_segment *) malloc ((int32) ((num_pts + 4)
1096  * sizeof (*new_segs)));
1097  }
1098  /* Build the polygon: */
1099 
1100  memset ((char *) new_poly, 1, sizeof (new_poly));
1101  new_poly->max_segs = num_pts + 4;
1102 
1103  max_row = image.height() - 1;
1104  max_col = image.width() - 1;
1105 
1106  poly->max_segs = num_pts;
1107  poly->segs = segs;
1108  coords_to_poly (num_pts, coords, poly, max_row, max_col);
1109 
1110  /* If the polygon is completely outside the image, exit now: */
1111 
1112  if (poly->min_col > max_col
1113  || poly->max_col < 0
1114  || poly->min_row > max_row
1115  || poly->max_row < 0)
1116  return;
1117 
1118  /* Clamp the polygon limits to image boundaries: */
1119 
1120  CLAMP (poly->min_row, 0, max_row);
1121  CLAMP (poly->max_row, 0, max_row);
1122  CLAMP (poly->min_col, 0, max_col);
1123  CLAMP (poly->max_col, 0, max_col);
1124 
1125  num_rows = poly->max_row - poly->min_row + 1;
1126  num_cols = poly->max_col - poly->min_col + 1;
1127 
1128  /* Extract colors: */
1129 
1130  fgnd = (uint8) (colors & 0xFF);
1131 
1132 #if DEBUG
1133  diag_flags = ((colors >> 8) & 0xFF);
1134 #endif
1135 
1136  /* Remember the intensity values at each vertex. This is to avoid
1137  time-consuming checking in inner loops; instead we blow away the
1138  values, and blend later, using the cached original values. */
1139 
1140  for (n = poly->num_segs, seg = poly->segs;
1141  n > 0;
1142  --n, seg = seg->next) {
1143 
1144  if (seg->vertex_in_image)
1145  seg->old_vert_int = image.getPixel (seg->ix, seg->iy);
1146  }
1147 
1148  /* Determine the X coordinate of the intersection of each segment with
1149  the line <y=min_row> and the Y coordinate of the intersection of the
1150  segment with the line <x=min_col>. */
1151 
1152  for (n = poly->num_segs, seg = poly->segs; n > 0; --n, seg = seg->next) {
1153 
1154  switch (seg->slope) {
1155  case SLOPE_HORZ:
1156  seg->x2 = poly->min_col;
1157  seg->y1_start = seg->y;
1158  break;
1159 
1160  case SLOPE_VERT:
1161  seg->x2 = seg->x;
1162  seg->y1_start = poly->min_row;
1163  break;
1164 
1165  default:
1166  seg->x2 = seg->x + (poly->min_row - seg->y) * seg->dx;
1167  seg->y1_start = seg->y + (poly->min_col - seg->x) * seg->dy;
1168  break;
1169  }
1170  }
1171 
1172 #if DEBUG
1173  if (diag_flags & DIAG_ENTRY) {
1174  dump_poly (poly);
1175  }
1176 #endif
1177 
1178  /* Loop through rows: */
1179 
1180 #if DEBUG
1181  if (diag_flags & DIAG_FILL) {
1182  printf ("\n%3s: %2s %2s %2s %3s %3s %3s %3s | %3s: %2s %6s %6s %6s %6s %2s %6s %1s\n\n",
1183  "row", "SB", "LB", "RB", "C1", "C2", "C3", "C4", "col", "nm", "x1", "x2", "y1", "y2", "it", "area", "I");
1184  }
1185 #endif
1186 
1187  row_incr = image.rowUpdate ();
1188 
1189  for (row_no = num_rows, y = poly->min_row,
1190  row = image.rowPointer(y);
1191  row_no > 0;
1192  --row_no, y += 1, row += row_incr) {
1193 
1194  /* Initialize the row loop state: */
1195 
1196  poly->left_bndry = NULL;
1197  poly->right_bndry = NULL;
1198  poly->single_bndry = TRUE;
1199  poly->col1 = poly->col3 = INT_MAX;
1200  poly->col2 = poly->col4 = INT_MIN;
1201 
1202  /* Update the state for each boundary segment. This stuff is used to
1203  compute X and Y intercepts, and limits for speed: */
1204 
1205  for (n = poly->num_segs, seg = poly->segs;
1206  n > 0;
1207  --n, seg = seg->next) {
1208 
1209  int32 this_col; /* utility column number */
1210 
1211  /* Compute the starting X and Y intercepts. Do the minimum
1212  updates required for all segments. Later we do more for the
1213  segments actually in use. */
1214 
1215  seg->x1 = seg->x2;
1216  seg->x2 += seg->dx;
1217  seg->y2 = seg->y1_start;
1218 
1219  /* Check if this segment crosses the current row. If not, ignore
1220  it: */
1221 
1222  if (seg->max_row < y || seg->min_row > y) {
1223  seg->use_me = FALSE;
1224  continue;
1225  }
1226  seg->use_me = TRUE;
1227 
1228  switch (seg->slope) {
1229  case SLOPE_VERT:
1230  seg->min_col = seg->max_col = seg->first_col;
1231  break;
1232 
1233  case SLOPE_HORZ:
1234  poly->single_bndry = FALSE;
1235  seg->min_col = seg->first_col;
1236  seg->max_col = seg->last_col;
1237  break;
1238 
1239  case SLOPE_POS:
1240  this_col = (int32) seg->x1;
1241  seg->min_col = max (this_col, seg->first_col);
1242  this_col = (int32) seg->x2;
1243  seg->max_col = min (this_col, seg->last_col);
1244  break;
1245 
1246  case SLOPE_NEG:
1247  this_col = (int32) seg->x2;
1248  seg->min_col = max (this_col, seg->first_col);
1249  this_col = (int32) seg->x1;
1250  seg->max_col = min (this_col, seg->last_col);
1251  break;
1252  }
1253  /* Update boundaries and limits. */
1254 
1255  if (seg->type & LEFT_BNDRY) {
1256  if (seg->min_col < poly->col1)
1257  poly->col1 = seg->min_col;
1258  if (seg->max_col > poly->col2)
1259  poly->col2 = seg->max_col;
1260 
1261  if (poly->single_bndry) {
1262  if (poly->left_bndry == NULL)
1263  poly->left_bndry = seg;
1264  else
1265  poly->single_bndry = FALSE;
1266  }
1267  } else {
1268 
1269  if (seg->min_col < poly->col3)
1270  poly->col3 = seg->min_col;
1271  if (seg->max_col > poly->col4)
1272  poly->col4 = seg->max_col;
1273 
1274  if (poly->single_bndry) {
1275  if (poly->right_bndry == NULL)
1276  poly->right_bndry = seg;
1277  else
1278  poly->single_bndry = FALSE;
1279  }
1280  }
1281  }
1282 
1283  /* Sanity check: make sure we have left and right boundary segments
1284  if single_bndry is still set: */
1285 
1286  if (poly->single_bndry
1287  && (poly->left_bndry == NULL || poly->right_bndry == NULL)) {
1288  printf ("Error! single boundary without both boundaries!\n");
1289  poly->single_bndry = FALSE;
1290  }
1291  /* Compensate Y intercept if not starting at leftmost column: */
1292 
1293  col_shift = poly->col1 - poly->min_col;
1294  if (col_shift > 0) {
1295  for (n = poly->num_segs, seg = poly->segs;
1296  n > 0;
1297  --n, seg = seg->next) {
1298 
1299  seg->y2 += col_shift * seg->dy;
1300  }
1301  }
1302 
1303  /* If this row is outside the image, go to next row: */
1304 
1305  if (poly->col4 < 0 || poly->col1 > max_col)
1306  continue;
1307 
1308 #if DEBUG
