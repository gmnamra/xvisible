60  class rcMatrix_2d : public boost::numeric::ublas::c_matrix< double, 2, 2 >
61  /* Named degrees of freedom:
62  * There are various ways of specifying the 4 degrees of freedom represented
63  * by a 2x2 matrix. Two sets of four "named" degrees of freedom are
64  * implemented for constructors and accessors:
65  * 1) xRot, yRot, xScale, yScale (Rx, Ry, Sx, Sy)
66  * 2) scale, aspect, shear, rotation (s, a, K, R)
67  *
68  * A 2x2 matrix is defined as follows in terms of these variables:
69  * |e11 e12| |Sx(cosRx) -Sy(sinRy)| |s(cosR) as(-sinR - cosR tanK)|
70  * |e21 e22| = |Sx(sinRx) Sy(cosRy)| = |s(sinR) as( cosR - sinR tanK)|
71  *
72  * The composition order is:
73  * |Sx(cosRx) -Sy(sinRy)| |cosRx -sinRy| |Sx 0 |
74  * |Sx(sinRx) Sy(cosRy)| = |sinRx cosRy| |0 Sy|
75  *
76  * Or:
77  * |s(cosR) as(-sinR - cosR tanK)| |cosR -sinR| |1 -tanK| |1 0|
78  * |s(sinR) as( cosR - sinR tanK)| = s * |sinR cosR| |0 1 | |0 a|
79  *
80  * The named degrees of freedom are extracted from a matrix as follows:
81  * Rx = atan2(e21, e11) s = sqrt(e11*e11 + e21*e21)
82  * Ry = atan2(-e12, e22) a = det / (e11*e11 + e21*e21)
83  * Sx = sqrt(e11*e11 + e21*e21) K = -atan((e11*e12 + e21*e22) / det)
84  * Sy = sqrt(e12*e12 + e22*e22) R = atan2(e21, e11)
85  * Note: det = a * (s^2)
86  * Note that none of these is meaningful if the matrix is singular, even though
87  * some values might be well-defined by the above formulas.
88  *
89  * Also note that the extraction formulas define the canonical ranges for the
90  * variables (for a non-singular matrix), as follows:
91  * Rx [-PI, PI] s (0, +Inf)
92  * Ry [-PI, PI] R [-PI, PI]
93  * Sx (0, +Inf) a (-Inf, +Inf), a != 0
94  * Sy (0, +Inf) K (-PI/2, PI/2)
95  *
96  * Be careful if you use both the (Rx, Ry, Sx, Sy) style of
97  * decomposition and the (s, a, K, R) style. There are some
98  * non-obvious interactions between the two. For example, when shear
99  * is present, aspect != yScale/xScale.
100  */

147  rcMatrix_2d(double scale, double aspect,
148  const double& shear, const rcRadian& rotation)
149  {
150  matrix_type (2,2);
151  double cosR = cos(rotation);
152  double sinR = sin(rotation);
153  double tanK = std::tan(shear);
154  matrix_type::insert_element (0,0, scale * cosR);
155  matrix_type::insert_element (0,0, aspect * scale * (- sinR - cosR * tanK ));
156  matrix_type::insert_element (0,0, scale * sinR);
157  matrix_type::insert_element (0,0, aspect * scale * ( cosR - sinR * tanK ));
158  setdt();
159  }
