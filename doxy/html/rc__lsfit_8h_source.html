138 bool rcLsFit<T>::solve3p (rc2dVector<T>& trans, rcRadian& rot)
139  {
140  if (mN < 3) return false;
141 
142  T e = mN * (mSux + mSvy) - mSu * mSx - mSv * mSy;
143  T d = mN * (mSvx + mSuy) + mSu * mSy - mSv * mSx;
144 
145  rot = atan2 ((double) d, (double) e);
146  T c = (T) cos (rot);
147  T s = (T) sin (rot);
148  trans.x ((mSu - c * mSx + s * mSy) / mN);
149  trans.y ((mSv - s * mSx - c * mSy) / mN);
150 
151 
152  return true;
153  }

342 template<class T>
343 rcLineSegment<T> rcLsFit<T>::fit () const
344 {
345  T
346  xnom = mSu / mN,
347  ynom = mSv / mN,
348  a = mSuu - mSu * xnom,
349  b = 2 * (mSuv - mSu * ynom),
350  c = mSvv - mSv * ynom;
351 
352  if (fabs(b) < 1e-12 && fabs(a - c) < 1e-12)
353  throw general_exception ("Unstable line Fit");
354 
355  T
356  hypotenuse = sqrt(rmSquare(b) + rmSquare(a - c)),
357  cos2theta = (a - c) / hypotenuse,
358  sn = sqrt( (1 - cos2theta) / 2),
359  cn = sqrt( (1 + cos2theta) / 2);
360 
361  if (b < 0) cn = -cn;
362 
363  T distance = -(xnom * sn)+(ynom * cn);
364  rcRadian angle(atan2(sn,cn));
365  if (distance >= 0)
366  return rcLineSegment<T>(angle,distance);
367  return rcLineSegment<T>(angle+rcRadian(rkPI),-distance);
368 
369 }
