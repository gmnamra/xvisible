192 // These will be called by a multi-threaded engine so they must be MT-safe.
193 // Qt is NOT MT-safe so all Qt calls should be protetced by a global application
194 // lock if they can be called from multiple threads.
195 
196 // Locking every Qt call in application code would be tedious and error-prone
197 // so Qt postEvent mechanism is used instead. Notifications coming from
198 // a MT engine will be posted as custom events to the main thread.
199 // Since the main thread is responsible for all Qt GUI activity,
200 // simultaneous unsafe calls to Qt methods cannot occur. When
201 // the main event loop processes the custom events, it will call
202 // rcModelDomain::customEvent() for each event.

516  notifyStatus( "Loading experiment data..." );
517

519  rcExperimentFileFormat format = eExperimentNativeFormat;
520 
521  // First, prompt for unsaved data from current experiment
522  rcExperimentState state = getExperimentState();
523 
524  if ( state != eExperimentEmpty && state != eExperimentLocked ) {
525  // No data has been saved, prompt user
526 
527  switch( QMessageBox::warning( 0, cAppName,
528  "Do you want to save experiment data before opening a new experiment?\n\n"
529  "If you don't save, your data will be lost.",
530  "&Don't Save", "Cancel", "&Save",
531  2, 1 ) ) {
532  case 0: // Don't Save clicked or Alt+D pressed
533  break;
534  case 1: // Cancel clicked or Alt+C pressed or Escape pressed
535  // Do nothing
536  return;
537  break;
538  case 2: // Save clicked or Alt+S pressed
539  requestSave( format );
540  break;
541  }
542  }
543 
544  // TODO: query (some of) these from engine settings
545  QString caption = QStrFrcStr (persistenceManager->fileFormatImportCaption( format ));
546 
547  switch ( mode ) {
548  case eExperimentImportAllSettings:
549  caption += " Settings";
550  break;
551  case eExperimentImportExperimentSettings:
552  caption += " Template";
553  break;
554  case eExperimentImportAllData:
555  caption += " Data";
556  break;

563 
564  // Ready to choose a file to open

566  "Untitled",
567  filter,
568  0,
569  caption,
570  caption );
571 
572  if (s != QString::null)
573  {

575  rcModelProgress loadProgress( this, "Loading experiment data...%.2f%% complete", 2.5, qApp );
576

579  QApplication::setOverrideCursor(Qt::waitCursor );
580 
581  // Determine import directory
582  int idx = s.findRev( '/' );
583  if ( idx != -1 ) {
584  QString dir = s.left( idx + 1 );

592 
593  // Special Reify handlers

622  QApplication::setOverrideCursor( Qt::waitCursor );

626  QApplication::restoreOverrideCursor();

628  // Errors during import, display warning

630  .arg( errors );
631  // Display error dialog
632  QMessageBox::critical( 0, cAppName, message, 1, 0 );
633  }
634  }
635  }
636  notifyStatus( "Ready" );
637  }
638  catch (general_exception& x)
639  {
640  QMessageBox::critical( 0 , cAppName , x.what() , 1 , 0 );
641  }
642 }

746  widgetName = QString( "export-dialog" );
747  progressMessage = std::string( "Exporting image data...%.2f%% complete" );
748  processExtension( defaultName, ext, nativeExt );
749  break;
750 
751  case eExperimentNativeMovieFormat:
752  widgetName = QString( "save-dialog" );
753  progressMessage = std::string( "Saving image data...%.2f%% complete" );
754  processExtension( defaultName, ext, nativeExt );
755  break;
756 
757  default:
758  rmAssert( 0 );
759  break;
760  };
761 
762  QString fileName = Q3FileDialog::getSaveFileName(
763  defaultName,
764  filter,
765  0,
766  widgetName,
767  caption,
768  &filter,
769  true );
770 
771  // If user canceled, the fileName will be QString::null
772  if (fileName == QString::null) {
773  notifyStatusInternal( "Ready" );
774  return;
775  }
776 
777  // Add extension if necessary
778  if (!fileName.endsWith( ext ))
779  fileName += ext;
780 
781  // Warn if file already exists
782  if (QFile::exists( fileName ))
783  {
784  if (QMessageBox::warning( 0 , cAppName , "File already exists: overwrite?" ,
785  QMessageBox::Ok , QMessageBox::Cancel ) == QMessageBox::Cancel) {
786  notifyStatusInternal( "Ready" );
787  return;
788  }
789  }
790 
791  // Export the experiment data.
792  QApplication::setOverrideCursor(Qt::waitCursor );
793  rcModelProgress progressIndicator( this, progressMessage, 2.5, qApp );
794 
795  int err = mDomain->saveExperiment( fileName.latin1(), format, &progressIndicator );
796  QApplication::restoreOverrideCursor();
797 
798  if ( err ) {
799  // Error during export/save!
800  QString error( caption );
801  error += " error: ";
802  error += strerror( err );
803  QMessageBox::critical( 0 , cAppName , error , 1 , 0 );
804  }
805  // Experiment file name may have changed, do a notification
806  notifySettingChange();
807  notifyStatus( "Ready" );
808  }
809  catch (general_exception& x)
810  {
811  QMessageBox::critical( 0 , cAppName , x.what() , 1 , 0 );
812  }

913  const QString& string = ev->myData();
914  // Use status bar widget
915  notifyStatusInternal( string.latin1() );
916  }
917  break;
918 
919  case eNotifyTimeEvent:
920  {
921  rcBaseTimeEvent* ev = static_cast<rcBaseTimeEvent *> (e);
922  // Produce elapsed time notification
923  const rcTimestamp& time = ev->myData();
924  emit elapsedTime( time );
925  }
926  break;
927 
928  case eNotifyCursorEvent:

930  // Produce cursor movement notification

933  emit cursorTime( time );
934  }
935  break;

943  }
944  break;

954  }
955  break;
