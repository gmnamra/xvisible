733  float m0 = 0.0, m1 = 0.0, m2 = 0.0;
734 
735  // Based on Horn
736  // Traversing from the end
737  vector<float>::const_iterator endp = profile.end();
738  endp--;
739  for(vector<float>::const_iterator p = endp;
740  p > profile.begin();
741  p--){
742  m0 += *p;
743  m1 += m0;
744  m2 += m1;
745  }
746 
747  // since we assume left tail == 0, we do not have to add i * hist[i] to m1
748  zero = m0 + profile[0]; // number of sample
749  first = m1; // mean * number of sample
750  second = m2 * 2 - m1; // var = second_/zero_ - (first_/zero_)^2
751 }

981 bool rcShape::peakDetect1d (const vector<float>& signal, rcDPair& extendPeaks)
982 {
983  // assume nothing.
984  extendPeaks.x() = 0.0; extendPeaks.y() = 0;
985 
986  // Take the highest positive and the highest negative.
987  rcEdgeFilter1DPeak posMax, negMax;
988  posMax._value = 0, posMax._location = -1, negMax._value = 0, negMax._location = -1;
989  vector<rcEdgeFilter1DPeak>peaks, posPeaks, negPeaks;
990  rcEdgeFilter1D pfinder;
991  vector<int32> efilter, filtered;
992  pfinder.genFilter (efilter, 2, 1);
993  pfinder.filter (efilter);
994  pfinder.genPeaks (signal, &filtered, peaks, 0, 1.0, 0.5);
995 
996  for (uint32 i = 0; i < peaks.size(); i++)
997  {
998  int32 left = (i == 0) ? 0 : peaks[i-1]._value;
999  int32 right = (i == (peaks.size() -1) ) ? 0 : peaks[i+1]._value;
1000  int32 val = peaks[i]._value;
1001 
1002  if (val <= 0)
1003  {
1004  if (val <= left && val <= right)
1005  {
1006  negPeaks.push_back (peaks[i]);
1007 
1008  if (val < negMax._value)
1009  {
1010  negMax._value = val;
1011  negMax._location = peaks[i]._location;
1012  }
1013  }
1014  }
1015  else
1016  {
1017  rmAssert (val > 0);
1018 
1019  if (val >= left && val >= right)
1020  {
1021  posPeaks.push_back (peaks[i]);
1022 
1023  if (val > posMax._value)
1024  {
1025  posMax._value = val;
1026  posMax._location = peaks[i]._location;
1027  }
1028  }
1029  }
1030  }
1031 
1032  if (posMax._location < 0 || negMax._location < 0) return false;
1033  // In case we needed space for all the score vector <float> scores (posPeaks.size() * negPeaks.size(), zf);
1034  // Score every pos/neg pair based on spatial distance and peak strength.
1035  float dMax (signal.size());
1036  float pnMax = posMax._value - negMax._value;
1037  float sdMax (0.0f);
1038  rcIPair sMax;
1039 
1040  for (uint32 i = 0; i < negPeaks.size(); i++)
1041  {
1042  for (uint32 j = 0; j < posPeaks.size(); j++)
1043  {
1044  float d = rmABS (posPeaks[j]._location - negPeaks[i]._location) / dMax;
1045  float s = (posPeaks[j]._value - negPeaks[i]._value) / pnMax;
1046  s *= d;
1047  if (s > sdMax)
1048  {
1049  sMax = rcIPair ((int32) i, (int32) j);
1050  sdMax = s;
1051  }
1052  }
1053  }
1054 
1055  // First things first
1056  // Given the filtering, peaks will never be the ends but right now be paranoid
1057 
1058  rcDPair extends;
1059  extends.x() = rfInterpolatePeak (filtered, negPeaks[sMax.x()]._location);
1060  extends.y() = rfInterpolatePeak (filtered, posPeaks[sMax.y()]._location);
1061  extendPeaks.x() = rmMin (extends.x(), extends.y());
1062  extendPeaks.y() = rmMax (extends.x(), extends.y());
1063 
1064 // Debugging Print
1065 
1066 #ifdef SHAPEDEBUG
1067 
1068  cerr << dec << "+++++++++++++++++++++++++" << endl;
1069  cerr << signal << endl;
1070  cerr << filtered << endl;
1071  cerr << efilter << endl;
1072  for (uint32 i = 0; i < peaks.size(); i++) cerr << "[" << i << "]" << peaks[i]._location << ":\t" << peaks[i]._value << endl;
1073  cerr << negPeaks[sMax.x()]._location << " ---- " << posPeaks[sMax.y()]._location << endl;
1074  cerr << extendPeaks << endl;
1075 
1076  cerr << "+++++++++++++++++++++++++" << endl;
1077 
1078 #endif
1079 
1080  return true;
1081 }
