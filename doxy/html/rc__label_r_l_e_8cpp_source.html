43 rcLabelRLE::Run* rcLabelRLE::image2runs (const rcWindow& img, T* pelPtr)
44 {
45  const int32 height(img.height());
46  const int32 width(img.width());
47  Run* runs = alloc_;
48  int32 lastLength;
49  uint32 lastValue;
50  int32 runsleft = numRuns ();
51 
52  for (int y = 0 ; y < height; y++)
53  {
54  const T *pels = (T *) img.rowPointer(y);
55  rat_[y] = runs;
56  int32 w = width;
57  const T *runStart = pels; //@note pointer size
58  lastValue = pels[0];
59 
60  while(w--)
61  {
62  if(*pels != lastValue)
63  {
64  lastLength = pels - runStart;
65  if (lastLength > 0)
66  {
67  runs->label = 0;
68  runs->value = lastValue;
69  runs++->length = lastLength;
70  }
71  lastValue = *pels;
72  runStart = pels;
73  }
74  ++pels;
75  }
76 
77  lastLength = pels - runStart;
78  if (lastLength > 0)
79  {
80  runs->label = 0;
81  runs->value = lastValue;
82  runs++->length = lastLength;
83  }
84 
85  const int32 runsUsed = runs - rat_[y];
86  runsleft -= runsUsed;
87 
88  // make EOR marker
89  runs->value = 0;
90  runs->length = 0;
91  ++runs;
92  }
93

105 static int32 rfRLEstimate (const rcWindow& image)
106 {
107  rmAssert (image.isBound());
108  int32 estimate = 0;
109  int32 bLast = image.width () - 2;
110  int32 height = image.height ();
111  for (int32 j = 0; j < image.height(); j++)
112  {
113  switch (image.depth ())
114  {
115  case rcPixel8:
116  estimate += rfGradHist1d ((uint8 *) image.pelPointer (0,j), (uint8 *) image.pelPointer (bLast,j));
117  break;
118  case rcPixel16:
119  estimate += rfGradHist1d ((uint16 *) image.pelPointer (0,j), (uint16 *) image.pelPointer (bLast,j));
120  break;
121  case rcPixel32:
122  estimate += rfGradHist1d ((uint32 *) image.pelPointer (0,j), (uint32 *) image.pelPointer (bLast,j));
123  break;
124  default:
125  rmExceptionMacro(<< "rcWindow::notImplemented");
126  }
127  }
128  return estimate+height+height;
129 }
