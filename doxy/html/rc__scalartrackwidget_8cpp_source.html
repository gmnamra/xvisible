270 // called to update the value label, etc.
271 void rcScalarTrackWidget::updateTrackDisplay( void )
272 {
273  rcModelDomain* domain = rcModelDomain::getModelDomain();
274 
275  // In camera mode we can only see the last value
276  if ( domain->getExperimentAttributes().liveInput )
277  _cursorTime = cCursorTimeCurrent;
278 
279  // determine the time to display
280  rcTimestamp timeToDisplay = _cursorTime;
281 
282  if (_cursorTime < cZeroTime)
283  timeToDisplay = _elapsedTime;
284 
285  // if we don't have a valid time we can skip out now.
286  if ( timeToDisplay < cZeroTime )
287  return;
288 
289  rcExperiment* experiment = domain->getExperiment();
290  int nTrackGroups = experiment->getNTrackGroups();
291 
292  if ( nTrackGroups > _trackGroupNo ) {
293  rcTrackManager* tmp = rcTrackManager::getTrackManager();
294 
295  if ( tmp->isTrackEnabled( _trackGroupNo , _trackNo ) ) {
296  // Show current values only for enabled tracks
297  rcTrackGroup* trackGroup = experiment->getTrackGroup( _trackGroupNo );
298  const int32 semantic = trackGroup->getType();
299 
300  rcTrack* track = trackGroup->getTrack( _trackNo );
301 
302  if ( track ) {
303  rcTrackType type = track->getTrackType();
304 
305  rcSegmentIterator* iter = 0;
306 
307  if ( type == eScalarTrack ) {
308  // get the scalar track.
309  rcScalarTrack* scalarTrack = dynamic_cast<rcScalarTrack*>( track );
310  if (scalarTrack == 0)
311  return;
312  // get the scalar value at the time to display
313  iter = scalarTrack->getDataSegmentIterator( timeToDisplay );
314  } else if ( type == eVideoTrack ) {
315  // get the video track
316  rcVideoTrack* videoTrack = dynamic_cast<rcVideoTrack*>( track );
317  if (videoTrack == 0)
318  return;
319  iter = videoTrack->getDataSegmentIterator( timeToDisplay );
320  } else if ( type == eGraphicsTrack ) {
321  // get the graphics track.
322  rcGraphicsTrack* graphicsTrack = dynamic_cast<rcGraphicsTrack*>( track );
323  if (graphicsTrack == 0)
324  return;
325  // get the value at the time to display
326  iter = graphicsTrack->getDataSegmentIterator( timeToDisplay );
327  } else if ( type == ePositionTrack ) {
328 
329  // get the position track.
330  rcPositionTrack* pTrack = dynamic_cast<rcPositionTrack*>( track );
331  if (pTrack == 0) {
332  rmAssert( 0 );
333  return;
334  }
335  // get the value at the time to display
336  iter = pTrack->getDataSegmentIterator( timeToDisplay );
337  } else {
338  rmAssert( 0 );
339  }
340 
341  if ( iter ) {
342  char valueBuf[ 128 ];
343  bool hasValue = iter->contains( timeToDisplay );
344 
345  // Trying to get the very last value?
346  if ( !hasValue && _cursorTime == cCursorTimeCurrent ) {
347  // The very last value may not be available yet so go back one segment
348  iter->advance( -1 );
349  if ( iter->getSegmentIndex() != -1 )
350  hasValue = true;
351  }
352 
353  if ( hasValue ) {
354  // timeToDisplay is within the segment
355  if ( type == eScalarTrack ) {
356  rcScalarSegmentIterator* si = dynamic_cast<rcScalarSegmentIterator*>(iter);
357  double value = si->getValue();
358  if ( semantic == eGroupSemanticsGlobalMeasurements)
359  value = log (value);
360  snprintf( valueBuf , rmDim(valueBuf),
361  track->getDisplayFormatString(), value );
362  } else if ( type == eVideoTrack ) {
363  int idx = iter->getSegmentIndex();
364  snprintf( valueBuf , rmDim(valueBuf), "Frame " );
365  int len = strlen( valueBuf );
366  snprintf( valueBuf + len , rmDim(valueBuf)-len,
367  track->getDisplayFormatString(), idx );
368  } else if ( type == eGraphicsTrack ) {
369  rcGraphicsSegmentIterator* gi = dynamic_cast<rcGraphicsSegmentIterator*>(iter);
370  uint32 graphicCount = gi->getCount();
371  snprintf( valueBuf, rmDim(valueBuf),
372  track->getDisplayFormatString(), graphicCount );
373  } else if ( type == ePositionTrack ) {
374  rcPositionSegmentIterator* pi = dynamic_cast<rcPositionSegmentIterator*>(iter);
375  rcFPair value = pi->getValue();
376  snprintf( valueBuf, rmDim(valueBuf),
377  track->getDisplayFormatString(), value.x(), value.y() );
378  }
379  } else {
380  // timeToDisplay is either before track start or after track end
381  snprintf( valueBuf , rmDim(valueBuf), "Undefined" );
382  }
383 
384  _value->setText( valueBuf );
385  // Add tool tip to show the value, too
386  QToolTip::remove( _value );
387  QToolTip::add( _value , valueBuf );
388  //cerr << *iter;
389  delete iter;
390  }
391  }
392  } else {
393  // Value not available
394  _value->setText( "N/A" );
395  // Add tool tip to show the value, too
396  QToolTip::remove( _value );
397  QToolTip::add( _value , "N/A" );
398  }
399 
400  QFont textFont = _value->font();
401  bool hilited = tmp->isHiliteTrack( _trackGroupNo , _trackNo );
402 #if 1
403  if ( hilited ) {
404  textFont.setWeight( QFont::Bold );
405  } else {
406  textFont.setWeight( QFont::Normal );
407  }
408 #else
409  if ( hilited ) {
410  textFont.setUnderline( true );
411  } else {
412  textFont.setUnderline( false );
413  }
414 #endif
415  _value->setFont( textFont );
416  _name->setFont( textFont );
417  }
418 
419  _currentTime = timeToDisplay;
420 }
