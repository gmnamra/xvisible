105 rcGenMovieFileError rcGenMovieFile::addFrame(const rcWindow& frame)
106 {
107  if ( _flushed )
108  return eGenMovieFileErrorFlush;
109  if ( !valid() )
110  return _lastError;
111 
112  if ( _toc.empty() ) {
113  _lastError = initHeader( frame );
114  if ( !valid() )
115  return _lastError;
116  } else {
117  if ( _rev <= movieFormatRev1 ) {
118  // All added frames must be the same size
119  rmAssert(int(_movieHdr.width()) == frame.width());
120  rmAssert(int(_movieHdr.height()) == frame.height());
121  } else if ( _rev >= movieFormatRev2 ) {
122  // All added frames must be the same size
123  rmAssert(int(_movieHdr2.width()) == frame.width());
124  rmAssert(int(_movieHdr2.height()) == frame.height());
125  }
126 
127  rmAssert( _inputDepth == frame.depth() );
128  }
129 
130  int64 timestamp = 0;
131 
132  if ( _frameInterval > 0.0f ) {
133  // Generate timestamps from forced frame rate
134  if ( !_toc.empty() )
135  timestamp = convertSecondsToTimestamp((rcTimestamp(_toc.back()) + rcTimestamp(_frameInterval)).secs());
136  } else
137  timestamp = frame.frameBuf()->timestamp()._timestamp;
138 
139  if ( frame.depth() == rcPixel8 ) {
140  // 8-bit, no conversion
141  write( frame, timestamp );
142  } else if ( frame.depth() == rcPixel32 ) {
143  if ( _conversion == rcSelectAll ) {
144  // 32-bit, no conversion
145  write( frame, timestamp );
146  } else {
147  // Convert 32-bit to 8-bit
148  rcWindow frame8(frame.width(), frame.height());
149  rfImageConvert32to8 (frame, frame8, _conversion);
150  rmAssert( rcPixel8 == frame8.depth() );
151  frame8.frameBuf()->setIsGray(true);
152  frame8.frameBuf()->setTimestamp(rcTimestamp(timestamp));
153  write( frame8, timestamp );
154  }
155  } else if ( frame.depth() == rcPixel16) {
156  if ( _conversion == rcSelectAll ) {
157  // 16-bit, no conversion
158  write( frame, timestamp );
159  } else {
160  // @todo merge this and rc_ipconvert{h,cpp} move them to visual
161  // Convert 16-bit to 8-bit
162  rcWindow frame8(frame.width(), frame.height());
163  rfImageConvert168 (frame, frame8, _conversion);
164  rmAssert( rcPixel8 == frame8.depth() );
165  frame8.frameBuf()->setIsGray(true);
166  frame8.frameBuf()->setTimestamp(rcTimestamp(timestamp));
167  write( frame8, timestamp );
168  }
169  } else {
170  // Unsupported depth
171  rmAssert( 0 );
172  }
173 
174  _toc.push_back(timestamp);
175 
176  return eGenMovieFileErrorOK;
177 }

239 rcGenMovieFileError rcGenMovieFile::flush()
240 {
241  if ( _flushed )
242  return eGenMovieFileErrorOK;
243  if ( !_saveStream )
244  return eGenMovieFileErrorNull;
245 
246  _flushed = true;
247 
248  if (_toc.size() < 2) {
249  if ( fclose(_saveStream)) {
250  perror("fclose in flush(0) failed");
251  return eGenMovieFileErrorClose;
252  }
253  return eGenMovieFileErrorNull;
254  }
255 
256  // Update basic header
257  int64 firstTime = _toc[0];
258  double movieTime = rcTimestamp(_toc[_toc.size()-1] - firstTime).secs();
259 
260 
261  switch ( _rev ) {
262  case movieFormatRev0:
263  _movieHdr.averageFrameRate((_toc.size()-1)/movieTime);
264  _movieHdr.frameCount(_toc.size());
265  break;
266 
267  case movieFormatRev1:
268  {
269  _movieHdr.averageFrameRate((_toc.size()-1)/movieTime);
270  _movieHdr.frameCount(_toc.size());
271  // Store TOC header
272  rcMovieFileTocExt tocHdr( _toc.size() );
273  if ( tocHdr.write( _saveStream, _toc ) )
274  return eGenMovieFileErrorWrite;
275  if ( _verbose )
276  cerr << tocHdr << endl;
277  /* Store EOF header.
278  */
279  rcMovieFileExt eofHdr(movieExtensionEOF);
280 
281  if (eofHdr.write(_saveStream))
282  return eGenMovieFileErrorWrite;
283  if ( _verbose )
284  cerr << eofHdr << endl;
285  }
286  break;
287 
288  case movieFormatRev2:
289  {
290  _movieHdr2.averageFrameRate((_toc.size()-1)/movieTime);
291  _movieHdr2.frameCount(_toc.size());
292  // Update rev2 header additions
293  const off_t movieRecordSz = sizeof(int64) + _movieHdr2.bytesInFrame();
294  // Offset to start of headers
295  _movieHdr2.extensionOffset( sizeof(_movieHdr2) +
296  movieRecordSz*_movieHdr2.frameCount());
297 
298  // Store TOC header
299  rcMovieFileTocExt tocHdr( _toc.size() );
300  if ( tocHdr.write( _saveStream, _toc ) )
301  return eGenMovieFileErrorWrite;
302  if ( _verbose )
303  cerr << tocHdr << endl;
304  std::string creator = _creator;
305 
306  // Each rev2 header needs a unique id
307  int32 id = 1;
308  // Store origin headers
309  if ( !_orgHdrs.empty() ) {
310  for ( uint32 i = 0; i < _orgHdrs.size(); ++i ) {
311  rcMovieFileOrgExt orgHdr = _orgHdrs[i];
312  orgHdr.id( id++ );
313  if ( orgHdr.write( _saveStream ) )
314  return eGenMovieFileErrorWrite;
315  if ( _verbose )
316  cerr << orgHdr << endl;
317  }
318  } else {
319  // User added no headers, produce a default header
320  rcMovieFileOrgExt orgHdr( _origin,
321  _movieHdr2.baseTime(),
322  _movieHdr2.frameCount(),
323  _movieHdr2.width(),
324  _movieHdr2.height(),
325  _inputDepth,
326  _rev,
327  _creator.c_str() );
328  orgHdr.id( id++ );
329  if ( orgHdr.write( _saveStream ) )
330  return eGenMovieFileErrorWrite;
331  if ( _verbose )
332  cerr << orgHdr << endl;
333  }
334 
335  // Store camera headers
336  for ( uint32 i = 0; i < _camHdrs.size(); ++i ) {
337  rcMovieFileCamExt cam = _camHdrs[i];
338 
339  cam.id( id++ );
340  if ( cam.write( _saveStream ) )
341  return eGenMovieFileErrorWrite;
342  }
343 
344  // Store conversion headers
345  for ( uint32 i = 0; i < _cnvHdrs.size(); ++i ) {
346  rcMovieFileConvExt cnv = _cnvHdrs[i];
347 
348  if ( cnv.channel() != movieChannelUnknown ) {
349  cnv.id( id++ );
350  if ( cnv.write( _saveStream ) )
351  return eGenMovieFileErrorWrite;
352  }
353  }
354 
355  // Store experiment headers
356  for ( uint32 i = 0; i < _expHdrs.size(); ++i ) {
357  rcMovieFileExpExt exp = _expHdrs[i];
358  exp.id( id++ );
359  if ( exp.write( _saveStream ) )
360  return eGenMovieFileErrorWrite;
361  }
362 
363  /* Store EOF header.
364  */
365  rcMovieFileExt eofHdr(movieExtensionEOF);
366 
367  if (eofHdr.write(_saveStream))
368  return eGenMovieFileErrorWrite;
369  if ( _verbose )
370  cerr << eofHdr << endl;
371  }
372  break;
373 
374  case movieFormatInvalid:
375  rmAssert( 0 );
376  break;
377  }
378 
379  if (fseek(_saveStream, 0, SEEK_SET)) {
380  perror("fseek in flush failed");
381  if (fclose(_saveStream))
382  perror("fclose in flush(5) failed");
383  return eGenMovieFileErrorSeek;
384  }
385  if ( _rev <= movieFormatRev1 ) {
386  // Write basic header
387  if ( _movieHdr.write( _saveStream ) )
388  return eGenMovieFileErrorWrite;
389  } else if ( _rev >= movieFormatRev2 ) {
390  // Write rev2 header
391  if ( _movieHdr2.write( _saveStream ) )
392  return eGenMovieFileErrorWrite;
393  } else {
394  rmAssert( 0 );
395  }
396 
397  if (fclose(_saveStream)) {
398  perror("fclose in flush(7) failed");
399  return eGenMovieFileErrorClose;
400  }
401 
402  return eGenMovieFileErrorOK;
403 }

443  // Write rev2 header

445  return eGenMovieFileErrorWrite;
446  } else {
447  rmAssert( 0 );
448  }
449 
450  return eGenMovieFileErrorOK;

453 // Write one frame to stream
454 rcGenMovieFileError rcGenMovieFile::write( const rcWindow& frame,
455  const int64& timestamp )
456 {
457  if ( _flushed )
458  return eGenMovieFileErrorFlush;
459 
460  if ( !valid() )
461  return _lastError;
462 
463  const uint32 rowDataLth = frame.width() * frame.depth();
464  const uint32 rowUpdateLth = (rowDataLth + 15) & 0xFFFFFFF0;
465  const uint32 byteCnt = sizeof(int64) + rowUpdateLth*frame.height();
466 
467  rcMovieFrameFormat* fPtr = (rcMovieFrameFormat*)malloc(byteCnt);
468  rmAssert(fPtr);
469 
470  fPtr->timestamp = timestamp;
471 
472  for (int32 yo = 0; yo < frame.height(); yo++) {
473  uint8* dPtr = (uint8*)&fPtr->rawPixels[yo*rowUpdateLth];
474  const uint8* sPtr = (uint8*)frame.rowPointer(yo);
475  memcpy(dPtr, sPtr, rowDataLth);
476  }
477 
478  if (fwrite(fPtr, byteCnt, 1, _saveStream) != 1) {
479  perror("fwrite of frame in saveToDisk failed");
480  if (fclose(_saveStream))
481  perror("fclose in saveToDisk(1) failed");
482  free (fPtr);
483  return eGenMovieFileErrorWrite;
484  }
485 
486  free(fPtr);
487 
488  return eGenMovieFileErrorOK;
489 }

491 // Initialize movie header with frame information
492 rcGenMovieFileError rcGenMovieFile::initHeader( const rcWindow& frame )
493 {
494  rmAssert(rcPixel8 == frame.depth() || rcPixel16 == frame.depth() ||
495  rcPixel32 == frame.depth());
496 
497  uint32 rowDataLth = frame.width();
498  uint32 rowUpdateLth = (rowDataLth + 15) & 0xFFFFFFF0;
499 
500  if ( _conversion == rcSelectAll && frame.depth() == rcPixel32) {
501  rowDataLth = frame.width() * frame.depth();
502  rowUpdateLth = (rowDataLth + 15) & 0xFFFFFFF0;
503  }
504 
505  if ( _conversion == rcSelectAll && frame.depth() == rcPixel16) {
506  rowDataLth = frame.width() * frame.depth();
507  rowUpdateLth = (rowDataLth + 15) & 0xFFFFFFF0;
508  }
509 
510  if ( _rev <= movieFormatRev1 ) {
511  _movieHdr = rcMovieFileFormat( frame.frameBuf(), _rev );
512  _movieHdr.rowUpdate( rowUpdateLth );
513  if ( _frameInterval > 0.0f && _origin == movieOriginSynthetic ) {
514  // Synthetic frame rate, set base time to current time
515  _movieHdr.baseTime( getCurrentTimestamp() );
516  }
517  } else if ( _rev >= movieFormatRev2 ) {
518  _movieHdr2 = rcMovieFileFormat2( frame.frameBuf(), _rev );
519  _movieHdr2.rowUpdate( rowUpdateLth );
520 
521  if ( _conversion != rcSelectAll ) {
522  // Output will be 8-bit
523  _movieHdr2.depth( rcPixel8 );
524  }
525  if ( _frameInterval > 0.0f && _origin == movieOriginSynthetic ) {
526  // Synthetic frame rate, set base time to current time
527  _movieHdr2.baseTime( getCurrentTimestamp() );
528  }
529  }
530 
531  _inputDepth = frame.depth();
532 
533  return eGenMovieFileErrorOK;
534 }
