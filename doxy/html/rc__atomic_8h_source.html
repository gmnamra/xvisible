45 /* rcAtomicValue - Allows the creation of data types that can safely be
46  * read "atomically" by one or more "consumer threads"
47  * without concern that a "producer thread" is
48  * simultaneously modifying the same data. Since no
49  * mutexes are used, the caller is guaranteed never to
50  * be blocked.
51  *
52  * The following restrictions apply:
53  *
54  * 1) T must work properly with the default copy constructor and
55  * assignment operator. See 3) for a feel of why this requried.
56  *
57  * 2) There must be only one "producer thread".
58  *
59  * 3) T should not be "too large". What's too large? I'll describe a
60  * little of how rcAtomicValue works and let you decide for yourself:
61  *
62  * rcAtomicValue based objects store 4 copies of the T object
63  * internally. When a call to getValue() is made, all four values
64  * are read into local stack variables and then compared with each
65  * other. If two copies are found that match, and some other
66  * conditions are met, a valid copy of the underlying T object is
67  * deemed to have been found. Otherwise, the process repeats. A
68  * final copy is done to pass a version of the current value of
69  * the T object back to the caller.
70  *
71  * As you can see, as the size of T grows, a large number of
72  * copy/comparison operations are required (and the likelyhood
73  * of a read/write collision grows as well).
74  *
75  * The interface consists of the following public functions:
76  *
77  * rcAtomicValue<T>(const T& v) - Init object to value of v
78  *
79  * rcAtomicValue<T>(const rcAtomicValue<T>& at) - Init object from another
80  * rcAtomicValue<T>
81  *
82  * T& rcAtomicValue<T>::getValue(T& v) const - Store the current value in v
83  * and return a reference to v
84  *
85  * void rcAtomicValue<T>::setValue(const T& v) - Set the value of the object
86  * to v
87  */

106  T& getValue(T& v) const
107  {
108  unsigned char local_v0[2][sizeof(T)];
109  unsigned char local_v1[2][sizeof(T)];
110 
111  UT_GET_DATA_PTR;
112  /* Note: In the following comments the terms "stable" and
113  * "matching" are used. By "stable", I mean: "the value, when read
114  * in, is not partially the old value and partially the new
115  * value". By "matching", I mean: "a binary comparison of the two
116  * values are identical".
117  */
118  /* Keep looping until we get a stable, matching pair of
119  * values. Note that all reads (and writes) are done through
120  * function calls to guarantee that the compiler doesn't try to
121  * reorder any of them.
122  */
123  for (;;)
124  {
125  UT_INSERT_RACE;
126  unsigned long beforeCtr = getCounter();
127 
128  /* Read values in order opposite of that used to do writes in
129  * setValue().
130  */
131  copyValue((char*)local_v0[0], (volatile char*)_v0[0]);
132  copyValue((char*)local_v0[1], (volatile char*)_v0[1]);
133  copyValue((char*)local_v1[0], (volatile char*)_v1[0]);
134  copyValue((char*)local_v1[1], (volatile char*)_v1[1]);
135 
136  /* If the counter has changed, 2 or more calls to setValue() may
137  * have occurred during this call. This is at least one more
138  * than we can safely handle. Try again.
139  */
140  UT_INSERT_RACE;
141  unsigned long afterCtr = getCounter();
142  if (beforeCtr != afterCtr)
143  continue;
144 
145  /* If the counter hasn't changed then at most 1 call to
146  * setValue() has occurred during this call. If setValue()
147  * hasn't been called during this call, one of two things has
148  * happened:
149  *
150  * 1) Both _v0[] and _v1[] have stable, matching data.
151  *
152  * 2) Some thread was preempted in the middle of a call to
153  * setValue() and it is still waiting to run again. In
154  * this case, either _v0[] or _v1[] may contain unstable
155  * values, but not both.
156  *
157  * In either case, a stable, matching pair of values is
158  * available.
159  *
160  * Now consider the case where a call to setValue() was
161  * actively writing _v0[] and _v1[] during our reads of the
162  * same values. Because these reads were done in the opposite
163  * order as the writes were done in setValue(), at most one
164  * of the values read in has an unstable value. As proof,
165  * consider the following:
166  *
167  * 1) For a value to be unstable, it must be in the process
168  * of being read while writes are occuring.
169  *
170  * 2) If the first unstable read is to be _v0[0], then
171  * setValue() must have already written _v0[1], _v1[0],
172  * _v1[1]. The subsequent reads of these must yield stable
173  * values.
174  *
175  * 3) If the first unstable read is to be _v0[1], then
176  * setValue() must have already written _v1[0], _v1[1].
177  * Moreover, the write to _v0[0] cannot have occurred yet,
178  * so the read of that value must have been stable.
179  *
180  * 4) If the first unstable read is to be _v1[0], then
181  * setValue() must have already written _v1[1]. Moreover,
182  * the writes to _v0[0], _v0[1] cannot have occurred yet,
183  * so the reads of those values must have been stable.
184  *
185  * 5) If the first unstable read is to be _v1[1], then the
186  * writes to _v0[0], _v0[1], _v1[0] cannot have occurred
187  * yet, so the reads of those values must have been stable.
188  *
189  * 6) From inspection of 2) through 5) it is clear that one
190  * of _v0[] or _v1[] is a stable, matching pair.
191  */
192  if (equalValue((char*)local_v0[0], (char*)local_v0[1]))
193  {
194  copyValue((char*)&v, (char*)local_v0[0]);
195  return v;
196  }
197  else if (equalValue((char*)local_v1[0], (char*)local_v1[1]))
198  {
199  copyValue((char*)&v, (char*)local_v1[0]);
200  return v;
201  }
202 
203  /* If, as claimed, a stable, matching pair is available
204  * this code should never be reached.
205  */
206  assert(0);
207  }
208 
209  return v; // Make compiler happy
210  }

212  void setValue(const T& v)
213  {
214  /* Note that all writes are done through function calls to
215  * guarantee that the compiler doesn't try to reorder any of them.
216  */
217  /* Note: In the following comments the terms "stable" and
218  * "matching" are used. By "stable", I mean: "the value, when read
219  * in, is not partially the old value and partially the new
220  * value". By "matching", I mean: "a binary comparison of the two
221  * values are identical".
222  */
223  /* If the calling thread is preempted at some point during these
224  * two writes, _v0[] still has stable, matching values.
225  */
226  copyValue((volatile char*)_v1[1], (char*)&v);
227  genSync();
228  copyValue((volatile char*)_v1[0], (char*)&v);
229  genSync();
230 
231  /* If the calling thread is preempted at some point during these
232  * two writes, _v1[] still has stable, matching values.
233  */
234  copyValue((volatile char*)_v0[1], (char*)&v);
235  genSync();
236  copyValue((volatile char*)_v0[0], (char*)&v);
237  genSync();
238 
239  /* Tells getValue() this function has been called.
240  */
241  incCounter();
242  }

250  static void copyValue(volatile char* vDest, const volatile char* vSrc)
251  {
252  UT_INSERT_RACE_N_CODE;
253 
254  if (sizeof(T) == sizeof(int))
255  *(volatile int*)((void*)vDest) = *(volatile int*)((void*)vSrc);
256  else if (sizeof(T) == sizeof(short))
257  *(volatile short*)((void*)vDest) = *(volatile short*)((void*)vSrc);
258  else if (sizeof(T) == sizeof(char))
259  *(volatile char*)((void*)vDest) = *(volatile char*)((void*)vSrc);
260  else
261  memcpy((void*)vDest, (void*)vSrc, sizeof(T));
262  }

264  static bool equalValue(const char* vA, const char* vB)
265  {
266  if (sizeof(T) == sizeof(int))
267  return *(int*)((void*)vA) == *(int*)((void*)vB);
268  else if (sizeof(T) == sizeof(short))
269  return *(short*)((void*)vA) == *(short*)((void*)vB);
270  else if (sizeof(T) == sizeof(char))
271  return *(char*)((void*)vA) == *(char*)((void*)vB);
272 
273  return memcmp((void*)vA, (void*)vB, sizeof(T)) == 0;
274  }
