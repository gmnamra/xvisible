42 class rcXMLNameMapper {
43  public:
44  rcXMLNameMapper( const std::string* elemArray, uint32 elemSize,
45  const std::string* attrArray, uint32 attrSize ) :
46  mElemNames( elemArray ), mAttrNames( attrArray ) {
47  rmAssert( elemArray );
48  rmAssert( attrArray );
49  rmAssert( elemSize > 0 );
50  rmAssert( attrSize > 0 );
51 
52  // Fill element map
53  for ( uint32 i = 0; i < elemSize; ++i ) {
54  mElementMap.insert( make_pair( elemArray[i], static_cast<rcXMLElementType>(i) ) );
55  // Immediate self-test, it can catch duplicates
56  rmAssert( elementType( elemArray[i] ) == static_cast<rcXMLElementType>(i) );
57  }
58 
59  // Fill attribute map
60  for ( uint32 i = 0; i < attrSize; ++i ) {
61  mAttributeMap.insert( make_pair( attrArray[i], static_cast<rcXMLAttributeType>(i) ) );
62  // Immediate self-test, it can catch duplicates
63  rmAssert( attributeType( attrArray[i] ) == static_cast<rcXMLAttributeType>(i) );
64  }
65 
66  // Each name should be unique, check final size
67  rmAssert( mElementMap.size() == elemSize );
68  rmAssert( mAttributeMap.size() == attrSize );
69  };
70 
71  // Map element type to name
72  const std::string& elementName( rcXMLElementType id ) const {
73  return mElemNames[id];
74  };
75 
76  // Map element name to type
77  rcXMLElementType elementType( const std::string& name ) const {
78  rcXMLElementNameMap::const_iterator pos = mElementMap.find( name );
79  if ( pos != mElementMap.end() )
80  return pos->second;
81  else
82  return eXMLElementMax;
83  };
84 
85  // Map attribute type to name
86  const std::string& attributeName( rcXMLAttributeType id ) const {
87  return mAttrNames[id];
88  };
89 
90  // Map attribute name to type
91  rcXMLAttributeType attributeType( const std::string& name ) const {
92  rcXMLAttributeNameMap::const_iterator pos = mAttributeMap.find( name );
93  if ( pos != mAttributeMap.end() )
94  return pos->second;
95  else
96  return eXMLAttributeMax;
97  };
98 
99  private:
100  rcXMLElementNameMap mElementMap;
101  rcXMLAttributeNameMap mAttributeMap;
102  const std::string* mElemNames; // Element name array
103  const std::string* mAttrNames; // Attribute name array

106 class rcXMLTimeMapper {
107  public:
108  rcXMLTimeMapper() {}
109  rcXMLTimeMapper( const vector<rcTimestamp>& timeArray )
110  {
111  mTimes = timeArray;
112  // Fill time map
113  for ( uint32 i = 0; i < timeArray.size(); ++i ) {
114  mTimeMap.insert( make_pair( timeArray[i], i ) );
115  // Immediate self-test, it can catch duplicates
116  rmAssert( timeIndex( mTimes[i] ) == int32(i) );
117  }
118 
119  // Each time should be unique, check final size
120  rmAssert( mTimeMap.size() == timeArray.size() );
121  };
122 
123  // Map index to time value
124  const rcTimestamp timeValue( uint32 i ) const {
125  if ( i < mTimes.size() )
126  return mTimes[i];
127  else
128  return rcTimestamp(-1.0);
129  };
130 
131  // Map time value to index
132  int32 timeIndex( const rcTimestamp& time ) const {
133  rcXMLTimeMap::const_iterator pos = mTimeMap.find( time );
134  if ( pos != mTimeMap.end() )
135  return pos->second;
136  else
137  return -1;
138  };
139 
140  private:
141  rcXMLTimeMap mTimeMap;
142  vector<rcTimestamp> mTimes;
143 };
