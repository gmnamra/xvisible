1049  T0_F = vec_madd(T1_F, f1, f0);

1052 
1053  *(vector float*)_vectorSumPtr = T0_F;
1054

1056  res.score[yo][1] = *((float*)_vectorSumPtr + 2);
1057  res.score[yo][2] = *((float*)_vectorSumPtr + 1);
1058  } // End of: for (int32 yo = 0; yo < 3; yo++) {
1059  } // End of: "Now calculate correlation scores ..." block
1060 #endif
