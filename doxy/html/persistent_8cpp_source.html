480 // Integer types
481 // format: {size}{byte}*size
482 // size can be zero!
483 //
484 // A major problem is that integer types may be different sizes on different
485 // machines or even with different compilers on the same machine (though I
486 // haven't come across that yet). Neither the C nor the C++ standards specify
487 // the size of integer types. Dumping an int on one machine might dump 16
488 // bits. Restoring it on another machine might try to restore 32 bits. With
489 // version 0 of these persistence routines, this was not dealt with and so the
490 // restore function could fail. Version 1 handles different type sizes. It
491 // does this by writing the size to the file as well as the data, so the
492 // restore can therefore know how many bytes to restore independent of the
493 // type size.
494 //
495 // In fact, the standard does not even specify the size of char (true! And
496 // mind-numbingly stupid...). However, to be able to do anything at all, I've
497 // had to assume that a char is 1 byte.
498 
499 static void dump_unsigned(dump_context& context, size_t bytes, unsigned char* data) throw(persistent_dump_failed)
500 {
501  // first skip zero bytes - this may reduce the data to zero bytes long
502  size_t i = bytes;
503  while(i >= 1 && data[INDEX(i-1)] == 0)
504  i--;
505  // put the remaining size
506  context.put((unsigned char)i);
507  // and put the bytes
508  while(i--)
509  context.put(data[INDEX(i)]);
510 }

512 static void dump_signed(dump_context& context, size_t bytes, unsigned char* data) throw(persistent_dump_failed)
513 {
514  // first skip all-zero or all-one bytes but only if doing so does not change the sign
515  size_t i = bytes;
516  if (data[INDEX(i-1)] < 128)
517  {
518  // positive number so discard leading zeros but only if the following byte is positive
519  while(i >= 2 && data[INDEX(i-1)] == 0 && data[INDEX(i-2)] < 128)
520  i--;
521  }
522  else
523  {
524  // negative number so discard leading ones but only if the following byte is negative
525  while(i >= 2 && data[INDEX(i-1)] == 255 && data[INDEX(i-2)] >= 128)
526  i--;
527  }
528  // put the remaining size
529  context.put((unsigned char)i);
530  // and put the bytes
531  while(i--)
532  context.put(data[INDEX(i)]);
533 }

535 static void restore_unsigned(restore_context& context, size_t bytes, unsigned char* data) throw(persistent_restore_failed)
536 {
537  // get the dumped size from the file
538  size_t dumped_bytes = (size_t)context.get();
539  // zero fill any empty space
540  size_t i = bytes;
541  for (; i > dumped_bytes; i--)
542  data[INDEX(i-1)] = 0;
543  // restore the dumped bytes but discard any that don't fit
544  // TODO - could detect overflow and throw an exception here
545  while(i--)
546  {
547  int ch = context.get();
548  if (i < bytes)
549  data[INDEX(i)] = (unsigned char)ch;
550  else
551  throw persistent_restore_failed(std::string("integer overflow: restoring byte ") + to_string(i) + " of " + to_string(bytes));
552  }
553 }

555 static void restore_signed(restore_context& context, size_t bytes, unsigned char* data) throw(persistent_restore_failed)
556 {
557  // get the dumped size from the file
558  size_t dumped_bytes = (size_t)context.get();
559  // restore the dumped bytes but discard any that don't fit
560  size_t i = dumped_bytes;
561  while(i--)
562  {
563  int ch = context.get();
564  if (i < bytes)
565  data[INDEX(i)] = (unsigned char)ch;
566  else
567  throw persistent_restore_failed(std::string("integer overflow: restoring byte ") + to_string(i) + " of " + to_string(bytes));
568  }
569  // sign extend if the dumped integer was smaller
570  if (dumped_bytes < bytes)
571  {
572  if (data[INDEX(dumped_bytes-1)] < 128)
573  {
574  // positive so zero fill
575  for (i = dumped_bytes; i < bytes; i++)
576  data[INDEX(i)] = 0;
577  }
578  else
579  {
580  // negative so one fill
581  for (i = dumped_bytes; i < bytes; i++)
582  data[INDEX(i)] = 0xff;
583  }
584  }

754 // Format: address [ size data ]
755 
756 void dump(dump_context& context, char*& data) throw(persistent_dump_failed)
757 {
758  // register the address and get the magic key for it
759  std::pair<bool,unsigned> mapping = context.pointer_map(data);
760  dump(context,mapping.second);
761  // if the address is null, then that is all that we need to do
762  // however, if it is non-null and this is the first sight of the address, dump the contents
763  if (data && !mapping.first)
764  {
765  unsigned size = strlen(data);
766  dump(context,size);
767  for (unsigned i = 0; i < size; i++)
768  dump(context,data[i]);
769  }
770 }

772 void restore(restore_context& context, char*& data) throw(persistent_restore_failed)
773 {
774  // destroy any previous contents
775  if (data)
776  {
777  delete[] data;
778  data = 0;
779  }
780  // get the magic key
781  unsigned magic = 0;
782  restore(context,magic);
783  // now lookup the magic key to see if this pointer has already been restored
784  // null pointers are always flagged as already restored
785  std::pair<bool,void*> address = context.pointer_map(magic);
786  if (!address.first)
787  {
788  // this pointer has never been seen before and is non-null
789  // restore the string
790  size_t size = 0;
791  restore(context,size);
792  data = new char[size+1];
793  for (size_t i = 0; i < size; i++)
794  restore(context,data[i]);
795  data[size] = '\0';
796  // add this pointer to the set of already seen objects
797  context.pointer_add(magic,data);
798  }
799 }

814 // specialisation for a vector of bool which has a different implementation to a vector of anything else
815 
816 void dump(dump_context& context, const std::vector<bool>& data) throw(persistent_dump_failed)
817 {
818  dump(context,data.size());
819  size_t size = data.size();
820  size_t bytes = ((size + 7) / 8);
821  for (size_t b = 0; b < bytes; b++)
822  {
823  unsigned char byte = 0;
824  unsigned char mask = 1;
825  for (size_t e = 0; e < 8; e++)
826  {
827  size_t i = b*8 + e;
828  if (i >= size) break;
829  if (data[i]) byte |= mask;
830  mask <<= 1;
831  }
832  dump(context,byte);
833  }
834 }

836 void restore(restore_context& context, std::vector<bool>& data) throw(persistent_restore_failed)
837 {
838  size_t size = 0;
839  restore(context,size);
840  data.resize(size);
841  size_t bytes = ((size + 7) / 8);
842  for (size_t b = 0; b < bytes; b++)
843  {
844  unsigned char byte = 0;
845  restore(context,byte);
846  unsigned char mask = 1;
847  for (size_t e = 0; e < 8; e++)
848  {
849  size_t i = b*8 + e;
850  if (i >= size) break;
851  data[i] = ((byte & mask) != 0);
852  mask <<= 1;
853  }
854  }
855 }
