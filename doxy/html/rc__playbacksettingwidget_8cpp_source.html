115 // Update the text shown in the widget with the
116 // current value in the setting.
117 void rcPlaybackSettingWidget::settingChanged()
118 {
119  _currentValueBits = mSetting.getValue();
120  int currentSpeed = _currentValueBits & rcSettingInfo::eCurrentSpeedMask;
121  bool fwd = (_currentValueBits & rcSettingInfo::eCurrentStateFwd) != 0;
122  bool rev = (_currentValueBits & rcSettingInfo::eCurrentStateRev) != 0;
123  bool stopped = (_currentValueBits & rcSettingInfo::eCurrentStateStopped) != 0;
124 
125  rmAssert(fwd ^ rev); // Exactly one should be true
126 
127  _speedWidget->setValue(currentSpeed);
128 
129  // Disable the widget if the setting is currently not editable.
130  if (!mSetting.isEditable())
131  setEnabled(false);
132  else {
133  setEnabled(true);
134  _revWidget->setEnabled(stopped);
135  _fwdWidget->setEnabled(stopped);
136  _speedWidget->setEnabled(stopped);
137  _stopWidget->setEnabled(!stopped);
138  }
139 
140  // hide the widget if the setting is currently not enabled.
141  if (isHidden() && mSetting.isEnabled()) {
142  show();
143  }
144  else if (!isHidden() && !mSetting.isEnabled()) {
145  hide();
146  }
147 }
