52 class rcWriterMapper {
53 public:
54  // ctor
55  rcWriterMapper( const rcWriterInfo* writerArray, uint32 size )
56  {
57  // Populate all maps
58  for ( uint32 i = 0; i < size; ++i ) {
59  rmAssert( writerArray[i].type() != eWriterUnknown );
60 
61  // Fill name map
62  mNameMap.insert( make_pair( writerArray[i].type(), writerArray[i] ) );
63  // Immediate self-test, it can catch duplicates
64  rmAssert( info( writerArray[i].type() ).type() == writerArray[i].type() );
65 
66  // Fill type map
67  mTypeMap.insert( make_pair( writerArray[i].name(), writerArray[i].type() ) );
68  // Immediate self-test, it can catch duplicates
69  rmAssert( type( writerArray[i].name() ) == writerArray[i].type() );
70 
71  // Fill tag map
72  mTagMap.insert( make_pair( writerArray[i].tag(), writerArray[i].type() ) );
73  // Immediate self-test, it can catch duplicates
74  rmAssert( typeTag( writerArray[i].tag() ) == writerArray[i].type() );
75  }
76  // Check for duplicate types in the list
77  rmAssert( mNameMap.size() == size );
78  // Check for duplicate display names in the list
79  rmAssert( mTypeMap.size() == size );
80  // Check for duplicate tag names in the list
81  rmAssert( mTagMap.size() == size );
82  }
83
