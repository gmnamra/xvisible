89 // Returns the median value of the vector, given that the values are
90 // circular, with the given modulus. Values may be signed or unsigned,
91 // eg range from -pi to pi (modulus 2pi) or from 0 to 2pi (modulus 2pi).
92 // NOTE that the array is shuffled, but the time taken is linear.
93 // An assumption is made that most of the values are spread over no more than
94 // half the range, but wrap-around is accounted for if the median is near
95 // the wrap-around point.
96 // Cannot be a member of GenericVector, as it makes heavy used of LLSQ.
97 // T must be an integer or float/double type.
98 template<typename T> T MedianOfCircularValues(T modulus, std::vector<T>* v) {
99  LLSQ stats;
100  T halfrange = static_cast<T>(modulus / 2);
101  int num_elements = v->size();
102  for (int i = 0; i < num_elements; ++i) {
103  stats.add((*v)[i], (*v)[i] + halfrange);
104  }
105  bool offset_needed = stats.y_variance() < stats.x_variance();
106  if (offset_needed) {
107  for (int i = 0; i < num_elements; ++i) {
108  (*v)[i] += halfrange;
109  }
110  }
111  int median_index = v->choose_nth_item(num_elements / 2);
112  if (offset_needed) {
113  for (int i = 0; i < num_elements; ++i) {
114  (*v)[i] -= halfrange;
115  }
116  }
117  return (*v)[median_index];
118 }
