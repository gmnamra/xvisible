65 // load an experiment from a tree and make it current.
66 // return number of errors
67 int rcExperimentDomainImpl::loadExperiment( const char* fileName,
68  const rcXMLElementTree& tree,
69  rcExperimentImportMode mode,
70  rcProgressIndicator* progress )
71 {
72  int errors = 1;
73  // Reset everything
74  newExperiment();
75 
76  if ( _experiment ) {
77  // Import XML tree into experiment

79 
80  if ( errors > 0 ) {
81  // We may be in an inconsistent state
82  // Reset everything
83  newExperiment();

86  // Set actual XML file name that was loaded

89  _state = eExperimentLocked;
90  _observer->notifyState( _state );
91  }
92  }
93 
94  return errors;
95 }

97 // save the experiment to disk
98 int rcExperimentDomainImpl::saveExperiment( const char* fileName,
99  rcExperimentFileFormat format,
100  rcProgressIndicator* progress )
101 {
102  if (_state == eExperimentEmpty)
103  rmExceptionMacro (<< "experiment not running, there is nothing to save" );
104 
105  rcPersistenceManager* persistenceManager = rcPersistenceManagerFactory::getPersistenceManager();
106  const std::string creator = persistenceManager->generatorComment();
107 
108  rcMovieConverterOptions opt;
109 
110  opt.creator( creator );
111 
112  int nPages = _experiment->getNSettingCategories();
113  for (int i = 0; i < nPages; i++)
114  {
115  rcSettingCategory category = _experiment->getSettingCategory( i );
116 
117  // Look for analysis settings
118  if ( !strcmp( category.getTag(), "analysis-settings" ) ) {
119  const rcSettingInfo rect = category.getSettingInfo( "analysis-rect" );
120  const rcSettingInfo firstFrame = category.getSettingInfo( "analysis-first-frame" );
121  const rcSettingInfo lastFrame = category.getSettingInfo( "analysis-last-frame" );
122 
123  rcRect analysisRect = rect.getValue();
124  // Hack: use analysis first and last frame to specify whoch frames to export
125  int32 first = firstFrame.getValue();
126  int32 last = lastFrame.getValue();
127 
128  opt.frameCount( last - first + 1 );
129  opt.firstFrameIndex( first );
130  if ( analysisRect.width() > 0 && analysisRect.height() > 0 )
131  opt.clipRect( analysisRect );
132 
133  }
134  }
135  int err = _experiment->saveTo( fileName, format, opt, this, progress );
136  _state = eExperimentLocked;
137  return err;
138 }

331 // called by the engine to notify us of a state change.
332 void rcExperimentDomainImpl::notifyState( rcEngineState engineState )
333 {
334  rcExperimentState newState = _state;
335 
336  switch (engineState)
337  {
338  case eEngineUninitialized:
339  if ( _experiment )
340  _experiment->editableSettings( true );
341  break;
342 
343  case eEngineRunning: // engine is running and generating data
344  newState = eExperimentRunning;
345  if ( _experiment )
346  _experiment->editableSettings( false );
347  break;
348 
349  case eEnginePlayback:
350  newState = eExperimentPlayback;
351  break;
352 
353  case eEngineStopped: // engine is stopped and not generating data
354  // If it's locked already, keep it locked
355  if ( _state == eExperimentLocked )
356  newState = eExperimentLocked;
357  else
358  newState = eExperimentEnded;
359  if ( _experiment )
360  _experiment->editableSettings( false );
361  break;
362 
363  case eEngineShuttingDown: // engine is in the process of shutting down
364  _experiment->editableSettings( false );
365  break;
366 
367  default:
368  return; // ignore.
369  }
370 
371  if ( _experiment ) {
372  // Set experiment image input source name
373  std::string inputName = _engine->getAttributes().inputName;
374  rcValue value( inputName );
375  _experiment->setSettingValue( cInputImageSourceNameSettingId, value );
376  }
377 
378  setInternalState( newState );
379 }
