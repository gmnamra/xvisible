383 // Return a RGB color vector from HSI color vector
384 uint32 rfHSI2RGB (float H, float S, float I)
385 {
386  float Htmp;
387  float R,B,G;
388 
389  if (H == 0.0)
390  R = G = B = I;
391  else if (H > 0.0 && H < rk2PI/3. )
392  {
393  Htmp = 1 / sqrt(3.0) * tan(H - rkPI/3.);
394  B = (1.0 - S) * I;
395  G = (1.5 + 1.5*Htmp ) * I - ((0.5 + 1.5*Htmp) * B);
396  R = 3.0 * I - G - B;
397  }
398  else if (H >= rk2PI/3. && H < (rk2PI+rk2PI)/3.)
399  {
400  Htmp = 1 / sqrt(3.0) * tan(H - rkPI);
401  R = (1.0 - S) * I;
402  B = (1.5 + 1.5*Htmp ) * I - ((0.5 + 1.5*Htmp) * R);
403  G = 3.0 * I - B - R;
404  }
405  else
406  {
407  Htmp = 1 / sqrt(3.0) * tan(H - (5.0 * rkPI)/3.);
408  G = (1.0 - S) * I;
409  R = (1.5 + 1.5*Htmp ) * I - ((0.5 + 1.5*Htmp) * G);
410  B = 3.0 * I - R - G;
411  }
412 
413  return rfRgb (uint32 (floor (R * 255)), uint32 (floor (G * 255)), uint32 (floor (B * 255)));
414
