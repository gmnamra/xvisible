131 template<class P>
132 void gaussEdge (P* pixelPtr, const rcWindow& src, rcWindow& dest)
133 {
134  const uint32 srcWidth(src.width());
135  const uint32 srcHeight(src.height());
136  const uint32 srcUpdate(src.rowPixelUpdate());
137  const uint32 destUpdate(dest.rowPixelUpdate());
138  const P *src0 = pixelPtr;
139  P* dest0 = (P *) dest.rowPointer(0);
140 
141  // TL and TR pixels
142  *dest0 = *src0;
143  dest0[srcWidth - 1] = src0[srcWidth - 1];
144  dest0++;
145 
146  // Top
147  uint32 left(uint32 (3* *src0 + *(src0 + srcUpdate)));
148  src0++;
149  uint32 centre(uint32 (3* *src0 + *(src0 + srcUpdate)));
150  src0++;
151  uint32 width;
152  for(width = srcWidth - 2; width--;)
153  {
154  uint32 right(uint32 (3* *src0 + *(src0 + srcUpdate)));
155  *dest0++ = P ((left + centre + centre + right + 8 )/ 16);
156  left = centre;
157  centre = right;
158  src0++;
159  }
160 
161  // BL and BR pixels
162  dest0 = (P *) dest.rowPointer (srcHeight - 1);
163  src0 = (P *) src.rowPointer (srcHeight - 1);
164  *dest0 = *src0;
165  dest0[srcWidth - 1] = src0[srcWidth - 1];
166  dest0++;
167 
168  // Bottom
169  left = uint32 (3* *src0 + *(src0 - srcUpdate));
170  src0++;
171  centre = uint32 (3* *src0 + *(src0 - srcUpdate));
172  src0++;
173  for(width = srcWidth - 2; width--;)
174  {
175  uint32 right(uint32 (3* *src0 + *(src0 - srcUpdate)));
176 
177  *dest0++ = P ((left + centre + centre + right + 8 )/ 16);
178  left = centre;
179  centre = right;
180  src0++;
181  }
182 
183  // Left
184  src0 = (P *) src.rowPointer (0);
185  dest0 = (P *) dest.rowPointer (1);
186 
187  uint32 top(uint32 (3* *src0 + *(src0+1)));
188  src0 += srcUpdate;
189  uint32 middle(uint32 (3* *src0 + *(src0+1)));
190  src0 += srcUpdate;
191 
192  uint32 height;
193  for(height = srcHeight - 2; height--;)
194  {
195  uint32 bottom(uint32 (3* *src0 + *(src0 + 1)));
196  *dest0 = P ((top + middle + middle + bottom + 8)/16);
197  src0 += srcUpdate;
198  dest0 += destUpdate;
199  top = middle;
200  middle = bottom;
201  }
202 
203  // Right
204  src0 = (P *) src.rowPointer (0) + srcWidth - 1;
205  dest0 = (P *) dest.rowPointer (1) + srcWidth - 1;
206  top = uint32 (3* *src0 + *(src0 - 1));
207  src0 += srcUpdate;
208  middle = uint32 (3* *src0 + *(src0 - 1));
209  src0 += srcUpdate;
210 
211  for(height = srcHeight - 2; height--;)
212  {
213  uint32 bottom(uint32 (3* *src0 + *(src0 - 1)));
214  *dest0 = P ((top + middle + middle + bottom + 8)/16);
215  src0 += srcUpdate;
216  dest0 += destUpdate;
217  top = middle;
218  middle = bottom;
219  }

469 static void rfGauss3by3 (const rcWindow& src, rcWindow& dest)
470 {
471  assert(src.width() == dest.width() + 2);
472  assert(src.height() == dest.height() + 2);
473 
474  uint32 ht(dest.height());
475  uint32 wd(dest.width());
476  uint32 x, y, sum1, sum2, sum3;
477  uint32 kernel_wds, k;
478  uint32 srcUpdate(src.rowPixelUpdate());
479  uint32 destUpdate(dest.rowPixelUpdate());
480 
481  uint8 *base_dst = dest.rowPointer(0);
482  uint8 *dst;
483 
484  const uint8 *top, *mid, *bot, *save;
485  uint8 curr_pel;
486 
487  // Initialize pointers to rows
488  top = src.rowPointer(0);
489  mid = top + srcUpdate;
490  bot = mid + srcUpdate;
491  save = mid;
492 
493  // Compute number of kernel multiples contained in image width
494  kernel_wds = wd / 3;
495  k = wd % 3;
496 
497  for(y = ht; y; y--, base_dst += destUpdate)
498  {
499  // SETUP FOR FIRST KERNEL POSITION:
500  dst = base_dst;
501  sum1 = uint32 (*top++);
502  curr_pel = *mid++;
503  sum1 += uint32 (curr_pel + curr_pel);
504  sum1 += uint32 (*bot++);
505 
506  sum2 = uint32 (*top++);
507  curr_pel = *mid++;
508  sum2 += uint32 (curr_pel + curr_pel);
509  sum2 += uint32 (*bot++);
510 
511  // Fill up the pipeline
512 
513  for(x = kernel_wds; x; x--)
514  {
515  gaussCompute(top,mid,bot,dst,sum1, sum2, sum3);
516 
517  gaussCompute(top,mid,bot,dst,sum2, sum3, sum1);
518 
519  gaussCompute(top,mid,bot,dst,sum3, sum1, sum2);
520  }
521 
522  // Compute any leftover pixels on the row
523  for(x = k; x; x--)
524  {
525  gaussCompute (top,mid,bot,dst,sum1, sum2, sum3);
526  sum1 = sum2;
527  sum2 = sum3;
528  }
529 
530  // Update row pointers for next row iteration
531  top = save;
532  mid = save + srcUpdate;
533  bot = mid + srcUpdate;
534  save = mid;
535  }
