30 /* rcSimilarator - Entropy signal generating class. There are two
31  * steps to this process. First, a self-similarity matrix is
32  * calculated between a set of images. Second, a measure of the
33  * entropy between the images in this set is calculated based upon the
34  * previously calculated self-similarity matrix.
35  *
36  * Note: Sometimes we generate the matrix info on the fly, removing
37  * the need to store what can become a large 2D array. In this case it
38  * is still valid, from a conceptual point of view, to think of this
39  * as a two step process. To simplify things, the following
40  * descriptions will assume both steps are being performed.
41  *
42  */

52  /* rcMatrixGeneration - Self-similarity matrix creation options.
53  *
54  * eExhaustive - Perform pairwise correlations on all images
55  * within the temporal window.
56  *
57  * eApproxNoMatrix - Perform pairwise correlation only on
58  * temporally adjacent images. Use algorithm
59  * that doesn't require storing entire matrix
60  * of approximate results to generate final
61  * entropy signal.
62  *
63  * eApproximate - Perform pairwise correlation only on
64  * temporally adjacent images. Use algorithm
65  * that first approximates the entire self
66  * similarity matrix before generating final
67  * entropy signal.
68  */
69  enum rcMatrixGeneration {
70  eExhaustive,
71  eApproxNoMatrix,
72  eApproximate
73  };

147  /* fill - Allow client to specify an initial temporal window's worth
148  * of images. If a full temporal window's worth of images are passed
149  * in, an entropy signal is generated.
150  *
151  * Clears any existing intermediate results, so passing a 0 length
152  * set of images can be used to just clear intermediate resutls.
153  *
154  * If firstImages is larger than the temporal window size, the
155  * beginning images are ignored. In all events, up to a temporal
156  * window's worth of self-similarity calculations are performed.
157  *
158  * Up to the last temporal window size number of images from
159  * firstImages are retained within the object for use in subsequent
160  * calls to update().
161  *
162  * If firstImages is at least temporal window size large, then there
163  * are enough images to calculate an entropy signal. In this case,
164  * an entropy signal is generated and true will be returned.
165  * Otherwise, no calculations are done and false is returned.
166  */
167  bool fill(vector<rcWindow>& firstImages);
168  bool fill(deque<rcWindow>& firstImages);

174  * temporal window's worth of images are available, a new entropy

411  class progressNotification
412  {
413  public:
414  progressNotification(int32 totCorr,
415  rcProgressIndicator* guiUpdate) :
416  _totCorr(totCorr), _curCorr(0), _guiUpdate(guiUpdate), _abort(false)
417  {
418  // The following is a hack to simulate the existing effect.
419  if (guiUpdate) {
420  _abort = _guiUpdate->progress(uint32(0));
421  _pctUpdate = 1;
422  }
423  else {
424  fprintf(stderr, " 0 pct done\n");
425  _pctUpdate = 5;
426  }
427  _nextPct = _pctUpdate;
428  }
429 
430  /* Return true if analysis should be aborted, false otherwise.
431  */
432  bool update()
433  {
434  if (_abort)
435  return true;
436  _curCorr++;
437  if (_nextPct <= ((_curCorr*100)/_totCorr)) {
438  if (_guiUpdate) {
439  bool abort = _guiUpdate->progress((uint32)_nextPct);
440  _nextPct += _pctUpdate;
441  return abort;
442  }
443  fprintf(stderr, "% 3d pct done\n", _nextPct);
444  _nextPct += _pctUpdate;
445  }
446  return false;
447  }
448 
449  private:
450  int32 _totCorr, _curCorr, _nextPct, _pctUpdate;
451  rcProgressIndicator* _guiUpdate;
452  bool _abort;
453  };
