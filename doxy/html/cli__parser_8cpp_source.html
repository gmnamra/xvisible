107 unsigned cli_parser_data::find_definition(const std::string& name)
108 {
109  // this does substring matching on the definitions and returns the first
110  // match - however, it requires at least one character in the substring so
111  // that the "" convention for command line arguments doesn't match with
112  // anything. It returns size() if it fails
113  for (unsigned i = 0; i < m_definitions.size(); i++)
114  {
115  std::string candidate = m_definitions[i].m_name;
116  if ((candidate.empty() && name.empty()) ||
117  (!name.empty() && candidate.size() >= name.size() && candidate.substr(0,name.size()) == name))
118  return i;
119  }
120  return m_definitions.size();
121 }

169 unsigned cli_parser_data::add_value(unsigned definition, const std::string& value, const std::string& source)
170 {
171  // behaviour depends on mode:
172  // - single: erase all previous values
173  // - multiple: erase values at a lower level than current
174  // - cumulative: erase no previous values
175  switch (m_definitions[definition].m_mode)
176  {
177  case cli_single_mode:
178  clear_level(definition, m_level);
179  break;
180  case cli_multiple_mode:
181  clear_level(definition, m_level-1);
182  break;
183  case cli_cumulative_mode:
184  break;
185  }
186  m_values.push_back(cli_value(definition,value,m_level,source));
187  return m_values.size()-1;
188 }

331 void cli_parser::set_defaults(const ini_manager& defaults, const std::string& ini_section) throw()
332 {
333  // import default values from the Ini Manager
334  m_data->increase_level();
335  // get the set of all names from the Ini manager so that illegal names generate meaningful error messages
336  std::vector<std::string> names = defaults.variable_names(ini_section);
337  for (unsigned i = 0; i < names.size(); i++)
338  {
339  std::string name = names[i];
340  unsigned definition = m_data->find_definition(name);
341  if (definition == m_data->m_definitions.size())
342  {
343  // not found - give an error report
344  error_position position(defaults.variable_filename(ini_section,name),
345  defaults.variable_linenumber(ini_section,name),
346  0);
347  m_data->m_errors.error(position,"CLI_INI_VARIABLE", name);
348  }
349  else
350  {
351  // found - so add the value
352  // multi-valued variables are entered as a comma-separated list and this is then turned into a vector
353  // the vector is empty if the value was empty
354  std::vector<std::string> values = defaults.variable_values(ini_section, name);
355  // an empty string is used to negate the value
356  if (values.empty())
357  m_data->erase_value(definition);
358  else
359  {
360  std::string source = filespec_to_relative_path(defaults.variable_filename(ini_section, name));
361  for (unsigned j = 0; j < values.size(); j++)
362  m_data->add_value(definition, values[j], source);
363  }
364  }
365  }
366  // add the set of ini files to the list for usage reports
367  for (unsigned j = 0; j < defaults.size(); j++)
368  m_data->add_ini_file(defaults.filename(j));
369 }

371 bool cli_parser::parse(char* argv[]) throw(cli_argument_error,error_handler_id_error,error_handler_format_error)
372 {
373  bool result = true;
374  if (!argv) throw cli_argument_error("Argument vector cannot be null");
375  m_data->increase_level();
376  if (argv[0])
377  m_data->m_executable = argv[0];
378  for (unsigned i = 1; argv[i]; i++)
379  {
380  std::string name = argv[i];
381  if (!name.empty() && name[0] == '-')
382  {
383  // we have a command line option
384  unsigned found = m_data->find_definition(name.substr(1, name.size()-1));
385  if (found < m_data->m_definitions.size())
386  {
387  // found it in its positive form
388  switch (m_data->m_definitions[found].m_kind)
389  {
390  case cli_switch_kind:
391  m_data->add_switch(found, true, "command line");
392  break;
393  case cli_value_kind:
394  // get the next argument in argv as the value of this option
395  // first chk that there is a next value
396  if (!argv[i+1])
397  result &= m_data->m_errors.error("CLI_VALUE_MISSING", name);
398  else
399  m_data->add_value(found, argv[++i], "command line");
400  break;
401  }
402  }
403  else if (name.size() > 3 && name.substr(1,2) == "no")
404  {
405  found = m_data->find_definition(name.substr(3, name.size()-3));
406  if (found < m_data->m_definitions.size())
407  {
408  // found it in its negated form
409  switch (m_data->m_definitions[found].m_kind)
410  {
411  case cli_switch_kind:
412  m_data->add_switch(found, false, "command line");
413  break;
414  case cli_value_kind:
415  m_data->erase_value(found);
416  break;
417  }
418  }
419  else
420  {
421  // could not find this option in either its true or negated form
422  result &= m_data->m_errors.error("CLI_UNRECOGNISED_OPTION", name);
423  }
424  }
425  else
426  {
427  // could not find this option and it could not be negated
428  result &= m_data->m_errors.error("CLI_UNRECOGNISED_OPTION", name);
429  }
430  }
431  else
432  {
433  // we have a command-line value which is represented internally as an option with an empty string as its name
434  // some very obscure commands do not have values - only options, so allow for that case too
435  unsigned found = m_data->find_definition("");
436  if (found < m_data->m_definitions.size())
437  m_data->add_value(found, name, "command line");
438  else
439  result &= m_data->m_errors.error("CLI_NO_VALUES", name);
440  }
441  }
442  if (!result) m_data->set_invalid();
443  return result;
444 }

522 void cli_parser::usage(void) const throw(std::runtime_error)
523 {
524  m_data->m_errors.information("CLI_USAGE_PROGRAM", m_data->m_executable);
525  m_data->m_errors.information("CLI_USAGE_DEFINITIONS");
526  for (unsigned d = 0; d < m_data->m_definitions.size(); d++)
527  {
528  m_data->m_errors.information(m_data->m_definitions[d].m_message);
529  }
530  if (size() != 0)
531  {
532  m_data->m_errors.information("CLI_USAGE_VALUES");
533  for (unsigned v = 0; v < size(); v++)
534  {
535  std::string source = m_data->m_values[v].m_source;
536  std::string key = name(v);
537  if (key.empty())
538  {
539  // command-line values
540  m_data->m_errors.information("CLI_USAGE_VALUE_RESULT", string_value(v), source);
541  }
542  else if (kind(v) == cli_switch_kind)
543  {
544  // a switch
545  m_data->m_errors.information("CLI_USAGE_SWITCH_RESULT", (switch_value(v) ? name(v) : "no" + name(v)), source);
546  }
547  else
548  {
549  // other values
550  std::vector<std::string> args;
551  args.push_back(name(v));
552  args.push_back(string_value(v));
553  args.push_back(source);
554  m_data->m_errors.information("CLI_USAGE_OPTION_RESULT", args);
555  }
556  }
557  }
558  if (m_data->ini_file_size() > 0)
559  {
560  m_data->m_errors.information("CLI_INI_HEADER");
561  for (unsigned i = 0; i < m_data->ini_file_size(); i++)
562  {
563  if (file_exists(m_data->ini_file(i)))
564  m_data->m_errors.information("CLI_INI_FILE_PRESENT", filespec_to_relative_path(m_data->ini_file(i)));
565  else
566  m_data->m_errors.information("CLI_INI_FILE_ABSENT", filespec_to_relative_path(m_data->ini_file(i)));
567  }
568  }
