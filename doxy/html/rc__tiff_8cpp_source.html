165 int TIFFReaderInternal::Open( const char *filename )
166 {
167  Clean();
168  struct stat fs;
169  if ( stat(filename, &fs) )
170  {
171  return 0;
172  }
173 
174 // TIFFOpen also reads the first directory
175  Image = TIFFOpen(filename, "r");
176 
177  if ( !Image)
178  {
179  Clean();
180  return 0;
181  }
182  if ( !Initialize() )
183  {
184  Clean();
185  return 0;
186  }
187 
188 
189 
190  TIFF* tif = Image;
191  register TIFFDirectory *td;
192  td = &tif->tif_dir;
193 
194  rmAssert (NumberOfPages == 1);
195 
196  for (int32 cf = 0; cf < td->td_customValueCount; cf++)
197  {
198  const TIFFFieldInfo *fip = td->td_customValues[cf].info;
199  if (!mIsSTKfile)
200  {
201  mIsSTKfile = fip->field_tag == uic2tag || fip->field_tag == uic3tag || fip->field_tag == uic4tag;
202  if (mIsSTKfile)
203  NumberOfPages = td->td_customValues[cf].count;
204  }
205  }
206 
207 // cerr << *this << endl;
208 
209 
210 /*
211  ** Custom tag support.
212  */
213  {
214  int i;
215  short count;
216 
217  count = (short) TIFFGetTagListCount(tif);
218  for(i = 0; i < count; i++)
219  {
220  ttag_t tag = TIFFGetTagListEntry(tif, i);
221  const TIFFFieldInfo *fip;
222  uint16 value_count;
223  int j, mem_alloc = 0;
224  void *raw_data;
225 
226  fip = TIFFFieldWithTag(tif, tag);
227  if(fip == NULL)
228  continue;
229 
230  if(fip->field_passcount) {
231  if(TIFFGetField(tif, tag, &value_count, &raw_data) != 1)
232  continue;
233  } else {
234  if (fip->field_readcount == TIFF_VARIABLE
235  || fip->field_readcount == TIFF_VARIABLE2)
236  value_count = 1;
237  else if (fip->field_readcount == TIFF_SPP)
238  value_count = td->td_samplesperpixel;
239  else
240  value_count = fip->field_readcount;
241  if (fip->field_type == TIFF_ASCII
242  || fip->field_readcount == TIFF_VARIABLE
243  || fip->field_readcount == TIFF_VARIABLE2
244  || fip->field_readcount == TIFF_SPP
245  || value_count > 1) {
246  if(TIFFGetField(tif, tag, &raw_data) != 1)
247  continue;
248  } else {
249  raw_data = _TIFFmalloc(
250  _TIFFDataSize(fip->field_type)
251  * value_count);
252  mem_alloc = 1;
253  if(TIFFGetField(tif, tag, raw_data) != 1)
254  continue;
255  }
256  }
257 
258  fprintf(stderr, " %s: ", fip->field_name);
259 
260  for(j = 0; j < value_count; j++) {
261  if(fip->field_type == TIFF_BYTE)
262  fprintf(stderr, "%u",
263  (unsigned int) ((uint16 *) raw_data)[j]);
264  else if(fip->field_type == TIFF_UNDEFINED)
265  fprintf(stderr, "0x%x",
266  (unsigned int) ((unsigned char *) raw_data)[j]); else if(fip->field_type == TIFF_SBYTE)
267  fprintf(stderr, "%d", (int) ((uint16 *) raw_data)[j]);
268  else if(fip->field_type == TIFF_SHORT)
269  fprintf(stderr, "%u",
270  (unsigned int)((unsigned short *) raw_data)[j]);
271  else if(fip->field_type == TIFF_SSHORT)
272  fprintf(stderr, "%d", (int)((short *) raw_data)[j]);
273  else if(fip->field_type == TIFF_LONG)
274  fprintf(stderr, "%lu",
275  (unsigned long)((unsigned long *) raw_data)[j]);
276  else if(fip->field_type == TIFF_SLONG)
277  fprintf(stderr, "%ld", (long)((long *) raw_data)[j]);
278  else if(fip->field_type == TIFF_RATIONAL
279  || fip->field_type == TIFF_SRATIONAL
280  || fip->field_type == TIFF_FLOAT)
281  fprintf(stderr, "%f", ((float *) raw_data)[j]);
282  else if(fip->field_type == TIFF_IFD)
283  fprintf(stderr, "0x%x",
284  (int)((unsigned long *) raw_data)[j]);
285  else if(fip->field_type == TIFF_ASCII) {
286  fprintf(stderr, "%s", (char *) raw_data);
287  break;
288  }
289  else if(fip->field_type == TIFF_DOUBLE)
290  fprintf(stderr, "%f", ((double *) raw_data)[j]);
291  else if(fip->field_type == TIFF_FLOAT)
292  fprintf(stderr, "%f", ((float *)raw_data)[j]);
293  else {
294  fprintf(stderr, "<unsupported data type in TIFFPrint>");
295  break;
296  }
297 
298  if(j < value_count - 1)
299  fprintf(stderr, ",");
300  }
301  fprintf(stderr, "\n");
302  if(mem_alloc)
303  _TIFFfree(raw_data);
304  }
305  }
306 
307  return 1;
308 }

396 void TIFFImageIO::ReadImageInformation()
397 {
398  m_Spacing[0] = 1.0; // We'll look for TIFF pixel size information later,
399  m_Spacing[1] = 1.0; // but set the defaults now
400 
401  m_Origin[0] = 0.0;
402  m_Origin[1] = 0.0;
403 
404  m_Dimensions[0] = width ();
405  m_Dimensions[1] = height ();
406 
407  switch ( GetFormat() )
408  {
409  case TIFFImageIO::GRAYSCALE:
410  case TIFFImageIO::PALETTE_GRAYSCALE:
411  SetNumberOfComponents( 1 );
412  SetPixelType(SCALAR);
413  break;
414  case TIFFImageIO::RGB_:
415  SetNumberOfComponents( mInternalImage->spp );
416  SetPixelType(RGB);
417  break;
418  case TIFFImageIO::PALETTE_RGB:
419  SetNumberOfComponents( 3 );
420  SetPixelType(RGB);
421  break;
422  default:
423  SetNumberOfComponents( 4 );
424  SetPixelType(RGBA);
425  }
426 
427  if ( !mInternalImage->CanRead() )
428  {
429  SetNumberOfComponents( 4 );
430  SetPixelType(RGBA);
431  }
432 
433  m_ComponentType = (mInternalImage->bps <= 8) ? UCHAR : USHORT;
434 
435 
436 // if the tiff file is multi-pages
437  if(mInternalImage->NumberOfPages>0)
438  {
439  SetNumberOfDimensions(3);
440  m_Dimensions[2] = mInternalImage->NumberOfPages;
441  m_Spacing[2] = 1.0;
442  m_Origin[2] = 0.0;
443  }
444 
445 // if the tiff is tiled
446  if(mInternalImage->NumberOfTiles>0)
447  {
448  SetNumberOfDimensions(3);
449  m_Dimensions[0] = mInternalImage->TileWidth;
450  m_Dimensions[1] = mInternalImage->TileHeight;
451  m_Dimensions[2] = mInternalImage->NumberOfTiles;
452  m_Spacing[2] = 1.0;
453  m_Origin[2] = 0.0;
454  mIsTiled = true;
455  }
456 
457  //@note set the rcPixel size that matched our reading
458  mRCdepth = (mInternalImage->bps == 8 && mInternalImage->spp == 1) ?
459  rcPixel8 : (mInternalImage->bps == 16 && mInternalImage->spp == 1) ?
460  rcPixel16 : (mInternalImage->bps == 8 && mInternalImage->spp == 3) ?
461  rcPixel32 : (mInternalImage->bps == 8 && mInternalImage->spp == 4) ?
462  rcPixel32 : rcPixelUnknown;
463 
464  mStripperimage = (uint32) mInternalImage->Image->tif_dir.td_stripsperimage;
465  mStripoffsets.resize (mStripperimage);
466  for (uint32 ii = 0; ii < mStripoffsets.size (); ii++) mStripoffsets[ii] = mInternalImage->Image->tif_dir.td_stripoffset[ii];
467  mStripbytecounts.resize (mStripperimage);
468  for (uint32 ii = 0; ii < mStripbytecounts.size (); ii++) mStripbytecounts[ii] = mInternalImage->Image->tif_dir.td_stripbytecount[ii];
469 
470  mRootHeight = mStripperimage * height ();
471  mRowperstrip = mInternalImage->Image->tif_dir.td_rowsperstrip;
472 
473  mFileBO = TIFFIsBigEndian(mInternalImage->Image) == 0 ? eByteOrderLittleEndian : eByteOrderBigEndian;
474 
475  return;
476 }

597 int TIFFImageIO::EvaluateImageAt( void* out, void* in )
598 {
599  unsigned char *image = (unsigned char*)out;
600  unsigned char *source = (unsigned char*)in;
601 
602  int increment;
603  unsigned short red, green, blue, alpha;
604  switch ( GetFormat() )
605  {
606  case TIFFImageIO::GRAYSCALE:
607  if ( mInternalImage->Photometrics ==
608  PHOTOMETRIC_MINISBLACK )
609  {
610  if(m_ComponentType == USHORT)
611  {
612  unsigned short *image = (unsigned short*)out;
613  unsigned short *source = (unsigned short*)in;
614  *image = *source;
615  }
616  else
617  {
618  *image = *source;
619  }
620  }
621  else
622  {
623  *image = ~( *source );
624  }
625  increment = 1;
626  break;
627  case TIFFImageIO::PALETTE_GRAYSCALE:
628  GetColor(*source, &red, &green, &blue);
629  *image = static_cast<unsigned char>(red >> 8);
630  increment = 1;
631  break;
632  case TIFFImageIO::RGB_:
633  red = *(source);
634  green = *(source+1);
635  blue = *(source+2);
636  *(image) = red;
637  *(image+1) = green;
638  *(image+2) = blue;
639  if ( mInternalImage->spp == 4 )
640  {
641  alpha = *(source+3);
642  *(image+3) = 255-alpha;
643  }
644  increment = mInternalImage->spp;
645  break;
646  case TIFFImageIO::PALETTE_RGB:
647  GetColor(*source, &red, &green, &blue);
648  *(image) = static_cast<unsigned char>(red >> 8);
649  *(image+1) = static_cast<unsigned char>(green >> 8);
650  *(image+2) = static_cast<unsigned char>(blue >> 8);
651  increment = 3;
652  break;
653  default:
654  return 0;
655  }
656 
657  return increment;
658 }

661 void TIFFImageIO::GetColor( int index, unsigned short *red,
662  unsigned short *green, unsigned short *blue )
663 {
664  *red = 0;
665  *green = 0;
666  *blue = 0;
667  if ( index < 0 )
668  {
669  std::cout << "Color index has to be greater than 0" << std::endl;
670  return;
671  }
672  if ( TotalColors > 0 &&
673  ColorRed && ColorGreen && ColorBlue )
674  {
675  if ( index >= TotalColors )
676  {
677  std::cout << "Color index has to be less than number of colors ("
678  << TotalColors << ")" << std::endl;
679  return;
680  }
681  *red = *(ColorRed + index);
682  *green = *(ColorGreen + index);
683  *blue = *(ColorBlue + index);
684  return;
685  }
686 
687  unsigned short photometric;
688 
689  if (!TIFFGetField(mInternalImage->Image, TIFFTAG_PHOTOMETRIC, &photometric))
690  {
691  if ( mInternalImage->Photometrics != PHOTOMETRIC_PALETTE )
692  {
693  std::cout << "You can only access colors for palette images" << std::endl;
694  return;
695  }
696  }
697 
698  unsigned short *red_orig, *green_orig, *blue_orig;
699 
700  switch (mInternalImage->bps)
701  {
702  case 1: case 2: case 4:
703  case 8: case 16:

860  if ( mInternalImage->Compression == COMPRESSION_OJPEG )

863  }
864

869  //@note using rcWindow since it is ref counted and will be deleted upon dtor

873  {
874
