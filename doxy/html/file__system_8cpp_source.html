127 bool file_specification::initialise_folder(const std::string& folder_spec)
128 {
129  std::string spec = folder_spec;
130  m_relative = true;
131  m_drive.erase();
132  m_path.clear();
133  m_filename.erase();
134  unsigned i = 0;
135 #if defined(MSWINDOWS) || defined(CYGWIN)
136  // first split off the drive letter or UNC prefix on Windows - the Cygwin environment supports these too
137  if (spec.size() >= 2 && isalpha(spec[0]) && spec[1] == ':')
138  {
139  // found a drive letter
140  i = 2;
141  m_drive = spec.substr(0, 2);
142  m_relative = false;
143  // if there is a drive but no path or a relative path, get the current path for this drive and prepend it to the path
144  if (i == spec.size() || !is_separator(spec[i]))
145  {
146 #ifdef MSWINDOWS
147  // getdcwd requires the drive number (1..26) not the letter (A..Z)
148  char path [MAX_PATH+1];
149  int drivenum = toupper(m_drive[0]) - 'A' + 1;
150  if (_getdcwd(drivenum, path, MAX_PATH+1))
151  {
152  // the path includes the drive so we have the drive info twice
153  // need to prepend this absolute path to the spec such that any remaining relative path is still retained
154  if (!is_separator(path[strlen(path)-1])) spec.insert(2, 1, preferred_separator);
155  spec.insert(2, path+2);
156  }
157  else
158 #endif
159  {
160  // non-existent drive - fill in just the root directory
161  spec.insert(2, 1, preferred_separator);
162  }
163  }
164  }
165  else if (spec.size() >= 2 && is_separator(spec[0]) && is_separator(spec[1]))
166  {
167  // found an UNC prefix
168  i = 2;
169  // find the end of the prefix by scanning for the next seperator or the end of the spec
170  while (i < spec.size() && !is_separator(spec[i])) i++;
171  m_drive = spec.substr(0, i);
172  m_relative = false;
173  }
174 #endif
175  // chk whether the path is absolute or relative and discard the leading / if absolute
176  if (i < spec.size() && is_separator(spec[i]))
177  {
178  m_relative = false;
179  i++;
180 #ifdef MSWINDOWS
181  // if there's no drive, fill it in on Windows since absolute paths must have a drive
182  if (m_drive.empty())
183  {
184  m_drive += (char)(_getdrive() - 1 + 'A');
185  m_drive += ':';
186  }
187 #endif
188  }
189  // now extract the path elements - note that a trailing / is not significant since /a/b/c/ === /a/b/c
190  // also note that the leading / has been discarded - all paths are relative
191  // if absolute() is set, then paths are relative to the drive, else they are relative to the current path
192  unsigned start = i;
193  while(i <= spec.size())
194  {
195  if (i == spec.size())
196  {
197  // path element terminated by the end of the string
198  // discard this element if it is zero length because that represents the trailing /
199  if (i != start)
200  m_path.push_back(spec.substr(start, i-start));
201  }
202  else if (is_separator(spec[i]))
203  {
204  // path element terminated by a separator
205  m_path.push_back(spec.substr(start, i-start));
206  start = i+1;
207  }
208  i++;
209  }
210  // TODO - some error handling?
211  return true;
212 }

235 bool file_specification::simplify(void)
236 {
237  // simplify the path by removing unnecessary . and .. entries - Note that zero-length entries are treated like .
238  for (unsigned i = 0; i < m_path.size(); )
239  {
240  if (m_path[i].empty() || m_path[i].compare(".") == 0)
241  {
242  // found . or null
243  // these both mean do nothing - so simply delete this element
244  m_path.erase(m_path.begin()+i);
245  }
246  else if (m_path[i].compare("..") == 0)
247  {
248  // found ..
249  if (i == 0 && !m_relative)
250  {
251  // up from the root does nothing so can be deleted
252  m_path.erase(m_path.begin()+i);
253  i++;
254  }
255  else if (i == 0 || m_path[i-1].compare("..") == 0)
256  {
257  // the first element of a relative path or the previous element is .. then keep it
258  i++;
259  }
260  else
261  {
262  // otherwise delete this element and the previous one
263  m_path.erase(m_path.begin()+i);
264  m_path.erase(m_path.begin()+i-1);
265  i--;
266  }
267  }
268  // keep all other elements
269  else
270  i++;
271  }
272  // TODO - error chking?
273  return true;
274 }

314 bool file_specification::make_relative(const file_specification& rootspec)
315 {
316  // test whether already an relative path in which case there's nothing to do
317  if (relative()) return true;
318  // initialise the result with the root and make the root absolute
319  file_specification absolute_root = rootspec;
320  absolute_root.make_absolute();
321  // now compare elements of the absolute root with elements of this to find the common path
322  // if the drives are different, no conversion can take place and the result must be absolute, else clear the drive
323  if (!path_compare(drive(), absolute_root.drive())) return true;
324  set_drive("");
325  // first remove leading elements that are identical to the corresponding element in root
326  unsigned i = 0;
327  while(subpath_size() > 0 && i < absolute_root.subpath_size() && path_compare(subpath_element(0), absolute_root.subpath_element(i)))
328  {
329  subpath_erase(0);
330  i++;
331  }
332  // now add a .. prefix for every element in root that is different from this
333  while (i < absolute_root.subpath_size())
334  {
335  m_path.insert(m_path.begin(), "..");
336  i++;
337  }
338  set_relative();
339  return true;
340 }

726 {
727  std::string dir = directory.empty() ? std::string(".") : directory;
728  std::vector<std::string> results;
729 #ifdef MSWINDOWS
730  std::string wildcard = create_filespec(dir, wild);

732  _finddata_t fileinfo;
733  for (bool OK = (handle = _findfirst((char*)wildcard.c_str(), &fileinfo)) != -1; OK; OK = (_findnext(handle, &fileinfo)==0))
734  {

736  if (strentry.compare(".")!=0 && strentry.compare("..")!=0)
737  if ((subdirs && (fileinfo.attrib & _A_SUBDIR)) || (files && !(fileinfo.attrib & _A_SUBDIR)))
738  results.push_back(strentry);
739  }
740  _findclose(handle);
741 #else
742  DIR* d = opendir(dir.c_str());
743  if (d)

746  {
747  std::string strentry = entry->d_name;
748  if (strentry.compare(".")!=0 && strentry.compare("..")!=0)
749  {
750  std::string subpath = create_filespec(dir, strentry);
751  if (((subdirs && is_folder(subpath)) || (files && is_file(subpath))) && (match_wildcard(wild, strentry)))
752  results.push_back(strentry);
753  }
754  }

757 #endif

795 static std::string full_path(const std::string& root, const std::string& path)
796 {
797  // convert path to a full path using root as the start point for relative paths
798  // decompose the path and test whether it is already an absolute path, in which case just return it
799  file_specification spec;
800  spec.initialise_folder(path.empty() ? std::string(".") : path);
801  if (spec.absolute()) return spec.image();
802  // okay, so the path is relative after all, so we need to combine it with the root path
803  // decompose the root path and chk whether it is relative
804  file_specification rootspec;
805  rootspec.initialise_folder(root.empty() ? std::string(".") : root);
806  if (rootspec.relative())
807  rootspec.make_absolute();
808  // Now do the conversion of the path relative to the root
809  spec.make_absolute(rootspec);
810  return spec.image();
811 }

814 {

816  // first convert both paths to full paths relative to CWD
817  file_specification rootspec;

819  if (rootspec.relative())

821  file_specification spec;

823  if (spec.relative())
824  spec.make_absolute();
825  // now make path spec relative to the root spec
826  spec.make_relative(rootspec);
827  return spec.image();

881 {
882  std::string fname = filename_part(spec);
883  // scan back through filename until a '.' is found and remove suffix
884  // the whole filename is the basename if there is no '.'

886  // observe Unix convention that a dot at the start of a filename is part of the basename, not the extension

970 std::string lookup (const std::string& command, const std::string& path, const std::string& splitter)
971 {
972  // first chk whether the command is already a path and chk whether it exists
973  if (!folder_part(command).empty())
974  {
975  if (file_exists(command))
976  return command;
977  }
978  else
979  {
980  // command is just a name - so do path lookup
981  std::vector<std::string> paths = split(path, splitter);
982  for (unsigned i = 0; i < paths.size(); i++)
983  {
984  std::string spec = create_filespec(paths[i], command);
985  if (file_exists(spec))
986  {
987  return spec;
988  }
989  }
990  }
991 #ifdef MSWINDOWS
992  // if there is no extension, try recursing on each possible extension
993  // TODO iterate through PATHEXT
994  if (extension_part(command).empty())
995  return lookup(create_filespec(folder_part(command), basename_part(command), "exe"), path, splitter);
996 #endif
997  // if path lookup failed, return empty string to indicate error
998  return std::string();
999 }
