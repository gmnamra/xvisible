16 /* rcPointCorrelation - Calculates correlation space for a model. Class is
17  * optimized for performing high speed calculations in certain cases
18  * by by both reading precalculated values from a set of integral
19  * images and performing a high speed calculation of the IM values for
20  * each location of the model within the search space.
21  *
22  * Current implementation is optimized for 3x3 and 5x5 cases where the
23  * search space dimensions are <= 16 pixels in both dimensions.
24  */

31  /* gen3by3Space - Fast generation of a 3x3 correlation space.
32  *
33  * modelOrigin specifies the origin of the model, in pixels,
34  * relative to the mdlWin specified in the last call to update().
35  *
36  * srchSpace sepcifies a rectangle, in pixels, relative to the
37  * imgWin specified in the last call to update().
38  *
39  * Correlation scores will be generated using integral images
40  * specified by the last call to update.
41  *
42  * See rsCorr3by3Point (in rc_moments.h) for a description of the
43  * result structure. The size of the model the operation is being
44  * performed upon always has dimensions srchSpace.width()-2 by
45  * srchSpace.height()-2.
46  *
47  * Unlike rcAutoCorrelation, the full correlation space is always
48  * generated.
49  *
50  * Note: the "moments" version of update() must be called at least
51  * once before any calls to this fct. If the other veriosn of
52  * update() gets called, the "moments" version must, once again, be
53  * called at least once before any calls to this fct. The search
54  * space must refer to an rcRect that fits within the last imgWin
55  * passed to update() and be at minimum, 3 by 3. Unlike
56  * rcAutoCorrelation, there is an additional restriction on the
57  * search space size that it must be less than 16 by 16.
58  *
59  */
60  void gen3by3Space(const rcRect& srchSpace, const rcIPair& modelOrigin,
61  rsCorr3by3Point& res) const;

63  /* gen5by5Space - Fast generation of a 5x5 correlation space.
64  *
65  * modelOrigin specifies the origin of the model, in pixels,
66  * relative to the mdlWin specified in the last call to update().
67  *
68  * srchSpace sepcifies a rectangle, in pixels, relative to the
69  * imgWin specified in the last call to update().
70  *
71  * See rsCorr5by5Point (in rc_moments.h) for a description of the
72  * result structure. The size of the model the operation is being
73  * performed upon always has dimensions srchSpace.width()-4 by
74  * srchSpace.height()-4.
75  *
76  * Unlike rcAutoCorrelation, the full correlation space is always
77  * generated.
78  *
79  * The fct will use the moment data specified by the last call to
80  * update, to generate SumI, SumII, SumM, and SumMM values as well
81  * as using the images specified by the last call to update to
82  * calculate SumIM values required to calculate correlation scores.
83  *
84  * Note: the "moments" version of update() must be called at least
85  * once before any calls to this fct. If the other veriosn of
86  * update() gets called, the "moments" version must, once again, be
87  * called at least once before any calls to this fct. The search
88  * space must refer to an rcRect that fits within the last imgWin
89  * passed to update() and be at minimum, 5 by 5. Unlike
90  * rcAutoCorrelation, there is an additional restriction on the
91  * search space size that it must be less than 16 by 16.
92  */
93  void gen5by5Space(const rcRect& srchSpace, const rcIPair& modelOrigin,
94  rsCorr5by5Point& res) const;

96  /* genSpace - Generate an NxN correlation space, where N = (1 +

98  *
99  * modelOrigin specifies the origin of the model, in pixels,
100  * relative to the mdlWin specified in the last call to update().
101  *
102  * srchSpace sepcifies a rectangle, in pixels, relative to the
103  * imgWin specified in the last call to update().
104  *
105  * The size of the model the operation is being performed upon
106  * always has dimensions srchSpace.width-(acRadius*2) by
107  * srchSpace.height()-(acRadius*2). acRadius must be > 0.
108  *
109  * Unlike rcAutoCorrelation, the full correlation space is always
110  * generated.
111  *
112  * The fct will use the moment data specified by the last call to
113  * update, to generate SumI, SumII, SumM, and SumMM values as well
114  * as using the images specified by the last call to update to
115  * calculate SumIM values required to calculate correlation scores.
116  *
117  * Note: update() must be called at least once before this function
118  * gets called. If the "moments" version of update() was the last
119  * to be called, the moments data will be used in generating the
120  * correlation space. The search space must refer to an rcRect that
121  * fits within the last imgWin passed to update() and be at minimum,
122  * N by N, where N = (1 + (acRadius*2).
123  */
124  void genSpace(const rcRect& srchSpace, const rcIPair& modelOrigin,
