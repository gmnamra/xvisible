182 rcVisualFunction::rcVisualFunction (rcKinetoscope* kin, const rcPolygon& poly,
183  const vector<rc2Fvector>& ends, const float rad)
184  : mKine (kin)
185 {
186  rmAssert (!mHaves.any());
187  rmAssert (!mState.any());
188  rmAssert (ends.size() == 2);
189  mRad = (int32) rad;
190 
191  initFunction ();
192  // Motion is detected using this frame and the previous frame
193  // Map using kinetoscopes time unit mapper
194  mGenesis = sharedKin()->secondsToUnits().mapVector (kin->startingAbsoluteTime().secs());
195  info (kin->organism(), kin->organismName ());
196 
197  mEstimatedShortening = gEstSideShortening; // an estimate to start things off
198  framesPerPeriod (kin->periodicMotion());
199 
200  // Create a polygon using the ellipse formed by the ends.
201  // Rotate and move it to the center of the line contecting the ends.
202  rc2Fvector m = (ends[0] + ends[1]) / 2.0f;
203  rc2Fvector d = ends[1] - ends[0];
204  if (d.isNull ())
205  {
206  rmExceptionMacro(<<"Length too short");
207  }
208  rc2Xform xf;
209  static const rcDPair noScale (1.0, 1.0);
210  rcMatrix_2d mx (d.angle(), noScale);
211  xf.matrix (mx);
212  xf.trans (m);
213 
214  float a =d.len (); a = a / 2.0f;
215  rcPolygon elip (a, mRad, 12);
216  elip.transform (xf);
217 
218  mMajorEnds = rcFPair (0.0f, a + a);
219  for (uint32 i = 0; i < 2; i++) mMajorEndPoints[i] = ends[i];
220  setState (eHasFullyExtended);
221  mMidPoint = m;
222 
223 
224  // Now find the union poly of the two
225  // Cell's registration point is with respect to the orthogonal rect
226  // Initial position
227  rcFRect fr;
228  elip.orthogonalEnclosingRect(fr);
229  mMotionCtr = m;
230  anchor (mMotionCtr);
231  mRectSnaps.push_back (fr);
232  const rc2Xform& xform = sharedKin()->xform();
233  mCurrentPosition = xform.mapPoint (m);
234 
235  // Keep a copy of the polygon
236  mPOLYs.push_back (elip);
237  mAffys.push_back (elip.minimumPerimeterEnclosingRect());
238 
239  // In production sharedKin is ready. In testing this may not be true
240  // @note: should this be on for cardiac ?
241  if (sharedKin()->canStep () && (sharedKin()->have (rcKinetoscope::eLineage)))
242  {
243  initializeSimilarator (mRectSnaps.back());
244  }
245  setState (rcVisualFunction::eIsInitialized);

249 void rcVisualFunction::createIrradianceModel ()
250 {
251  // Create a Gauss Model. Currently only for catching daughters after mitosis
252  // Default is a gaussian with std of 1.0
253  static const int32 dummy (0);
254  int32 d = rfRound (sqrt (mRectSnaps.back().area() / rkPI), dummy);
255  d = rmMax (16, d);
256  rcPixel depth (rcPixel8);
257  rcWindow tmp (d, d, depth);
258  mIradModel.gauss (tmp, rcMathematicalImage::eHiPeak);
259  mGaussModel = rcCorrelationWindow<uint8> (tmp);

282 bool rcVisualFunction::updateSims (const rcWindow& win, const rcFRect& rect)
283 {
284  rmAssert (isState (eHasSimilarator));
285  clear (eCoarseSS);
286 
287  rcFRect pm, fm; rcIRect bm, im;
288  static const rcFPair stationary (0.0f, 0.0f);
289  static const bool doClip (true);
290 
291  // Get the current and estimated rect positons, window the est and setup mask if size has changed
292  // Note that the current rect might have been outside itself. We check the inteded size with estiamted
293  // window size, and act on clipping done or not.
294  // rcFPair ds = rect.size() - mCurrentSimRect.size();
295  //rcFPair dp = rect.ul() - mCurrentSimRect.ul();
296 
297  updateRects (rect, stationary, fm, im);
298  updateRects (mCurrentSimRect, stationary, pm, bm);
299  rcWindow tmp (win, im, doClip);
300 
301  if (tmp.size () == bm.size() && tmp.size() == im.size() && im.size() == mCoarseSimRef->fillImageSize() &&
302  (mCoarseSimRef->update (tmp) || steps() < mCoarseSimRef->matrixSz()))
303  {
304  mark(eCoarseSS);
305  mCurrentSimRect = fm;
306  return true;
307  }
308  else
309  {
310 
311  rcWindow content (win, im.ul().x(), im.ul().y(), rmMin (tmp.width(), bm.width()), rmMin (tmp.height(), bm.height()));
312  rcWindow extended (bm.width(), bm.height());
313  rcWindow masked (bm.width(), bm.height());
314  extended.setAllPixels (0);
315  masked.setAllPixels (255);
316  rcWindow exw (extended, content.width(), content.height());
317  rcWindow mw (extended, content.width(), content.height());
318  exw.copyPixelsFromWindow (content);
319  mw.setAllPixels (0);
320 
321  if (extended.size() == bm.size() && masked.size() == bm.size())
322  {
323  mCoarseSimRef->clearMask ();
324  mCoarseSimRef->setMask (masked);
325  if (mCoarseSimRef->update (extended) || steps() < mCoarseSimRef->matrixSz())
326  {
327  mark(eCoarseSS);
328  mCurrentSimRect = fm;
329  return true;
330  }
331  }
332  }
333 
334  // @note for now only one similarator
335  return false;

350 float rcVisualFunction::framesPerPeriod (float p)
351 {
352  clearState (eIsPeriodic);
353  mFramesPerPeriod = p;
354  if (p > 0.0) setState (eIsPeriodic);
355  mIntegralFramesPerPeriod = rfRoundPlus (framesPerPeriod (), sDummy);
356 
357  // Use a window (up adjusted to power of 2) of two periods to estimate contraction frequency
358  frexp (p+p, &mFramesPerPeriodPowerOf2);
359  mFramesPerPeriodPowerOf2 = (int32) ldexp (1.0f, mFramesPerPeriodPowerOf2);
360  mBeatAC.resize (mFramesPerPeriodPowerOf2);
361  return p;
362 }

396 double rcVisualFunction::computeAcceleration ()
397 {
398  mAcceleration = 0.0; //gInValidAcceleration;
399  clear (eAcceleration);
400  if (have (eCoarseSS))
401  {
402  deque<double> signal;
403  bool hasSignal = mCoarseSimRef->entropies (signal);
404 
405  if (hasSignal && signal.size())
406  {
407  rmAssert (signal.size () == mCoarseSimRef->matrixSz ());
408  mSS.push_back (signal[mCoarseSimRef->matrixSz () / 2]);
409 
410  if (mSS.size() > (uint32) ssWindowSize ())
411  {
412  mAcceleration = log10(mSS[mSS.size() - 1] / mSS[mSS.size() - 2]);
413  mAcl.push_back ((float) acceleration ());
414  mark (eAcceleration);
415  }
416  }
417  }
418  return mAcceleration;
419 }
