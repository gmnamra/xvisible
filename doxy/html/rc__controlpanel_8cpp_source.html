429 static const double cSecsPerHour = cSecsPerMinute * 60.0;

433 QString rcControlPanel::timestampToString( const rcTimestamp& time )
434 {
435  double dSecs = time.secs();
436  int nDays = int(dSecs / cSecsPerDay);
437  dSecs -= cSecsPerDay * nDays;
438  int nHours = int(dSecs / cSecsPerHour);
439  dSecs -= cSecsPerHour * nHours;
440  int nMinutes = int(dSecs / cSecsPerMinute);
441  dSecs -= cSecsPerMinute * nMinutes;
442  int nSecs = int(dSecs);
443  dSecs -= nSecs;
444  int nMillis = int(dSecs * cMilliSecsPerSec);
445  QString string;
446  if (nDays == 0)
447  string.sprintf( "%02d:%02d:%02d.%03d" , nHours , nMinutes , nSecs , nMillis );
448  else
449  string.sprintf( "%dd %02d:%02d:%02d.%03d" , nDays , nHours , nMinutes , nSecs , nMillis );
450  return string;
451 }
