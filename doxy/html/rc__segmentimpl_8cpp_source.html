137 // advance the iterator by the specified amount (new implementation).
138 void rcSegmentIteratorImpl::advanceNoLock( const rcTimestamp& amount )
139 {
140  const int nSegments = _track->segments();
141 
142  if ( nSegments > 0 ) {
143  // First advance call
144  if ( _segmentIndex < 0 ) {
145  _position = cZeroTime;
146  _segmentOffset = cZeroTime;
147  _segmentIndex = 0;
148  _currentSegment = _track->segment( _segmentIndex );
149  }
150 
151  // Sequential advance call
152  if ( amount > cZeroTime ) {
153  rcTimestamp start = _currentSegment.start();
154  rcTimestamp target = _segmentOffset + amount;
155  if ( _position >= start )
156  target += start;
157  // Find closest segment
158  _currentSegment = findClosestSegment( target, _segmentIndex );
159  start = _currentSegment.start();
160  const rcTimestamp end = start + _currentSegment.length();
161 
162  if ( target < start ) {
163  // Requested time is before start of track
164  _position = target;
165  _segmentOffset = target;
166  } else {
167  if ( target < end ) {
168  // Requested time is within a segment
169  _position = target;
170  _segmentOffset = _position - start;
171  }
172  else if ( target > end ) {
173  // Requested time is after end of track
174  _position = end;
175  _segmentOffset = cZeroTime;
176  } else {
177  // Requested time is on a segment boundary
178  _position = target;
179  _segmentOffset = cZeroTime;
180  }
181  }
182  }
183  }

197 // Advance by count frame and round _position to a segment boundary
198 bool rcSegmentIteratorImpl::advanceNoLock( int32 count )
199 {
200  int32 nSegments = _track->segments();
201  int32 newIndex = _segmentIndex + count;
202 
203  if ( newIndex >= nSegments ) {
204  // Can't advance forward any more
205  return false;
206  }
207  else if ( newIndex < 0 ) {
208  // Can't advance backwards any more
209  return false;
210  }
211  _currentSegment = _track->segment( newIndex );
212  _segmentOffset = cZeroTime;
213  _position = _currentSegment.start();
214  _segmentIndex = newIndex;

275 // Return segment that contains (or is closest to) time
276 // Binary search
277 rcSegmentImpl rcSegmentIteratorImpl::findClosestSegment( const rcTimestamp& time,
278  int32& index )
279 {
280  index = -1;
281  const int32 nSegments = _track->segments();
282 
283  if ( nSegments > 0 ) {
284  int32 first = 0;
285  int32 last = nSegments - 1;
286 
287  // Binary search the segments
288  while ( first <= last ) {
289  int32 mid = (first + last) / 2;
290  rcSegmentImpl currentSegment = _track->segment( mid );
291  rcTimestamp start = currentSegment.start();
292 
293  if ( time < start ) {
294  last = mid - 1;
295  }
296  else if ( time > start ) {
297  rcTimestamp end = start + currentSegment.length();
298  if ( time < end ) {
299  index = mid;
300  return currentSegment;
301  }
302  first = mid + 1;
303  }
304  else {
305  index = mid;
306  return currentSegment;
307  }
308  }
309  if ( last < 0 )
310  index = 0; // Time is before start of track
311  else
312  index = last; // Time is after end of track
313 
314  return _track->segment( index );
315  }
316 
317  rmAssert( nSegments > 0 );
318  // Error, this should never happen
319  return _currentSegment;
320 }
