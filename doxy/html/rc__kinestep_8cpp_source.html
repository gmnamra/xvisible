124 void rcKinetoscope::preSegmentedBodies (const vector< vector <rc2Fvector> >& bodyEnds, vector<float>& rads)
125 {
126  if (bodyEnds.size() == 0 || bodyEnds.size() != rads.size())
127  {
128  rmExceptionMacro(<<"Inconsistent Selection Data");
129  }
130 
131  createFunctionSpecificObjects ();
132 
133  if (mCount <= startFrame())
134  {
135  for (uint32 i = 0; i < bodyEnds.size(); i++)
136  {
137  vector<rc2Fvector> ends = bodyEnds[i];
138  // Create a polygon using the ellipse formed by the ends.
139  // Rotate and move it to the center of the line contecting the ends.
140  rc2Fvector m = (ends[0] + ends[1]) / 2.0f;
141  rc2Fvector d = ends[1] - ends[0];
142  if (d.isNull ())
143  {
144  rmExceptionMacro(<<"Length too short");
145  }
146 
147  float a =d.len ();
148  a = a / 2.0f;
149  rcPolygon poly (a, rads[i], 12);
150  static const rcDPair noscale (1.0, 1.0);
151  rc2Xform xt (rcMatrix_2d (d.angle (), noscale), m);
152  poly.transform (xt);
153  mPrePOLYs.push_back (poly);
154  mBodyEnds.push_back (ends);
155  mBodyRads.push_back(rads[i]);
156  }
157  mark (eSelectedMyo);
158  }
159 }
