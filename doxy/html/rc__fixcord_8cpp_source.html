40 /* To rotate a vector through an angle of theta, we calculate:
41  *
42  * x' = x cos(theta) - y sin(theta)
43  * y' = x sin(theta) + y cos(theta)
44  *
45  * The rotate() routine performs multiple rotations of the form:
46  *
47  * x[i+1] = cos(theta[i]) { x[i] - y[i] tan(theta[i]) }
48  * y[i+1] = cos(theta[i]) { y[i] + x[i] tan(theta[i]) }
49  *
50  * with the constraint that tan(theta[i]) = pow(2, -i), which can be
51  * implemented by shifting. We always shift by either a positive or
52  * negative angle, so the convergence has the ringing property. Since the
53  * cosine is always positive for positive and negative angles between -90
54  * and 90 degrees, a predictable magnitude scaling occurs at each step,
55  * and can be compensated for instead at the end of the iterations by a
56  * composite scaling of the product of all the cos(theta[i])'s.
57  */
58

100 static void PseudoRotate(int32 *px, int32 *py, register int32 theta)
101 {
102  register int i;
103  register int32 x, y, xtemp;
104  register int32 *arctanptr;
105 
106  x = *px;
107  y = *py;
108 
109  /* Get angle between -90 and 90 degrees */
110  while (theta < -QUARTER) {
111  x = -x;
112  y = -y;
113  theta += 2 * QUARTER;
114  }
115  while (theta > QUARTER) {
116  x = -x;
117  y = -y;
118  theta -= 2 * QUARTER;
119  }
120 
121  /* Initial pseudorotation, with left shift */
122  arctanptr = arctantab;
123  if (theta < 0) {
124  xtemp = x + (y << 1);
125  y = y - (x << 1);
126  x = xtemp;
127  theta += *arctanptr++;
128  }
129  else {
130  xtemp = x - (y << 1);
131  y = y + (x << 1);
132  x = xtemp;
133  theta -= *arctanptr++;
134  }
135 
136  /* Subsequent pseudorotations, with right shifts */
137  for (i = 0; i <= MAXITER; i++) {
138  if (theta < 0) {
139  xtemp = x + (y >> i);
140  y = y - (x >> i);
141  x = xtemp;
142  theta += *arctanptr++;
143  }
144  else {
145  xtemp = x - (y >> i);
146  y = y + (x >> i);
147  x = xtemp;
148  theta -= *arctanptr++;
149  }
150  }
151 
152  *px = x;
153  *py = y;
154 }

157 static void PseudoPolarize(int32 *argx, int32 *argy)
158 {
159  register int32 theta;
160  register int32 yi, i;
161  register int32 x, y;
162  register int32 *arctanptr;
163 
164  x = *argx;
165  y = *argy;
166 
167  /* Get the vector into the right half plane */
168  theta = 0;
169  if (x < 0) {
170  x = -x;
171  y = -y;
172  theta = 2 * QUARTER;
173  }
174 
175  if (y > 0)
176  theta = - theta;
177 
178  arctanptr = arctantab;
179  if (y < 0) { /* Rotate positive */
180  yi = y + (x << 1);
181  x = x - (y << 1);
182  y = yi;
183  theta -= *arctanptr++; /* Subtract angle */
184  }
185  else { /* Rotate negative */
186  yi = y - (x << 1);
187  x = x + (y << 1);
188  y = yi;
189  theta += *arctanptr++; /* Add angle */
190  }
191 
192  for (i = 0; i <= MAXITER; i++) {
193  if (y < 0) { /* Rotate positive */
194  yi = y + (x >> i);
195  x = x - (y >> i);
196  y = yi;
197  theta -= *arctanptr++;
198  }
199  else { /* Rotate negative */
200  yi = y - (x >> i);
201  x = x + (y >> i);
202  y = yi;
203  theta += *arctanptr++;
204  }
205  }
206 
207  *argx = x;
208  *argy = theta;
209 }

213 /* FxPreNorm() block normalizes the arguments to a magnitude suitable for
214  * CORDIC pseudorotations. The returned value is the block exponent.
215  */
216 static int
217 FxPreNorm(int32 *argx, int32 *argy)
218 {
219  register int32 x, y;
220  int signx, signy;
221  register int shiftexp;
222 
223  shiftexp = 0; /* Block normalization exponent */
224  signx = signy = 1;
225 
226  if ((x = *argx) < 0) {
227  x = -x;
228  signx = -signx;
229  }
230  if ((y = *argy) < 0) {
231  y = -y;
232  signy = -signy;
233  }
234  /* Prenormalize vector for maximum precision */
235  if (x < y) { /* |y| > |x| */
236  while (y < (1 << 27)) {
237  x <<= 1;
238  y <<= 1;
239  shiftexp--;
240  }
241  while (y > (1 << 28)) {
242  x >>= 1;
243  y >>= 1;
244  shiftexp++;
245  }
246  }
247  else { /* |x| > |y| */
248  while (x < (1 << 27)) {
249  x <<= 1;
250  y <<= 1;
251  shiftexp--;
252  }
253  while (x > (1 << 28)) {
254  x >>= 1;
255  y >>= 1;
256  shiftexp++;
257  }
258  }
259 
260  *argx = (signx < 0) ? -x : x;
261  *argy = (signy < 0) ? -y : y;
262  return(shiftexp);
263 }
264 #endif FASTER

278 /* Fxrotate() rotates the vector (argx, argy) by the angle theta. */
279 void rfFxRotate(int32 *argx, int32 *argy, int32 theta)
280 {
281 #ifndef FASTER
282  int shiftexp;
283 #endif FASTER
284 
285  if (((*argx == 0) && (*argy == 0)) || (theta == 0))
286  return;
287 
288 #ifndef FASTER
289  shiftexp = FxPreNorm(argx, argy); /* Prenormalize vector */
290 #endif FASTER
291  PseudoRotate(argx, argy, theta); /* Perform CORDIC pseudorotation */
292  *argx = FFracMul(*argx, COSCALE); /* Compensate for CORDIC enlargement */
293  *argy = FFracMul(*argy, COSCALE);
294 #ifndef FASTER
295  if (shiftexp < 0) { /* Denormalize vector */
296  *argx >>= -shiftexp;
297  *argy >>= -shiftexp;
298  }
299  else {
300  *argx <<= shiftexp;
301  *argy <<= shiftexp;
302  }
303 #endif FASTER
304 }
