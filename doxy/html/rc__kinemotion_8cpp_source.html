163 //
164 // log2max
165 //
166 // This function computes the ceiling of log2(n). For example:
167 //
168 // log2max(7) = 3
169 // log2max(8) = 3
170 // log2max(9) = 4
171 //
173 static int32 log2max(int32 n)
174 {
175  int32 power = 1;
176  int32 k = 1;
177 
178  if (n==1) {
179  return 0;
180  }
181 
182  while ((k <<= 1) < n) {
183  power++;
184  }
185 
186  return power;
187 }
