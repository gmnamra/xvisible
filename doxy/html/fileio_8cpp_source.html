47 void ofbuff::open(const std::string& fname, size_t bufsize, otext::open_t mode, bool line_buffer)
48 {
49  // open a managed file
50  if (fname.empty())
51  {
52  // an empty filename is an error without even trying to open the file
53  // on Windows fopen raises an assertion failure in this case which is unacceptable so I trap it first
54  set_error(textio_open_failed);
55  }
56  else
57  {
58  m_managed = true;
59  // open in binary mode so that *I* have control over newline conventions
60  m_handle = fopen(fname.c_str(),(mode==otext::overwrite ? "wb" : "ab"));
61  if (!m_handle)
62  {
63  if (errno)
64  set_error(errno);
65  else
66  set_error(textio_open_failed);
67  }
68  else if (setvbuf(m_handle,0,(bufsize ? _IOFBF : _IONBF),bufsize) != 0)
69  {
70  int error = ferror(m_handle);
71  if (error)
72  set_error(error);
73  else
74  set_error(textio_open_failed);
75  }
76  }
77  set_line_buffer(line_buffer);
78 }

231 int ifbuff::get (void)
232 {
233  if (!m_handle) return -1;
234  int ch = getc(m_handle);
235  if (ch == EOF)
236  {
237  // may be a stdio error or the user may be trying to read past the last valid character
238  // in the latter case it is not actually an error - it is perfectly legal to read the EOF
239  int error = ferror(m_handle);
240  if (error)
241  set_error(error);
242  }
243  return ch;
244 }
