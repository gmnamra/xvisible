91  /*
92  * effect Maps the point by the xform: result = scale * pt + offset.
93  * note Both mapPoint and '*' map the input like a full 1D-point.
94  * (i.e. a location on the 1-D number line.)
95  */
96 
97  double invMapPoint (double pt) const { return (inverse() * pt); }
98  /*
99  * effect Maps the point by the inverse xform:
100  * result = 1/scale * (pt - offset)

104  // The following two map the input like a vector.
105  // (i.e. the input is an 'unlocated' length along the 1-D number line.)
106  double mapVector (double vect) const { return (mScale * vect); }
107  /*
108  * effect Scales the vector: result = scale * vect
109  */
110  inline double invMapVector (double vect) const

112  if (isSingular()) throw general_exception ("xform is Singular");

114  }

118  */

132  */

272 // Mapping vectors (1,0), and (0,1) in to the new vector,
