483 {

490  if (scalarTrack == 0)

492 
493  strstream s;

495  bool nextSegment = true;
496  uint32 values = 0;
497  // Output with high accuracy

500 
501  //@note create util function or use boost

504  std::string iname = domain->getExperimentAttributes().inputName;

513  s << "{" << experimentName << "}" << endl;
514  }
515  else
516  s << experimentName << endl << track->getName() << endl;
517 
518  if ( iter->getSegmentIndex() == -1 ) {

522

531

533  iter->lock();
534  // Copy all values separated by a newline
535  while ( nextSegment ) {
