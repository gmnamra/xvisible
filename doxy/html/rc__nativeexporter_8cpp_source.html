164 uint32 rcNativeExporter::produceSettings( rcXMLElementTree& tree,
165  const rcXMLElementTree::iterator& curNode ) const
166 {
167  uint32 errors = 0;
168  uint32 nCategories = mExperiment->getNSettingCategories();
169 
170  if ( nCategories > 0 ) {
171  // Add comment
172  rcXMLElement commentElem( eXMLElementComment, cSettingCategoryComment );
173  tree.append_child( curNode, commentElem );
174  }
175 
176  // Output all settings
177  for ( uint32 i = 0; i < nCategories; i++ )
178  {
179  rcSettingCategory category = mExperiment->getSettingCategory( i );
180  uint32 nSettings = category.getNSettings();
181  uint32 persistableCount = 0;
182 
183  // Count the number of persistable setting in this category
184  for ( uint32 j = 0; j < nSettings; j++ )
185  {
186  rcSettingInfo setting = category.getSettingInfo( j );
187  if ( setting.isPersistable() )
188  ++persistableCount;
189  }
190 
191  if ( persistableCount > 0 ) {
192  rcXMLElement categoryElem( eXMLElementExperimentSettingCategory );
193  rcXMLElementTree::iterator categoryNode = tree.append_child( curNode, categoryElem );
194  // Add category name and description
195  rcXMLElement nameElem( eXMLElementName, category.getName() );
196  rcXMLElement descriptionElem( eXMLElementDescription, category.getDescription() );
197  tree.append_child( categoryNode, nameElem );
198  tree.append_child( categoryNode, descriptionElem );
199 
200  for ( uint32 j = 0; j < nSettings; j++ )
201  {
202  rcSettingInfo setting = category.getSettingInfo( j );
203 
204  // Persistability check
205  if ( setting.isPersistable() ) {
206  rcXMLElement settingElem( eXMLElementSetting );
207  const rcXMLElementTree::iterator settingNode = tree.append_child( categoryNode, settingElem );
208  // Add setting name and value
209  rcXMLElement nameElem( eXMLElementName, setting.getTag() );
210  rcXMLElement valueElem( eXMLElementValue, setting.getValue() );
211 
212  tree.append_child( settingNode, nameElem );
213  tree.append_child( settingNode, valueElem );
214  }
215  }
216  }
217  }
218 
219  return errors;
220 }

329  tree.append_child( trackNode, analysisAreaElem );

331

333

335  // Produce scalar-specific elements
336  rcScalarTrack* scalarTrack = dynamic_cast<rcScalarTrack*>( track );
337  if ( scalarTrack != 0 ) {
338  rcValue min( scalarTrack->getExpectedMinimumValue() );

355 
356  // Expected min/max

369  cerr << "rcNativeExporter internal error: graphics track version " << rcGraphicsTrack::ePersistenceVersion;
370  cerr << " is different from graphics token version " << rcVisualSegment::ePersistenceVersion << endl;
371  ++errors;
372  }
373  }
374 
375  // Persistence version number
376  rcValue version( versionNumber );

378  tree.append_child( trackNode, versionElem );
379  }

421 // Add all analysis result elements to document tree
422 // Return number of errors
423 uint32 rcNativeExporter::produceResults( rcXMLElementTree& tree,
424  const rcXMLElementTree::iterator& curNode,
425  double completedPercentage, double maxPercentage )
426 {
427  uint32 errors = 0;
428  uint32 processedElements = 0;
429 
430  if ( mExportIterators.size() > 0 ) {
431  rcExperimentDomain* domain = rcExperimentDomainFactory::getExperimentDomain();
432  rcTimestamp experimentStart = domain->getExperimentStart();
433 
434  // Add comment
435  rcXMLElement commentElem( eXMLElementComment, cValueComment );
436  tree.append_child( curNode, commentElem );
437 
438  // Produce values for each track
439  rcTimestamp current = cZeroTime;
440  vector<rcAttribute> trackIdAttributes( mExportIterators.size() );
441  uint32 activeTracks = 0;
442 
443  // Time mapper
444  rcXMLTimeMapper timeMap( mTimes );
445 
446  // Precompute some values
447  for ( uint32 i = 0; i < mExportIterators.size(); i++ ) {
448  // Track id attribute
449  const rcAttribute trackAttribute( eXMLAttributeTrackId,
450  produceIdString( eXMLAttributeTrackId, i ) );
451  trackIdAttributes[i] = trackAttribute;
452  if ( mExportActiveIterators[i] ) {
453  rcSegmentIterator* iter = mExportIterators[ i ];
454  // Deactivate empty tracks
455  if ( iter->getSegmentCount() > 0 )
456  ++activeTracks;
457  else
458  mExportActiveIterators[i] = false;
459  }
460  }
461 
462  if ( activeTracks > 0 )
463  {
464  // Iterate all the active tracks
465  for ( uint32 a = 0; a < mTimes.size(); ++a ) {
466  rcTimestamp time = mTimes[a] - experimentStart;
467  // Init all iterators with the correct start time
468  init( time );
469  // Current absolute time
470  current = mTimes[a];
471 
472  // Time id attribute
473  int idx = a;
474 
475  if ( idx < 0 ) {
476  ostrstream timestr;
477  timestr << current << ends;
478  timestr.freeze( false ); // Without freeze(), strstream::str() leaks the string it returns
479  fprintf( stderr, "rcNativeExporter error: time value %s cannot be mapped to index\n",
480  timestr.str());
481  ++errors;
482  } else {
483  rmAssertDebug( current == timeMap.timeValue( idx ) );
484  if ( time < cZeroTime ) {
485  ostrstream timestr;
486  timestr << current << ends;
487  timestr.freeze( false ); // Without freeze(), strstream::str() leaks the string it returns
488  fprintf( stderr, "rcNativeExporter error: current time value %s is negative\n",
489  timestr.str());
490  ++errors;
491  }
492  }
493  rcAttribute timeIdAttribute( eXMLAttributeTimeId,
494  produceIdString( eXMLAttributeTimeId, idx ) );
495  // Produce frame element
496  rcAttributeList attrs;
497  attrs.push_back( timeIdAttribute );
498  rcXMLElement frameElem( eXMLElementFrame, attrs );
499  const rcXMLElementTree::iterator& timeNode = tree.append_child( curNode, frameElem );
500 
501  // Output all the values
502  for ( uint32 i = 0; i < mExportIterators.size(); i++)
503  {
504  if ( mExportActiveIterators[i] ) {
505  rcSegmentIterator* iter = mExportIterators[ i ];
506 
507  // Skip if iterator doesn't point to a valid segment
508  if ( !iter->contains( time ) )
509  continue;
510 
511  rcAttributeList attrs;
512  // Track id attribute
513  attrs.push_back( trackIdAttributes[i] );
514 
515  // Get value
516  ostrstream str;
517  rcTrackType type = mExportTracks[i]->getTrackType();
518  if ( !produceValue( str, iter, type, eExperimentNativeFormat ) ) {
519  fprintf( stderr, "rcNativeExporter error: produceValue() failed\n" );
520  ++errors;
521  }
522  str << ends;
523  str.freeze( false ); // Without freeze(), strstream::str() leaks the string it returns
524  // Produce value element
525  rcXMLElement valueElem( eXMLElementTrackValue, attrs, std::string( str.str() ) );
526  tree.append_child( timeNode, valueElem );
527 
528  if ( mProgress ) {
529  if ( !(processedElements % 256)) {
530  // Progress range is [completedPercentage-maxPercentage] %
531  double completed = 1.0 * processedElements / mValuesToExport;
532  mProgress->progress( completedPercentage + (maxPercentage-completedPercentage) * completed);
533  }
534  ++processedElements;
535  }
536  }
537 
538  }
539  }
540  }
541  }
542 
543  return errors;
544 }
