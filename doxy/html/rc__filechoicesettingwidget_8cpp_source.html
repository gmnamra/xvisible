217 void rcFileChoiceSettingWidget::browseRequest( void )
218 {
219  switch (mSetting.getDisplayType())
220  {
221  case eFileChoice: // setting is a file name (text + "browse" button)
222  singleFileBrowse();
223  break;
224 
225  case eMultiFileChoice: // setting is a semicolon separated list of file names (text + "browse" button)
226  multiFileBrowse();
227  break;
228 
229  case eDirectoryChoice: // setting is a directory name (text + "browse" button)
230  directoryBrowse4Tiffs();
231  break;
232 
233  default:
234  rmExceptionMacro ( "unknown setting type" );
235  }
