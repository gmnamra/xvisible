17 rcMuscleTracker::rcMuscleTracker(uint32 smoothingRadius)
18  : _filterOffset(4), _stdDev(4.0),
19  _persistPct(.6), _smoothing(smoothingRadius*2 + 1)
20 {
21  uint32 curValue = 1;
22  uint32 curSum = 0;
23  uint32 backI = smoothingRadius*2;
24  uint32 frtI = 0;
25  while (frtI < smoothingRadius) {
26  _smoothing[frtI++] = _smoothing[backI--] = curValue;
27  curValue <<= 1;
28  curSum += curValue;
29  }
30  _smoothing[frtI] = curValue;
31  curSum += curValue;
32 
33  _smoothingNorm = curSum >> 1;
34 
35  setEdgeFilter(4, 0);
36 }

70  const uint32 height = images[0].height();

72  rcWindow hProj(height, 1, rcPixel32);
73 
74  /* The following vectors will hold the number of times a strong edge
75  * is found at each vertical/horizontal position. The array names
76  * are chosen to reflect the fact that each position is a potential
77  * candidate location for a cell edge.
78  */
79  if (_vCandPosDist.size() != width)
80  _vCandPosDist.resize(width);
81  if (_vCandNegDist.size() != width)
82  _vCandNegDist.resize(width);
83  for (uint32 i = 0; i < width; i++)
84  _vCandPosDist[i] = _vCandNegDist[i] = 0;
85 
86  if (_hCandPosDist.size() != height)
87  _hCandPosDist.resize(height);
88  if (_hCandNegDist.size() != height)
89  _hCandNegDist.resize(height);
90  for (uint32 i = 0; i < height; i++)
91  _hCandPosDist[i] = _hCandNegDist[i] = 0;
92 
93  /* Place to store the indices of frames that can be used as a source
94  * of model pixels.
95  */
96  vector<uint32> vCandPosFrmNum(width), vCandNegFrmNum(width);
97 
98  rcRect loc(0, 0, width, height);
99 
100  fprintf(stderr, "loc x %d y %d w %d h %d fi x %d y %d w %d h %d\n",
101  loc.x(), loc.y(), loc.width(), loc.height(),
102  images[0].x(), images[0].y(),
103  images[0].width(), images[0].height());
104 
105  /* Generate the frequency distribution of strong "positive" and
106  * "negative" edges.
107  */
108  for (uint32 i = images.size() - 1; i != 0xFFFFFFFF; i--) {
109  if ( progressIndicator )
110  if ( progressIndicator->progress( 100.0 * (images.size() - i)/images.size() ) )
111  return false;
112  mom.update(images[i]);
113  images[i].frameBuf().unlock();
114  mom.vProject(loc, vProj);
115  mom.hProject(loc, hProj);
116 
117  if (showDiags)
118  (*showDiags)(vProj, hProj);
119 
120  vector<rcEdgeFilter1DPeak> vPeaks, hPeaks;
121 
122  /* Generate edge points that have a strength value >= _StdDev
123  * std. dev. of norm.
124  */
125  edgeGen.genPeaks(vProj, 0, vPeaks, _filterOffset, 0, _stdDev);
126 
127  /* Now update running history of peaks found at each location,
128  * both in the vertical and horizontal dimensions. Also store
129  * indice of frame where, given a particular location is chosen,
130  * model pixels may be found. Note that this is only needed in the
131  * vertical direction, since it assumed that cells only move along
132  * the horizontal axis.
133  */
134  for (uint32 pIndex = 0; pIndex < vPeaks.size(); pIndex++) {
135  const int32 curLoc = vPeaks[pIndex]._location;
136  if (vPeaks[pIndex]._value > 0) {
137  _vCandPosDist[curLoc]++;
138  vCandPosFrmNum[curLoc] = i;
139  }
140  else {
141  _vCandNegDist[curLoc]++;
142  vCandNegFrmNum[curLoc] = i;
143  }
144  }
145 
146  /* Lather, Rinse, and Repeat in the horizontal direction.
147  */
148  edgeGen.genPeaks(hProj, 0, hPeaks, _filterOffset, 0, _stdDev);
149 
150  for (uint32 pIndex = 0; pIndex < hPeaks.size(); pIndex++) {
151  const int32 curLoc = hPeaks[pIndex]._location;
152  if (hPeaks[pIndex]._value > 0)
153  _hCandPosDist[curLoc]++;
154  else
155  _hCandNegDist[curLoc]++;
156  }
157  }
158 
159  /* The following heurisitic is used to calculate the edge positions
160  * that will form the boundary of the cell:
161  *
162  * 1) Edge persistence is used to cull out transient edges. The
163  * previously calculated frequency distribution is used to
164  * determine this. An edge will be considered to be persisitent
165  * if it was found at least _persistPct percent of the time.
166  *
167  * 2) It is assumed that the scene is not confusing, so the
168  * outermost persistent edges will be chosen as the final cell
169  * limits.
170  */
171  uint32 pCutOff = (uint32)(images.size()*_persistPct);
172  uint32 leftEdge = 0xFFFFFFFF, rightEdge = 0;
173  uint32 topEdge = 0xFFFFFFFF, bottomEdge = 0;
174 
175  for (uint32 x = 0; x < width; x++)
176  if (_vCandPosDist[x] > pCutOff) {
177  leftEdge = x;
178  _leftFrameIndex = vCandPosFrmNum[x];
179  break;
180  }
181  else if (_vCandNegDist[x] > pCutOff) {
182  leftEdge = x;
183  _leftFrameIndex = vCandNegFrmNum[x];
184  break;
185  }
186 
187  for (uint32 x = width; x-- > 0; )
188  if (_vCandPosDist[x] > pCutOff) {
189  rightEdge = x;
190  _rightFrameIndex = vCandPosFrmNum[x];
191  break;
192  }
193  else if (_vCandNegDist[x] > pCutOff) {
194  rightEdge = x;
195  _rightFrameIndex = vCandNegFrmNum[x];
196  break;
197  }
198 
199  for (uint32 y = 0; y < height; y++)
200  if ((_hCandPosDist[y] > pCutOff) || (_hCandNegDist[y] > pCutOff)) {
201  topEdge = y;
202  break;
203  }
204 
205  for (uint32 y = height; y-- > 0; )
206  if ((_hCandPosDist[y] > pCutOff) || (_hCandNegDist[y] > pCutOff)) {
207  bottomEdge = y;
208  break;
209  }
210 
211  if ((leftEdge >= rightEdge) || (topEdge >= bottomEdge)) {
212  printf("Couldn't find cell boundaries l %d r %d t %d b %d\n",
213  leftEdge, rightEdge, topEdge, bottomEdge);
214  _lMdlWin = rcWindow();
215  _rMdlWin = rcWindow();
216  return false;
217  }
218 
219  /* Use the previously calculated edge locations to generate the results:
220  * models, search spaces, and initial offset within search space.
221  */
222 
223  const uint32 cellWidth = rightEdge - leftEdge + 1;
224  const uint32 cellHeight = bottomEdge - topEdge + 1;
225  const uint32 outerSlop = 6;
226  const uint32 innerSlop = cellWidth/20 > 8 ? cellWidth/20 : 8;
227 
228  const uint32 heightSlop = 2;
229 
230  _leftModel = rcRect(leftEdge - outerSlop,
231  topEdge - heightSlop,
232  innerSlop + outerSlop,
233  cellHeight + heightSlop*2);
234 
235  _leftSpace = rcRect(_leftModel.x() - 4,
236  _leftModel.y() - 2,
237  _leftModel.width() + 4 + (cellWidth*10)/100,
238  _leftModel.height() + 4);
239 
240  _leftOffset = rcIPair(outerSlop, 0);
241 
242  _rightModel = rcRect(rightEdge - innerSlop,
243  topEdge - heightSlop,
244  innerSlop + outerSlop,
245  cellHeight + heightSlop*2);
246 
247  _rightSpace = rcRect(_rightModel.x() - (cellWidth*10)/100,
248  _rightModel.y() - 2,
249  _rightModel.width() + 4 + (cellWidth*10)/100,
250  _rightModel.height() + 4);
251 
252  _rightOffset = rcIPair(innerSlop, 0);
253 
254  _lMdlWin = rcWindow(_leftModel.width(), _leftModel.height());
255  rcWindow lp(images[_leftFrameIndex],
256  _leftModel.x(), _leftModel.y() - 2,
257  _leftModel.width(), _leftModel.height() + 4);
258  for (int32 yo = 0; yo < _lMdlWin.height(); yo++) {
259  for (int32 xo = 0; xo < _lMdlWin.width(); xo++) {
260  uint32 pixel =
261  (lp.getPixel(xo, yo) + lp.getPixel(xo, yo+4))*3 +
262  (lp.getPixel(xo, yo+1) + lp.getPixel(xo, yo+3))*11 +
263  lp.getPixel(xo, yo+2)*17;
264  _lMdlWin.setPixel(xo, yo, (pixel+22)/45);
265  }
266  }
267 
268  _rMdlWin = rcWindow(_rightModel.width(), _rightModel.height());
269  rcWindow rp(images[_rightFrameIndex],
270  _rightModel.x(), _rightModel.y() - 2,
271  _rightModel.width(), _rightModel.height() + 4);
272  for (int32 yo = 0; yo < _rMdlWin.height(); yo++) {
273  for (int32 xo = 0; xo < _rMdlWin.width(); xo++) {
274  uint32 pixel =
275  (rp.getPixel(xo, yo) + rp.getPixel(xo, yo+4))*3 +
276  (rp.getPixel(xo, yo+1) + rp.getPixel(xo, yo+3))*11 +
277  rp.getPixel(xo, yo+2)*17;
278  _rMdlWin.setPixel(xo, yo, (pixel+22)/45);
279  }
280  }
281 
282  _lSrchWin = rcWindow(_leftSpace.width(), _leftSpace.height());
283  _rSrchWin = rcWindow(_rightSpace.width(), _rightSpace.height());
284 
285  _rightToLeftOffset = _rightSpace.x() - _leftSpace.x();
286 
287  return true;

291  vector<double>* rCorrSpace)
292 {
293  rmAssert(_lMdlWin.isBound() && _rMdlWin.isBound());

295  /* Create smoothed versions of image pixels and generate xorrelation
296  * spaces.
297  */

299  _leftSpace.width(), _leftSpace.height() + 4);

305  lp.getPixel(xo, yo+2)*17;

312  rcWindow rp(image, _rightSpace.x(), _rightSpace.y() - 2,
313  _rightSpace.width(), _rightSpace.height() + 4);

315  for (int32 xo = 0; xo < _rSrchWin.width(); xo++) {
316  uint32 pixel =
317  (rp.getPixel(xo, yo) + rp.getPixel(xo, yo+4))*3 +
318  (rp.getPixel(xo, yo+1) + rp.getPixel(xo, yo+3))*11 +
319  rp.getPixel(xo, yo+2)*17;
320  _rSrchWin.setPixel(xo, yo, (pixel+22)/45);
321  }
322  }
323  uint32 rPeakOffsetX, rPeakOffsetY;
324  genSpace(_rSrchWin, _rMdlWin, _rResSpace, rPeakOffsetX, rPeakOffsetY);
325 
326  /* Generate interpolated position information.
327  */
328  double lPeakInterpolatedOffsetX, rPeakInterpolatedOffsetX;
329 
330  if ((lPeakOffsetX == 0) ||
331  (lPeakOffsetX == _lResSpace[lPeakOffsetY].size() - 1)) {
332  fprintf(stderr, "WARNING: Left peak at edge!!!: ");
333  lPeakInterpolatedOffsetX = lPeakOffsetX;
334  }
335  else {
336  double offset = parabolicFit(_lResSpace[lPeakOffsetY][lPeakOffsetX-1],
337  _lResSpace[lPeakOffsetY][lPeakOffsetX],
338  _lResSpace[lPeakOffsetY][lPeakOffsetX+1],
339  (double*)0);
340  lPeakInterpolatedOffsetX = lPeakOffsetX + offset;
341  }
342 
343  if ((rPeakOffsetX == 0) ||
344  (rPeakOffsetX == _rResSpace[rPeakOffsetY].size() - 1)) {
345  printf("WARNING: Right peak at edge!!!: ");
346  rPeakInterpolatedOffsetX = rPeakOffsetX;
347  }
348  else {
349  double offset = parabolicFit(_rResSpace[rPeakOffsetY][rPeakOffsetX-1],
350  _rResSpace[rPeakOffsetY][rPeakOffsetX],
351  _rResSpace[rPeakOffsetY][rPeakOffsetX+1],
352  (double*)0);
353  rPeakInterpolatedOffsetX = rPeakOffsetX + offset;
354  }
355 
356  /* If client has passed a pointer to a 1D correlation space vector,
357  * fill it with a compressed version of the the actual correlation
358  * space: for each location in x, pick the highest score in the y
359  * dimension.
360  */
361  if (lCorrSpace) {
362  if (lCorrSpace->size() != _lResSpace[0].size())
363  lCorrSpace->resize(_lResSpace[0].size());

365  double maxX = -1.0;
366  for (uint32 y = 0; y < _lResSpace.size(); y++)
367  if (_lResSpace[y][x] > maxX)
368  maxX = _lResSpace[y][x];
369  (*lCorrSpace)[x] = maxX;
370  }
371  }
372 
373  if (rCorrSpace) {
374  if (rCorrSpace->size() != _rResSpace[0].size())
375  rCorrSpace->resize(_rResSpace[0].size());
376  for (uint32 x = 0; x < _rResSpace[0].size(); x++) {
377  double maxX = -1.0;
378  for (uint32 y = 0; y < _rResSpace.size(); y++)
379  if (_rResSpace[y][x] > maxX)
380  maxX = _rResSpace[y][x];
381  (*rCorrSpace)[x] = maxX;
382  }
383  }
384 
385  return rcDPair(lPeakInterpolatedOffsetX,
386  rPeakInterpolatedOffsetX + _rightToLeftOffset);
387 }

389 void rcMuscleTracker::genSpace(rcWindow& srchWin, rcWindow& model,
390  vector<vector<double> >& space,
391  uint32& peakOffsetX, uint32& peakOffsetY)
392 {
393  rmAssert(srchWin.width() >= model.width());
394  rmAssert(srchWin.height() >= model.height());
395 
396  uint32 spaceWidth = srchWin.width() - model.width() + 1;
397  uint32 spaceHeight = srchWin.height() - model.height() + 1;
398 
399  if (space.size() != spaceHeight) {
400  space.resize(spaceHeight);
401  for (uint32 y = 0; y < spaceHeight; y++)
402  space[y].resize(spaceWidth);
403  }
404  else {
405  for (uint32 y = 0; y < spaceHeight; y++)
406  if (space[y].size() != spaceWidth)
407  space[y].resize(spaceWidth);
408  }
409 
410  rsCorrParams params;
411  rcCorr res;
412  double maxScore = -1.0;
413  const rcIPair shiftRightOne(1, 0);
414 
415  for (uint32 y = 0; y < spaceHeight; y++) {
416  rcWindow image(srchWin, 0, y, model.width(), model.height());
417 
418  for (uint32 x = 0; x < spaceWidth; ) {
419  rfCorrelate(model, image, params, res);
420  const double score = res.r();
421 
422  if (score > maxScore) {
423  maxScore = score;
424  peakOffsetX = x;
425  peakOffsetY = y;
426  }
427  space[y][x++] = score;
428  if (x != spaceWidth)
429  image.translate(shiftRightOne);
430  }
431  }
432 }
