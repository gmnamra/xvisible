58  // Add a new mapping for a group, return number of errors
59  uint32 addGroup( const std::string& idString, rcWriterGroup* inst ) {
60  uint32 errors = 0;
61 
62  mGroupMap.insert( make_pair( idString, inst ) );
63  // Immediate self-test, it can catch duplicates
64  if ( group( idString ) != inst )
65  ++errors;
66 
67  return errors;
68  };
69 
70  // Add a new mapping for a track, return number of errors
71  uint32 addTrack( const std::string& idString, rcTrackElement inst ) {
72  uint32 errors = 0;
73 
74  mTrackMap.insert( make_pair( idString, inst ) );
75  // Immediate self-test, it can catch duplicates
76  rcTrackElement e = track( idString );
77  if ( e.track() != inst.track() )
78  ++errors;
79  if ( e.rect() != inst.rect() )
80  ++errors;
81 
82  return errors;
83  };
84 
85  // Map group id string to group instance
86  rcWriterGroup* group( const std::string& idString ) const {
87  rcGroupIdMap::const_iterator pos;
88 
89  pos = mGroupMap.find( idString );
90  if ( pos != mGroupMap.end() )
91  return pos->second;
92  else
93  return 0;
94  };
95 
96  // Map track id string to writer instance
97  rcTrackElement track( const std::string& idString ) const {
98  rcTrackIdMap::const_iterator pos;
99 
100  pos = mTrackMap.find( idString );
101  if ( pos != mTrackMap.end() )
102  return pos->second;
103  else
104  return rcTrackElement();
105  };
106 
107  // Flush all scalar writers
108  void flushWriters() {
109  rcTrackIdMap::iterator pos;
110 
111  for ( pos = mTrackMap.begin(); pos != mTrackMap.end(); ++pos )
112  {
113  rcTrackElement e = pos->second;
114  if ( e.isValid() ) {
115  e.track()->flush();
116  }
117  }
118  };

122  rcTrackIdMap mTrackMap;
123 };
