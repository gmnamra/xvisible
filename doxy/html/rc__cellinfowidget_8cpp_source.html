69 // Get track value string
70 static void getTrackValue( rcTrack* track, const rcTimestamp& timeToDisplay, QString& value )
71 {
72  rcTrackType type = track->getTrackType();
73 
74  if ( track ) {
75  char valueBuf[ 256 ];
76  rcSegmentIterator* iter = 0;
77 
78  if ( type == eScalarTrack ) {
79  // get the scalar track.
80  rcScalarTrack* scalarTrack = dynamic_cast<rcScalarTrack*>( track );
81  if (scalarTrack == 0)
82  return;
83  // get the scalar value at the time to display
84  iter = scalarTrack->getDataSegmentIterator( timeToDisplay );
85  } else if ( type == ePositionTrack ) {
86 
87  // get the position track.
88  rcPositionTrack* pTrack = dynamic_cast<rcPositionTrack*>( track );
89  if (pTrack == 0) {
90  return;
91  }
92  // get the value at the time to display
93  iter = pTrack->getDataSegmentIterator( timeToDisplay );
94  }
95 
96  if ( iter ) {
97  if ( iter->contains( timeToDisplay ) ) {
98  // timeToDisplay is within the segment
99  if ( type == eScalarTrack ) {
100  rcScalarSegmentIterator* si = dynamic_cast<rcScalarSegmentIterator*>(iter);
101  double value = si->getValue();
102  // Special case for cell state track
103  if ( rcWriterManager::tagType( track->getTag() ) == eWriterBodyState ) {
104  rfVisualFunctionStateName( static_cast<rcVisualFunction::State>(value),
105  valueBuf,
106  rmDim(valueBuf) );
107  } else {
108  // Generic scalar track
109  snprintf( valueBuf , rmDim(valueBuf),
110  track->getDisplayFormatString(), value );
111  }
112  } else if ( type == ePositionTrack ) {
113  rcPositionSegmentIterator* pi = dynamic_cast<rcPositionSegmentIterator*>(iter);
114  rcFPair value = pi->getValue();
115  snprintf( valueBuf, rmDim(valueBuf),
116  "%.4f, %.4f", value.x(), value.y() );
117  }
118  } else {
119  // timeToDisplay is either before track start or after track end
120  snprintf( valueBuf , rmDim(valueBuf), "Undefined" );
121  }
122  value.sprintf( "%s\n", valueBuf );
123  delete iter;
124  }
125  }
