34 float rfSurfaceHessian (rcWindow& src)
35 {
36  float gxx(0.0), gyy(0.0), gxy(0.0);
37  int32 rup (src.rowUpdate());
38  float nn (src.pixelCount() * 2);
39 
40  for (int32 j = 1; j < src.height() - 1; j++)
41  {
42  uint8 * g = src.rowPointer (j);
43 
44  for (int32 i = 1; i < src.width() - 1; i++, g++)
45  {
46  float gp (g[0] );
47  float gx (gp - g[1]);
48  float gy (gp - g[rup]);
49  gxx += (rmSquare (gx));
50  gyy += (rmSquare (gy));
51  gxy += (gx * gy);
52  }
53  }
54 
55  // Return the min eigenvalue
56  return (gxx + gyy - sqrt((gxx - gyy)*(gxx - gyy) + 4*gxy*gxy))/ nn;
57 }
