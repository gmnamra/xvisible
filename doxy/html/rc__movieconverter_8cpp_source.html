693  rcFrameGrabberStatus status = inputGrabber.getNextFrame( framePtr, true );
694 
695  if ( status == eFrameStatusOK ) {
696  if ( i < first )
697  continue;
698  if ( frames == frameCount )
699  break;
700  if ( (i - lastIndex) < mOptions.samplePeriod() )
701  continue;
702 
703  rcTimestamp curTimeStamp = framePtr->timestamp();
704  rcTimestamp frameInt = curTimeStamp - prevTimeStamp;
705 
706  if ( firstFrame ) {
707  firstFrame = false;
708  } else {
709  // Zero time between frames is not allowed
710  if ( frameInt <= 0.0 )
711  if ( verbose() )
712  fprintf( stderr, "Error: invalid frame interval %f for frame %i\n", frameInt.secs(), i );
713  rmAssert( frameInt > 0.0 );
714  }
715  framePtr = clippedFrame( framePtr, mOptions );
716  lastIndex = i;
717  prevTimeStamp = curTimeStamp;
718  rcWindow image;
719  image = rcWindow( framePtr );
720  ++frames;
721 
722  if ( image.depth() == rcPixel8 ) {
723  if ( mOptions.reversePixels() ) {
724  rfReversePixels8( image );
725  pixelsReversed = true;
726  }
727  }
728  // Write a frame to disk
729  rcGenMovieFileError err = generator.addFrame( image );
730  if ( err != eGenMovieFileErrorOK )
731  break;
732 
733  if ( mProgress )
734  mProgress->progress( double(frames)/frameCount * 100 );
735  } else if ( status == eFrameStatusEOF ) {
736  // no more frames available
737  break;
738  }
739  else {
740  // getNextFrame() failed, bail out
741  error = inputGrabber.getLastError();
742  break;
743  }
744  }
745 
746  } else {
747  // start() failed
748  error = inputGrabber.getLastError();
749  err = eMovieConverterErrorRead;
750  }
751 
752  if ( ! inputGrabber.stop() ) {
753  // stop() failed
754  error = inputGrabber.getLastError();
755  err = eMovieConverterErrorRead;
756  }
757  } else {
758  // isValid() failed
759  error = inputGrabber.getLastError();
760  err = eMovieConverterErrorOpen;
761  }
762 
763  if ( error != eFrameErrorOK ) {
764  if ( verbose() )
765  cerr << rcFrameGrabber::getErrorString( error ) << endl;
766  return err;
767  } else
768  err = eMovieConverterErrorOK;
769 
770  // Add conversion header
771  rcMovieFileConvExt cnvHdr( mOptions.firstFrameIndex() + mOptions.frameOffset(),
772  frames,
773  mOptions.samplePeriod(),
774  mOptions.clipRect(),
775  channelType(generator.conversion()),
776  mOptions.frameInterval(),
777  pixelsReversed,
778  mOptions.creator().c_str() );
779  rcGenMovieFileError gerr = generator.addHeader( cnvHdr );
780  if ( gerr != eGenMovieFileErrorOK ) {
781  if ( verbose() )
782  cerr << "Output movie addHeader error: " << gerr << " " << generator.fileName() << endl;
783  return eMovieConverterErrorHeaderWrite;
784  }
785 
786  // Flush and close output file
787  gerr = generator.flush();
788  if ( gerr != eGenMovieFileErrorOK ) {
789  if ( verbose() )
790  cerr << "Output movie flush error: " << gerr << " " << generator.fileName() << endl;
791  return eMovieConverterErrorFlush;
792  }
793 
794  return err;
795 }
