246 static bool genSignatureFromMov (cli_parser& pars, string& rfymovf, deque<double>& _entropies)
247 {
248  double physMem (rfSystemRam () / rmBytesInGig); // In GigaBytes
249  const double maxMem = rmBytesInGig / 4.; // Minimum 256 MB
250  physMem = rmMin (physMem / 4 , maxMem);
251  uint32 maxMemToCache = (uint32) (physMem * rmBytesInGig);
252  rcVideoCache* _videoCacheP = rcVideoCache::rcVideoCacheCtor(rfymovf, 0, true, false,true,
253  maxMemToCache, 0 );
254 
255  if (! _videoCacheP->frameCount() ) return false;
256 
257  // VideoCache has grabbed all frames at sampling specified
258  // Now at the offset and count specified, run similarator
259  // get other vars
260  int32 offset = startframe (pars);
261  offset = (offset < 0) ? 0 : offset;
262  int32 _count = count (pars);
263  _count = (_count < 0) ? 0 : _count;
264 
265  int32 framesToUse = (int32) (_videoCacheP->frameCount()) - offset;
266  if (_count > 0 && framesToUse > _count) framesToUse = _count;
267  int32 frameIndex = offset;
268  vector<rcWindow> movie;
269  for (int32 usedFrames = 0; usedFrames++ < framesToUse;frameIndex += 1)
270  movie.push_back(rcWindow(*_videoCacheP, (uint32) (frameIndex)));
271 
272  int32 cacheSize = (uint32)(_videoCacheP->cacheSize());
273  // Do not accesss images in the vector. They are cached.
274  rcSimilaratorRef simi ( new rcSimilarator (rcSimilarator::eExhaustive,
275  _videoCacheP->frameDepth(),
276  framesToUse,
277  cacheSize,
278  rcSimilarator::eLowSpT,
279  true));
280  simi->fill(movie);
281  bool ok = simi->entropies (_entropies, rcSimilarator::eVisualEntropy);
282  rcVideoCache::rcVideoCacheDtor (_videoCacheP);
283  return ok;
284 }

513 {

515  string rfymovf = create_filespec (val, filename_part (contentf) + string (".rfymov"));
516  if (m2vf.empty() || rfymovf.empty()) return string ();
517 
518  // First we need to extract the video track
519  string m2vprog ("/extract_m2v -s 0xe0 ");
520  string cmd = execpath + m2vprog + contentf + string (" > ") + m2vf;
521  std::system( cmd.c_str() );
522  std::string whichMovie = std::string (m2vf);
523  if (file_size (m2vf) == 0) whichMovie = std::string (contentf);
524 
525  // Convert mpg to rfymov and store it loclly
526  // ini : channelConveson, frameInterval, extractm2v
527  rcMPEG2Grabber mpegg (whichMovie, NULL);
