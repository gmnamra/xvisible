30 void offsetGraphics( rcVisualGraphicsCollection& graphics, const rc2Fvector& offset )
31 {
32  if ( offset.x() != 0.0 || offset.y() != 0.0 ) {
33  rcVisualSegmentCollection& lines = graphics.segments();
34 
35  for ( uint32 i = 0; i < lines.size(); ++i ) {
36  rcVisualSegment line = lines[i];
37  const rcVisualSegment::rcVisualSegmentType type = line.type();
38  switch( type ) {
39  case rcVisualSegment::eEllipse:
40  case rcVisualSegment::eCross:
41  // Do not offset second point

43  break;
44 
45  case rcVisualSegment::ePoints:
46  case rcVisualSegment::eLine:
47  case rcVisualSegment::eLineStrip:
48  case rcVisualSegment::eLineLoop:
49  case rcVisualSegment::eArrow:
50  case rcVisualSegment::eRect:
51  // Offset both points

56  case rcVisualSegment::eEmpty:
57  case rcVisualSegment::eStyle:
58  case rcVisualSegment::eLast:
59  // Do nothing

111 // Call this to update status bar with completion percentage
112 bool rcEngineProgress::progress( double percentComplete )
113 {
114  bool ret = true;
115 
116  if ( mEngine ) {
117  if ( mObserver ) {
118  rcTimestamp now = rcTimestamp::now();
119  // If interval is zero, update with every call
120  // Otherwise update after mUpdateInterval seconds have passd since last update
121  if ( mUpdateInterval == cZeroTime ||
122  ( (now - mLastUpdate) > mUpdateInterval ) ) {
123  mLastUpdate = now;
124  snprintf( mMessageBuf, rmDim( mMessageBuf ), mFormatMessage.c_str(), percentComplete );
125  // Update status bar
126  mObserver->notifyStatus( mMessageBuf );
127  // Update elapsed time to refresh display
128  mObserver->notifyTime( mEngine->getElapsedTime() );
129  }
130  }
131  if ( mEngine->getState() == eEngineShuttingDown ||
132  mEngine->getState() == eEngineStopped )
133  ret = true; // Abort analysis
134  else
135  ret = false;
136  }
137  return ret;
138 }

292 bool rcEngineBaseImpl::checkSettings()

300 
301  // Verify that all settings have a unique tag

316  // Verify that all categories have a unique tag

318  rcSettingCategory c2 = *j;

330 // our internal state changing method
331 void rcEngineBaseImpl::setInternalState( rcEngineState newState )
332 {
333  // ignore setting to current state
334  if (newState == _state)
335  return;
336 
337  // validate state transition
338  switch (_state)
339  {
340  case eEngineUninitialized: // engine has not been initialized
341  if (newState != eEngineStopped)
342  throw general_exception( "illegal state transition from eUninitialized" );
343  break;
344 
345  case eEngineStopped: // engine is stopped and not generating data
346  if (newState != eEngineRunning && newState != eEngineProcessing )
347  throw general_exception( "illegal state transition from eEngineStopped" );
348  break;
349 
350  case eEngineRunning: // engine is running and generating data
351  if (newState != eEngineStopped)
352  throw general_exception( "illegal state transition from eEngineRunning" );
353  break;
354 
355  case eEnginePlayback: // engine is running and generating data
356  if (newState != eEngineStopped)
357  throw general_exception( "illegal state transition from eEnginePlayback" );
358  break;
359 
360  case eEngineProcessing: // engine is processing a focus area
361  if (newState != eEngineStopped)
362  throw general_exception( "illegal state transition from eEngineProcessing" );
363  break;
364 
365  case eEngineShuttingDown: // engine is in the process of shutting down
366  throw general_exception( "illegal state transition from eEngineShuttingDown" );
367  break;
368  }
369 
370  PRINT_STATE_CHANGE("setInternalState", _state, newState);
371  _state = newState;
372  if ( _observer )
373  _observer->notifyState( _state );
374 }
