155 // Returns the sqrt of the mean squared error measured perpendicular from the
156 // line through mean_point() in the direction dir.
157 //
158 // Derivation:
159 // Lemma: Let v and x_i (i=1..N) be a k-dimensional vectors (1xk matrices).
160 // Let % be dot product and ' be transpose. Note that:
161 // Sum[i=1..N] (v % x_i)^2
162 // = v * [x_1' x_2' ... x_N'] * [x_1' x_2' .. x_N']' * v'
163 // If x_i have average 0 we have:
164 // = v * (N * COVARIANCE_MATRIX(X)) * v'
165 // Expanded for the case that k = 2, where we treat the dimensions
166 // as x_i and y_i, this is:
167 // = v * (N * [VAR(X), COV(X,Y); COV(X,Y) VAR(Y)]) * v'
168 // Now, we are trying to calculate the mean squared error, where v is
169 // perpendicular to our line of interest:
170 // Mean squared error
171 // = E [ (v % (x_i - x_avg))) ^2 ]
172 // = Sum (v % (x_i - x_avg))^2 / N
173 // = v * N * [VAR(X) COV(X,Y); COV(X,Y) VAR(Y)] / N * v'
174 // = v * [VAR(X) COV(X,Y); COV(X,Y) VAR(Y)] * v'
175 // = code below
176 
177 double LLSQ::rms_orth(const rc2Fvector &dir) const {
178  rc2Fvector v = dir.perpendicular ();
179  v = v.unit ();
180  return sqrt(v.x() * v.x() * x_variance() +
181  2 * v.x() * v.y() * covariance() +
182  v.y() * v.y() * y_variance());
183 }

185 // Returns the direction of the fitted line as a unit vector, using the
186 // least mean squared perpendicular distance. The line runs through the
187 // mean_point, i.e. a point p on the line is given by:
188 // p = mean_point() + lambda * vector_fit() for some real number lambda.
189 // Note that the result (0<=x<=1, -1<=y<=-1) is directionally ambiguous
190 // and may be negated without changing its meaning.
191 // Fitting a line m + 𝜆v to a set of N points Pi = (xi, yi), where
192 // m is the mean point (𝝁, 𝝂) and
193 // v is the direction vector (cos𝜃, sin𝜃)
194 // The perpendicular distance of each Pi from the line is:
195 // (Pi - m) x v, where x is the scalar cross product.
196 // Total squared error is thus:
197 // E = ∑((xi - 𝝁)sin𝜃 - (yi - 𝝂)cos𝜃)²
198 // = ∑(xi - 𝝁)²sin²𝜃 - 2∑(xi - 𝝁)(yi - 𝝂)sin𝜃 cos𝜃 + ∑(yi - 𝝂)²cos²𝜃
199 // = NVar(xi)sin²𝜃 - 2NCovar(xi, yi)sin𝜃 cos𝜃 + NVar(yi)cos²𝜃 (Eq 1)
200 // where Var(xi) is the variance of xi,
201 // and Covar(xi, yi) is the covariance of xi, yi.
202 // Taking the derivative wrt 𝜃 and setting to 0 to obtain the min/max:
203 // 0 = 2NVar(xi)sin𝜃 cos𝜃 -2NCovar(xi, yi)(cos²𝜃 - sin²𝜃) -2NVar(yi)sin𝜃 cos𝜃
204 // => Covar(xi, yi)(cos²𝜃 - sin²𝜃) = (Var(xi) - Var(yi))sin𝜃 cos𝜃
205 // Using double angles:
206 // 2Covar(xi, yi)cos2𝜃 = (Var(xi) - Var(yi))sin2𝜃 (Eq 2)
207 // So 𝜃 = 0.5 atan2(2Covar(xi, yi), Var(xi) - Var(yi)) (Eq 3)
208 
209 // Because it involves 2𝜃 , Eq 2 has 2 solutions 90 degrees apart, but which
210 // is the min and which is the max? From Eq1:
211 // E/N = Var(xi)sin²𝜃 - 2Covar(xi, yi)sin𝜃 cos𝜃 + Var(yi)cos²𝜃
212 // and 90 degrees away, using sin/cos equivalences:
213 // E'/N = Var(xi)cos²𝜃 + 2Covar(xi, yi)sin𝜃 cos𝜃 + Var(yi)sin²𝜃
214 // The second error is smaller (making it the minimum) iff
215 // E'/N < E/N ie:
216 // (Var(xi) - Var(yi))(cos²𝜃 - sin²𝜃) < -4Covar(xi, yi)sin𝜃 cos𝜃
217 // Using double angles:
218 // (Var(xi) - Var(yi))cos2𝜃 < -2Covar(xi, yi)sin2𝜃 (InEq 1)
219 // But atan2(2Covar(xi, yi), Var(xi) - Var(yi)) picks 2𝜃 such that:
220 // sgn(cos2𝜃) = sgn(Var(xi) - Var(yi)) and sgn(sin2𝜃) = sgn(Covar(xi, yi))
221 // so InEq1 can *never* be true, making the atan2 result *always* the min!
222 // In the degenerate case, where Covar(xi, yi) = 0 AND Var(xi) = Var(yi),
223 // the 2 solutions have equal error and the inequality is still false.
224 // Therefore the solution really is as trivial as Eq 3.
225 
226 // This is equivalent to returning the Principal Component in PCA, or the
227 // eigenvector corresponding to the largest eigenvalue in the covariance
228 // matrix. However, atan2 is much simpler! The one reference I found that
229 // uses this formula is http://web.mit.edu/18.06/www/Essays/tlsfit.pdf but
230 // that is still a much more complex derivation. It seems Pearson had already
231 // found this simple solution in 1901.
232 // http://books.google.com/books?id=WXwvAQAAIAAJ&pg=PA559
233 
234  rcRadian LLSQ::vector_fit_angle () const
235 {
236  double x_var = x_variance();
237  double y_var = y_variance();
238  double covar = covariance();
239  double theta = 0.5 * atan2(2.0 * covar, x_var - y_var);
240  return rcRadian(theta);
241  }
