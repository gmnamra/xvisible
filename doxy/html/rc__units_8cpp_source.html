115 double rfRanOne(int32 &idum)
116 {
117  const int32 IA=16807,IM=2147483647,IQ=127773,IR=2836,NTAB=32;
118  const int32 NDIV=(1+(IM-1)/NTAB);
119  const double EPS=3.0e-16,AM=1.0/IM,RNMX=(1.0-EPS);
120  static int32 iy=0;
121  static std::vector<int32> iv(NTAB);
122  int32 j,k;
123  double temp;
124 
125  if (idum <= 0 || !iy) {
126  if (-idum < 1) idum=1;
127  else idum = -idum;
128  for (j=NTAB+7;j>=0;j--) {
129  k=idum/IQ;
130  idum=IA*(idum-k*IQ)-IR*k;
131  if (idum < 0) idum += IM;
132  if (j < NTAB) iv[j] = idum;
133  }
134  iy=iv[0];
135  }
136  k=idum/IQ;
137  idum=IA*(idum-k*IQ)-IR*k;
138  if (idum < 0) idum += IM;
139  j=iy/NDIV;
140  iy=iv[j];
141  iv[j] = idum;
142  if ((temp=AM*iy) > RNMX) return RNMX;
143  else return temp;
144 }

169 int32 rfGCDeuclid(int32 a, int32 b)
170  {
171  /* Returns the greatest common divisor of u and v by Euclid's method.
172  * I have experimented also with Stein's method, which involves only
173  * subtraction and left/right shifting; Euclid is faster, both for
174  * integers of size 0 - 1024 and also for random integers of a size
175  * which fits in a long integer. Stein's algorithm might be better
176  * when the integers are HUGE, but for our purposes, Euclid is fine.
177  *
178  * Walter H. F. Smith, 25 Feb 1992, after D. E. Knuth, vol. II */
179 
180  int32 u,v,r;
181 
182  u = rmMax (rmABS(a), rmABS(b));
183  v = rmMin (rmABS(a), rmABS(b));
184 
185  while (v > 0) {
186  r = u%v; /* Knuth notes that u < 2v 40% of the time; */
187  u = v; /* thus we could have tried a subtraction */
188  v = r; /* followed by an if test to do r = u%v */
189  }
190  return(u);
191  }
