262 // Get human-readable sliding window attribute string
263 std::string rcWriterManager::optionString( rcWriterSemantics type,
264  rcWriterWindowOrigin origin,
265  uint32 windowSize )
266 {
267  rmUnused( type );
268 
269  std::string options;
270  char originStr = '#';
271 
272  switch ( origin ) {
273  case eWindowOriginLeft:
274  originStr = 'L';
275  break;
276  case eWindowOriginCenter:
277  originStr = 'C';
278  break;
279  case eWindowOriginRight:
280  originStr = 'R';
281  break;
282  }
283  char buf[256];
284 
285  // Add window size and origin indicator to string
286  snprintf( buf, rmDim(buf), " [%i%c]",
287  windowSize, originStr );
288 
289  options += buf;
290 
291  return options;
292 }
