80 bool rcWindow::tiff (std::string& outfile, bool doCompress) const // int argc, char* argv[])

84  off_t hdr_size = 0; /* size of the header to skip */
85  TIFFDataType dtype = TIFF_BYTE;
86  int16 _depth = 1; /* bytes per pixel in input image */
87  InterleavingType interleaving = PIXEL; /* interleaving type flag */
88  uint32 rowsperstrip = (uint32) 1;
89  uint16 photometric = PHOTOMETRIC_MINISBLACK;

98  switch (depth())
99  {
100  case rcPixel8:
101  dtype = TIFF_BYTE;
102  _depth = 1;
103  break;
104  case rcPixel16:
105  dtype = TIFF_SHORT;
106  _depth = 2;
107  break;
108  case rcPixel32:
109  dtype = isD32Float() ? TIFF_FLOAT : TIFF_LONG;
110  _depth = 4;
111  break;
112  case rcPixelDouble:
113  dtype = TIFF_DOUBLE;
114  _depth = 8;
115  break;
116  default:
117  case rcPixelUnknown:
118  return false;
119  }
120 
121  // Photometric needs to be handle
122  out = TIFFOpen(outfile.c_str(), "wb");
123  if (out == NULL)
124  {
125  cerr << outfile << "Cannot open file for output. " << endl;
126  return 0;
127  }
128  TIFFSetField(out, TIFFTAG_IMAGEWIDTH, _width);
129  TIFFSetField(out, TIFFTAG_IMAGELENGTH, length);
130  TIFFSetField(out, TIFFTAG_ORIENTATION, ORIENTATION_TOPLEFT);
131  TIFFSetField(out, TIFFTAG_SAMPLESPERPIXEL, nbands);
132  TIFFSetField(out, TIFFTAG_BITSPERSAMPLE, _depth * 8);
133  TIFFSetField(out, TIFFTAG_FILLORDER, fillorder);
134  TIFFSetField(out, TIFFTAG_PLANARCONFIG, config);
135  TIFFSetField(out, TIFFTAG_PHOTOMETRIC, photometric);
136  switch (dtype) {
137  case TIFF_BYTE:
138  case TIFF_SHORT:
139  case TIFF_LONG:
140  TIFFSetField(out, TIFFTAG_SAMPLEFORMAT, SAMPLEFORMAT_UINT);
141  break;
142  case TIFF_SBYTE:
143  case TIFF_SSHORT:
144  case TIFF_SLONG:
145  TIFFSetField(out, TIFFTAG_SAMPLEFORMAT, SAMPLEFORMAT_INT);
146  break;
147  case TIFF_FLOAT:
148  case TIFF_DOUBLE:
149  TIFFSetField(out, TIFFTAG_SAMPLEFORMAT, SAMPLEFORMAT_IEEEFP);
150  break;
151  default:
152  TIFFSetField(out, TIFFTAG_SAMPLEFORMAT, SAMPLEFORMAT_VOID);
153  break;
154  }
155 
156  if (doCompress) compression = (uint16) -1;
157 
158  if (compression == (uint16) -1)
159  compression = COMPRESSION_PACKBITS;
160  TIFFSetField(out, TIFFTAG_COMPRESSION, compression);
161  switch (compression) {
162  case COMPRESSION_JPEG:
163  if (photometric == PHOTOMETRIC_RGB
164  && jpegcolormode == JPEGCOLORMODE_RGB)
165  photometric = PHOTOMETRIC_YCBCR;
166  TIFFSetField(out, TIFFTAG_JPEGQUALITY, quality);
167  TIFFSetField(out, TIFFTAG_JPEGCOLORMODE, jpegcolormode);
168  break;
169  case COMPRESSION_LZW:
170  case COMPRESSION_DEFLATE:
171  if (predictor != 0)
172  TIFFSetField(out, TIFFTAG_PREDICTOR, predictor);
173  break;
174  }
175  switch(interleaving) {
176  case BAND: /* band interleaved data */
177  linebytes = _width * _depth;
178  buf = (unsigned char *)_TIFFmalloc(linebytes);
179  break;
180  case PIXEL: /* pixel interleaved data */
181  default:
182  linebytes = _width * nbands * _depth;
183  break;
184  }
185 
186  TIFFSetField(out, TIFFTAG_ROWSPERSTRIP, TIFFDefaultStripSize(out, rowsperstrip));
187  lseek(fd, hdr_size, SEEK_SET); /* Skip the file header */
188  for (row = 0; row < length; row++)
189  {
190  if (TIFFWriteScanline(out, (unsigned char *) rowPointer (row), row, 0) < 0)
191  {
192  cerr << outfile << ": scanline" << row << ": Write error." << endl;
193  break;
194  }
195  }
196 
197  if (buf)
198  _TIFFfree(buf);
199  TIFFClose(out);
200 
201  return true;
202 }
