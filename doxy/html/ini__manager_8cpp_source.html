290  bool read_file(const std::string& filename)
291  {
292  m_filename = filename;
293  m_changed = false;
294  m_writable = file_writable(m_filename);
295  // create a dummy top section to hold file comments
296  std::list<smart_ptr<ini_section> >::iterator section = m_sections.insert(m_sections.end(), ini_section(""));
297  if (!file_exists(m_filename))
298  return true;
299  iftext input(m_filename);
300  std::string line;
301  unsigned line_number = 0;
302  while(input.getline(line))
303  {
304  line_number++;
305  unsigned i = 0;
306  while(i < line.size() && isspace(line[i]))
307  i++;
308  if (i < line.size() && line[i] == '[')
309  {
310  // found a section title
311  // skip the [
312  i++;
313  // get the text up to the end ] or the end of the line
314  std::string title;
315  while (i < line.size() && line[i] != ']')
316  title += line[i++];
317  // create a new section and make it the current section
318  section = m_sections.insert(m_sections.end(), ini_section(title));
319  }
320  else if (i < line.size() && line[i] == ';')
321  {
322  // found a comment
323  // skip the ; because that is not part of the comment
324  i++;
325  // add the rest of the line as a comment to the current section
326  (*section)->add_comment(line_number, line.substr(i, line.size()-1));
327  }
328  else if (i == line.size())
329  {
330  // found a blank line
331  (*section)->add_blank(line_number);
332  }
333  else
334  {
335  // found a *possible* variable - now scan forwards for the = operator
336  std::string name;
337  while (i < line.size() && line[i] != '=')
338  name += line[i++];
339  // skip the = sign
340  // TODO - detect a missing = sign here and convert to an error
341  if (i < line.size())
342  i++;
343  // trim trailing whitespace off the name
344  name = trim(name);
345  // now get the value, minus any leading whitespace
346  std::string value;
347  while(i < line.size() && isspace(line[i]))
348  i++;
349  while (i < line.size())
350  value += line[i++];
351  // trim trailing whitespace off the value
352  value = trim(value);
353  // and finally add the name/value pair to the data structure
354  (*section)->add_variable(line_number, name, value);
355  }
356  }
357  return true;
358  }

800 std::vector<std::string> ini_manager::variable_values(const std::string& section, const std::string variable, unsigned depth) const
801 {
802  // get the unprocessed value and then do the processing into processed separate values
803  std::string value = variable_value(section, variable, depth);
804  std::vector<std::string> result;
805  if (!value.empty())
806  {
807  result.push_back(std::string());
808  unsigned i = 0;
809  // loop which is repeated for each element in the comma-separated list
810  while(i < value.size())
811  {
812  // skip leading whitespace
813  while (i < value.size() && isspace(value[i])) i++;
814  // get the value - this means all text up to the next comma or end of line
815  // also allow escaped characters
816  while (i < value.size())
817  {
818  if (value[i] == ',') break;
819  if (value[i] == '\\')
820  {
821  // found an escaped character - de-escape it by only getting the next character
822  // beware of an escape character at the end of the line which just gets ignored
823  i++;
824  if (i < value.size()) result.back() += value[i++];
825  }
826  else if (value[i] == '"')
827  {
828  // get a quoted substring
829  // first skip the opening quote
830  i++;
831  // keep getting characters until a close-quote, but allow the quote character to be escaped by itself
832  while (i < value.size())
833  {
834  if (value[i] == '"')
835  {
836  // found a quote skip it
837  i++;
838  // now establish whether its an escaped quote
839  // if it is, keep it, but de-escape it by only getting the next character
840  // it it isn't, break out and continue processing the value as a non-quoted string
841  if (i < value.size() && value[i] != '"') break;
842  if (i < value.size()) result.back() += value[i++];
843  }
844  else
845  {
846  // non-quote and non-escape so just get the character
847  result.back() += value[i++];
848  }
849  }
850  }
851  else
852  {
853  // non-escape so just get the character
854  result.back() += value[i++];
855  }
856  }
857  // trim trailing whitespace off the value
858  while (!result.back().empty() && isspace(result.back()[result.back().size()-1])) result.back().erase(result.back().size()-1,1);
859  // now chk for whether there is a comma or end of line
860  if (i <= value.size() && value[i] == ',')
861  {
862  // skip the comma and add a new string to the result ready for the next iteration
863  i++;
864  result.push_back(std::string());
865  }
866  else
867  {
868  // end of line found so break out
869  break;
870  }
871  }
872  }
873  return result;
874 }

882 bool ini_manager::add_variable(const std::string& section, const std::string& variable, const std::vector<std::string>& values, unsigned depth)
883 {
884  // convert the values vector into a comma-separated string with each value escaped so that special characters do not confuse the reader
885  // the characters escaped are: '\', ',', '"'
886  std::string value;
887  for (unsigned v = 0; v < values.size(); v++)
888  {
889  const std::string& element = values[v];
890  // add the comma between values === add a comma before all but the first value
891  if (v > 0) value += ',';
892  for (unsigned i = 0; i < element.size(); i++)
893  {
894  // add a character at a time, escaping special characters
895  if (element[i] == '"' || element[i] == ',' || element[i] == '\\')
896  {
897  // escape the character
898  value += '\\';
899  }
900  value += element[i];
901  }
902  }
903  return add_variable(section, variable, value, depth);
904 }
