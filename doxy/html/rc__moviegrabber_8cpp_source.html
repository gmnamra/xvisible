95 bool rcMovieGrabber::start()
96 {
97  lock();
98  ClearMoviesStickyError();
99 
100  // open a Movie file using the FSSpec and create a Movie from that file.
101  OSErr err = OpenMovieFile( &mFileSpec, &mRefnum, fsRdPerm );
102  mFrameTimes.clear ();
103 
104  if ( err == noErr )
105  {
106  NewMovieFromFile( &mMovie, mRefnum, NULL, NULL, newMovieActive, NULL );
107 
108  mMedia = GetTrackMedia( GetMovieIndTrack ( mMovie, 1 ) );
109  // Allocate a handle for sample description
110  SampleDescriptionHandle anImageDesc = (SampleDescriptionHandle) NewHandle( sizeof(Handle) );
111 
112  // Get the image description for the movie
113  err = getVideoMediaImageDesc( mMedia, anImageDesc );
114 
115  if ( err == noErr )
116  {
117  getPixelInfoFromImageDesc ( (ImageDescriptionHandle) anImageDesc);
118 
119  mTimeScale = GetMovieTimeScale(mMovie);
120  Rect tmp = {0,0,0,0};
121  GetMovieBox(mMovie, &tmp);
122  m_width = tmp.right - tmp.left;
123  m_height = tmp.bottom - tmp.top;
124  m_qt_movie_rect.right = m_width; m_qt_movie_rect.bottom = m_height;
125  m_qt_movie_rect.left = tmp.left; m_qt_movie_rect.top = tmp.top;
126  // count the number of video 'frames' in the movie by stepping through all of the
127  // video 'interesting times', or in other words, the places where the movie displays
128  // a new video sample. The time between these interesting times is not necessarily constant.
129  {
130  OSType whichMediaType = VisualMediaCharacteristic;
131  TimeValue theTime = -1;
132 
133  // find out movie start time
134  GetMovieNextInterestingTime (mMovie, short (nextTimeMediaSample + nextTimeEdgeOK),
135  1, & whichMediaType, TimeValue (0), 0, & theTime, NULL);
136  if (theTime == -1)
137  {
138  fprintf (stderr, "Couldn't inquire first frame time\n");
139  return 0;
140  }
141  mCurrentTime = theTime;
142  // count all 'interesting times' of the movie
143  while (theTime >= 0)
144  {
145  mFrameTimes.push_back (theTime);
146  GetMovieNextInterestingTime (mMovie, short (nextTimeMediaSample),
147  1, & whichMediaType, theTime, 0, & theTime, NULL);
148  }
149  mFrameCount = mFrameTimes.size ();
150  }
151 
152  }
153  else {
154  setLastError( eFrameErrorUnsupportedFormat );
155  }
156  DisposeHandle( (Handle) anImageDesc );
157  }
158 
159 
160  unlock();
161 
162  if ( getLastError() == eFrameErrorOK )
163  return true;
164  else
165  return false;
166 }

250 // From an a quickTime imagedescription we can figure out a few things about the frame in the image:
251 // depth: Contains the pixel depth specified for the compressed image:
252 // Values of 1, 2, 4, 8, 16, 24, and 32 indicate the depth of color images.
253 // Values of 34, 36, and 40 indicate 2-bit, 4-bit, and 8-bit grayscale, respectively, for grayscale images.
254 // For us the only acceptable images are 8, 16, 24, and 32 in color and 40 which is 8 bit grey.
255 // We detect a gray image through size and equality of color mappings
256 // TBD: what is the best way of handling 32 bit images? The issue here is how to treat alpha and also byte ordering.
257 
258 void rcMovieGrabber::getPixelInfoFromImageDesc (ImageDescriptionHandle anImageDesc)
259 {
260 
261  int16 aPixelDepth = (*anImageDesc)->depth;
262  // Get the color table
263  CTabHandle m_ctb = NULL;
264  GetImageDescriptionCTable (anImageDesc,&m_ctb);
265  int16 csize = (*m_ctb)->ctSize;
266 
267  m_isGray = false;
268  m_isWhiteReversed = false;
269 
270  // Check grayness
271  if ( csize > 0 ) {
272  m_isGray = true;
273  // Test each color table entry
274  for ( int16 i = 0; i < csize + 1; ++i ) {
275  const int r = (*m_ctb)->ctTable[i].rgb.red;
276  const int g = (*m_ctb)->ctTable[i].rgb.green;
277  const int b = (*m_ctb)->ctTable[i].rgb.blue;
278  if ( r != g || g != b ) {
279  m_isGray = false;
280  break;
281  }
282  }
283  if ( m_isGray ) {
284  // Test whether index 0 contains white
285  m_isWhiteReversed = true;
286 
287  for ( int16 i = 1; i < csize + 1; ++i ) {
288  const int prev = (*m_ctb)->ctTable[i-1].rgb.red;
289  const int cur = (*m_ctb)->ctTable[i].rgb.red;
290  if ( prev < cur ) {
291  m_isWhiteReversed = false;
292  break;
293  }
294  }
295  }
296  }
297 
298 
299  switch( aPixelDepth ) {
300  case 8:
301  m_pd = rcPixel8;
302  m_pf = k8IndexedPixelFormat;
303  break;
304  case 16:
305  m_pd = rcPixel32;
306  m_pf = k32ARGBPixelFormat;
307  break;
308  case 24:
309  m_pd = rcPixel32;
310  m_pf = k32ARGBPixelFormat;
311  break;
312  case 32:
313  m_pd = rcPixel32;
314  m_pf = k32ARGBPixelFormat;
315  break;
316  case 40:
317  m_pd = rcPixel8;
318  m_pf = k8IndexedGrayPixelFormat;
319  break;
320  default:
321  m_pd = rcPixelUnknown;
322  m_pf = k32ARGBPixelFormat;
323  break;
324  }
325 
326 #ifdef DEBUG_LOG
327  fprintf (stderr, "Pixel depth in movie file is %d ==> %d bytes in depth - %s - %s - %s - color table size %i\n",
328  aPixelDepth, m_pd, m_isGray ? "is Gray" : "is not Gray",
329  m_isWhiteReversed ? "0 is white" : "0 is black",
330  m_pf == k8IndexedPixelFormat ? "k8IndexedPixelFormat" :
331  (m_pf == k8IndexedGrayPixelFormat) ? "k8IndexedGrayPixelFormat" :
332  (m_pf == k32ARGBPixelFormat) ? "k32ARGBPixelFormat" : "Other",
333  csize );
334 #endif
335 }

338 OSErr rcMovieGrabber::getNextVideoSample(rcWindow& image, Media media, TimeValue fromTimePoint,
339  bool reduceGrayTo8bit)
340 {
341  OSErr anErr = noErr;
342  GWorldPtr gWorld = NULL;
343  rcWindow tmp, tmp2;
344  PixMapHandle pixMap = NULL;
345 
346  assert(mMovie != NULL);
347 
348  // Create aGWorld for writing the movie in to
349  // We will allocate GWorld memory ourselves as a framebuf ===> Dispose will not free this memory
350  try
351  {
352  rcPixel depth = m_pd;
353 
354  rcSharedFrameBufPtr buf(new rcFrame (m_width, m_height, depth));
355  buf->setIsGray(m_isGray);
356  rcWindow rw (buf);
357  tmp = rw;
358  tmp2 = rw;
359  assert (m_pf);
360  anErr = QTNewGWorldFromPtr( &gWorld, m_pf, &m_qt_movie_rect, m_ctb, NULL, 0, (char *) (tmp.rowPointer(0)), tmp.rowUpdate ());
361 
362  if (noErr != anErr)
363  {
364  goto Closure; // Should throw an exception ????
365  }
366 
367  if ( reduceGrayTo8bit && m_isGray && depth > rcPixel8 )
368  {
369  depth = rcPixel8;
370  rcSharedFrameBufPtr buf8(new rcFrame (m_width, m_height, depth));
371  buf8->setIsGray(m_isGray);
372  // Produce color map
373  rfFillColorMap( m_ctb, buf8 );
374  rcWindow image8(buf8);
375  // Reduce image depth
376  rfRcWindow32to8( tmp, image8 );
377  tmp2 = image8;
378  } else {
379  // Produce color map if necessary
380  if ( depth < rcPixel32 && !m_isWhiteReversed) {
381  rfFillColorMap( m_ctb, buf );
382  }
383  }
384  }
385  catch (...)
386  {
387  goto Closure;
388  }
389 
390  //? Draw the next first video sample (frame) to GWorld number.
391  SetMovieTimeValue(mMovie, fromTimePoint);
392  SetMovieGWorld(mMovie, gWorld, GetGWorldDevice(gWorld));
393 
394  // Note that the work in rebuilding the drawing pipeline is not actually done in SetMovieGWorld,
395  // but during the next call to MoviesTask.
396  // invalidates the movie's display state ==>
397  // Movie Toolbox redraws the movie the next time MoviesTask is called
398  UpdateMovie (mMovie);
399  anErr = GetMoviesError ();
400  if (anErr != noErr)
401  {
402  fprintf (stderr, "Couldn't UpdateMovie() \n");
403  return 0;
404  }
405 
406  // (= redraw immediately)
407  MoviesTask (mMovie, 0L);
408  anErr = GetMoviesError ();
409  if (anErr != noErr)
410  {
411  fprintf (stderr, "MoviesTask() didn't succeed \n");
412  return 0;
413  }
414 
415  pixMap = GetGWorldPixMap(gWorld);
416  LockPixels(pixMap);
417  UnlockPixels(pixMap);
418 
419  if ( m_isWhiteReversed && tmp2.depth() == rcPixel8 ) {
420  // Color map indicates that pixel value 0 is white
421  // Reverse pixels and use default color map
422  rfReversePixels8( tmp2 );
423  }
424 
425  // Copy it to the vector
426  image = tmp2;
427 
428  //? Closure. Clean up if we have handles.
429 
430 Closure:
431  if(gWorld != NULL)
432  DisposeGWorld(gWorld);
433  return anErr;
434 }
