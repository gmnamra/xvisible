181  * integrals.
182  */
183  if (_type == eMoment1D) {

185  int32 validHeight = VALID_1D_MOM_INT_HEIGHT(frame.height());
186

226 void rcMomentGenerator::genSumSumSq1D(const rcRect& loc, float& sum,
227  float& sumSq) const
228 {
229  rmAssert(_imgSqBase);
230 
231  const int32 x0 = loc.x()*2;
232  const int32 y0 = loc.y();
233  const int32 x1 = x0 + (loc.width() - 1)*2;
234  const int32 y1 = y0 + loc.height();
235  rmAssert(0 <= x0);
236  rmAssert(x0 <= x1);
237  rmAssert(x1 <= _imgSq.width());
238  rmAssert(0 <= y0);
239  rmAssert(y0 < y1);
240  rmAssert(y1 < (_imgSq.height()-1)); // Subtract 1 to discount guard row
241 
242  uint32* iBottom = imgSumPtr(x0, y1);
243  const uint32* iBLast = iBottom + (x1 - x0);
244  uint32* iTop = imgSumPtr(x0, y0);
245  const uint32* iTLast = iTop + (x1 - x0);
246 
247  int64 iSum = 0;
248  int64 iSumSq = 0;
249  while (iBottom <= iBLast) {
250  iSum += *iBottom++ - *iTop++;
251  iSumSq += *iBottom++ - *iTop++;
252  }
253  sum = iSum;
254  sumSq = iSumSq;
255  rmAssert(iBottom == (iBLast + 2));
256  rmAssert(iTop == (iTLast + 2));
257 }

738 void rcAutoCorrelation::genPoint(const rcRect& srchSpace,
739  vector<vector<rcCorr> >& res, const int32 acRadius, resultSpace space ) const
740 {
741  rmAssert(acRadius > 0);
742  const int32 diameter = acRadius*2 + 1;
743  int32 xOrigin = srchSpace.x();
744  int32 yOrigin = srchSpace.y();
745 
746  /* For a diameter by diameter correlation space, target model
747  * dimensions are the search space dimensions reduced by
748  * (diameter-1) in both width and height.
749  */
750  rmAssert(srchSpace.width() > (diameter-1));
751  rmAssert(srchSpace.height() > (diameter-1));
752  const int32 width = srchSpace.width() - (diameter-1);
753  const int32 height = srchSpace.height() - (diameter-1);
754 
755  int32 xBound = xOrigin + width + (diameter-2);
756  int32 yBound = yOrigin + height + (diameter-2);
757  rmAssert(xOrigin < xBound);
758  rmAssert(yOrigin < yBound);
759 
760  const bool useIntegrals = false;
761  rmAssert(_updateFrame.frameBuf());
762  rmAssert(xBound <= _updateFrame.width());
763  rmAssert(yBound <= _updateFrame.height());
764 
765 
766  /* Both verify that the 2D result vector is large enough and
767  * initialize result entries all to -1.0. The latter allows for a
768  * simple mechanism to determine whether or not a correlation score
769  * needs to be calculated.
770  */
771  rmAssert((int32)(res.size()) >= diameter);
772  for (int32 i = 0; i < diameter; i++)
773  {
774  rmAssert((int32)(res[i].size()) >= diameter);
775  for (int32 j = 0; j < diameter; j++)
776  res[i][j].r (-1.0);
777  }
778 
779  rsCorrParams cp;
780  rcCorr cr;
781  rcWindow model(_updateFrame, xOrigin + acRadius, yOrigin + acRadius,
782  width, height);
783 
784  rfCorrelate(model, model, cp, res[acRadius][acRadius]);
785 
786  if (space == eFull)
787  {
788  for (int32 y = 0; y < diameter; y++)
789  for (int32 x = 0; x < diameter; x++)
790  if (res[y][x].r() == -1.0)
791  {
792  rcWindow image(_updateFrame, xOrigin + x, yOrigin + y, width, height);
793  rfCorrelate(image, model, cp, res[y][x]);
794  }
795  }
796  else
797  {
798  for (int32 i = 0; i < diameter; i++) {
799  if (res[acRadius][i].r() == -1.0) {
800  rcWindow image(_updateFrame, xOrigin + i, yOrigin + acRadius,
801  width, height);
802  rfCorrelate(image, model, cp, res[acRadius][i]);
803  }
804  if (res[i][acRadius].r() == -1.0) {
805  rcWindow image(_updateFrame, xOrigin + acRadius, yOrigin + i, width, height);
806  rfCorrelate(image, model, cp, res[i][acRadius]);
807  }
808  }
809  }

814 void rcAutoCorrelation::gen3by3Point (const rcRect& srchSpace, rsCorr3by3Point& res, resultSpace space ) const
815 {
816  /* For 3x3 correlation space, target dimensions are always (-2,-2)
817  * of the search space dimensions.
818  */
819  vector < vector < rcCorr > > cres;
820  cres.resize (3);
821  for (int i = 0; i < 3; i++)
822  cres[i].resize (3);
823 
824  genPoint (srchSpace, cres, 1, space);
825 
826  for (uint32 yM = 0; yM < 3; yM++)
827  for (uint32 xM = 0; xM < 3; xM++)
828  {
829  if ((xM & yM) == 1)
830  continue;
831  else if (_crossOnly && (((xM+yM) & 1) == 0))
832  res.score[yM][xM] = 1.;
833  else
834  {
835  res.score[yM][xM] = cres[yM][xM].r();
836  }
837  }

875  rmAssert(_baseP);
876  free(_baseP);

880  4*sizeof(float) + // Space for storing pixel count
881  (width+2)*sizeof(float)*3 + // Space for image sums
882  (width+2)*sizeof(float)*3 + // Space for image "var X sum squares"
883  (width-1)*sizeof(float) + // Space for w & e correlation results
884  (width-1)*sizeof(float)*2 + // Space for n & s correlation results
885  16*10; // Allow for rounding to 0 mod 16 address
886 
887  if (_radius == 2) {
888  allocCnt +=
889  (width-1)*sizeof(float) + // Space for fw & fe correlation results
890  (width-1)*sizeof(float)*3 + // Space for fn & fs correlation results
891  16*4; // Allow for rounding to 0 mod 16 address
892  }
893 
894  if (!_crossOnly) {
895  allocCnt +=
896  (width-1)*sizeof(float)*2 + // Space for nw & se correlation results
897  (width-1)*sizeof(float)*2 + // Space for ne & sw correlation results
898  16*4; // Allow for rounding to 0 mod 16 address
899 
900  if (_radius == 2) {
901  allocCnt +=
902  (width-1)*sizeof(float)*2 + // Space for wnw & ese correlation results
903  (width-1)*sizeof(float)*2 + // Space for wsw & ene correlation results
904  (width-1)*sizeof(float)*3 + // Space for fnw & fse correlation results
905  (width-1)*sizeof(float)*3 + // Space for fne & fsw correlation results
906  (width-1)*sizeof(float)*3 + // Space for nnw & sse correlation results
907  (width-1)*sizeof(float)*3 + // Space for ssw & nne correlation results
908  16*16; // Allow for rounding to 0 mod 16 address
