483  public:
484  rcMotionCenter () : mSumUU (0), mSumVV (0), mSumUV (0), mSumUR (0), mSumVR (0), mValid (false) {}
485 
486  void add (const rc2Fvector& pos, const rc2Fvector& motion)
487  {
488  mValid = false;
489  double u = motion.x();
490  double v = motion.y();
491  double r = pos.x() * u + pos.y() * v;
492  mSumUU += u * u;
493  mSumVV += v * v;
494  mSumUV += u * v;
495  mSumUR += u * r;
496  mSumVR += v * r;
497  }
498 
499  bool center (rc2Fvector& center)
500  {
501  double d = mSumUU * mSumVV - mSumUV * mSumUV;
502  mValid = d > 0;
503  if (mValid == true)
504  {
505  center.x((mSumUR * mSumVV - mSumUV * mSumVR)/d);
506  center.y((mSumVR * mSumUU - mSumUV * mSumUR)/d);
507  }
508  return mValid;
509  }
510 
511  private:
512  double mSumUU;
513  double mSumVV;
514  double mSumUV;
515  double mSumUR;
516  double mSumVR;
517  bool mValid;
